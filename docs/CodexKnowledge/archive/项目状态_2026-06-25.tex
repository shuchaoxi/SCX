\subsubsection{核心命题（已完成证明）}\label{ux6838ux5fc3ux547dux9898ux5df2ux5b8cux6210ux8bc1ux660e}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  \textbf{Proposition 1}: 全局专家排序不足 --- 严格反证 +
  Arrow不可能定理类比
\item
  \textbf{Proposition 2}: 高误差采样在噪声下非最优 --- 下界证明
\item
  \textbf{Proposition 3}: 状态条件权重优于全局权重 --- Gibbs不等式证明
\end{enumerate}

\subsubsection{四类数据}\label{ux56dbux7c7bux6570ux636e}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  \textbf{Valuable} (值得标注) --- 高误差 + 高密度 + 高一致性 + 低冗余 →
  acquire
\item
  \textbf{Redundant} (冗余) --- 低误差 + 高覆盖 → compress
\item
  \textbf{Noisy} (噪声) --- 高误差 + 孤立 + 标签不一致 →
  downweight/discard
\item
  \textbf{Expert-dependent} (依赖专家) --- 仅特定专家可靠 → route to
  expert
\end{enumerate}
