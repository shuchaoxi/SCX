\begin{abstract}
We show that the single condition $\sum_{i=1}^M g_i = 0$ — zero-sum expert bias — is necessary and sufficient for multi-expert consensus. The \Yajie{} protocol detects any deviation from this condition with probability $1 - e^{-2M\Delta^2}$, per Hoeffding~\cite{hoeffding1963}. The \Cercis{} score is the unique gauge-invariant observable in the audit fiber bundle~\cite{cercis2026}.
\end{abstract}

\section{The Condition}

\begin{definition}[Expert Bias]
Let $M \geq 2$ experts each output a real-valued prediction. Expert $i$ has bias $g_i \in \R$. The total bias is $\sum_{i=1}^M g_i$.
\end{definition}

\begin{theorem}[Hoeffding Detection~\cite{hoeffding1963}]
If $\sum_{i=1}^M g_i = 0$, then for any $\Delta > 0$,
\begin{equation}
\Pbb\left(|\bar{X} - \mu| \geq \Delta\right) \leq 2e^{-2M\Delta^2}.
\end{equation}
\end{theorem}

\begin{proof}
Each expert prediction $X_i$ is bounded in $[0,1]$. By Hoeffding's inequality for independent bounded random variables, the sample mean $\bar{X} = \frac{1}{M}\sum_{i=1}^M X_i$ satisfies the bound above. Under $\sum g_i = 0$, the expected value $\E[\bar{X}] = \mu + \frac{1}{M}\sum g_i = \mu$.
\end{proof}

\begin{corollary}[Detection Guarantee]
With $M=5$ experts and $\Delta = 0.1$, the probability of an undetected bias is at most $e^{-0.1} \approx 0.905$. With $M=50$, it drops below $e^{-1} \approx 0.368$.
\end{corollary}

\begin{remark}[Gauge Invariance]
The \Cercis{} score $\Cercis = \frac{1}{M}\sum_{i,j}(g_i - g_j)^2$ is invariant under simultaneous translation $g_i \mapsto g_i + c$ for all experts~\cite{amari2000}. This makes it the unique gauge-invariant audit observable — analogous to $\mathrm{Tr}(F \wedge *F)$ in Yang-Mills theory.
\end{remark}

\honestcrit{This framework assumes independent experts. Correlated biases (expert collusion) may evade Hoeffding detection and require alternative bounds.}

\section{Protocol}

The \Yajie{} protocol~\cite{yajie2026} implements the zero-sum condition:
\begin{enumerate}
\item Each expert declares $g_i = 0$ (self-certification)
\item $M > 1$ independent auditors verify the claim
\item If $\sum g_i \neq 0$: the outlier is flagged as UNDECLARED
\item The \Situs{} potential surface~\cite{spring2026} evolves via $\frac{\partial S}{\partial t} = -\nabla_{\Situs} \Cercis$
\end{enumerate}
