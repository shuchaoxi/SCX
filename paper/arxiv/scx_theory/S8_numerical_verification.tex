\section{Numerical Verification of Theoretical Constants}

This section provides numerical verification of the exact constant minimax optimality result (Theorem~4' in Supplementary Information S4). All computations are performed using the Python script \texttt{numerical\_verify.py} with standard double-precision arithmetic.

\subsection{Verification Protocol}

For a given parameter set $(p_0, p_1, \eta)$, we compute the following theoretical quantities:

\begin{enumerate}
    \item \textbf{Chernoff point} $\theta^*$: the unique solution to $\mathrm{KL}(\theta\|p_0) = \mathrm{KL}(\theta\|p_1)$,
    \item \textbf{Chernoff information} $\kappa = \mathrm{KL}(\theta^*\|p_0) = \mathrm{KL}(\theta^*\|p_1)$,
    \item \textbf{Saddlepoints} $\lambda_0^* > 0$, $\lambda_1^* < 0$ defined by the log-odds at $\theta^*$,
    \item \textbf{Total log-odds} $D^* = \lambda_0^* + |\lambda_1^*|$ and exponent fraction $s = |\lambda_1^*|/D^*$,
    \item \textbf{Canonical minimax constant} $C_{\min}$ (Lemma~E form):
        \begin{equation}
        C_{\min} = \frac{\eta}{2} \left(\frac{1-\eta}{\eta}\right)^{s}
        \frac{1/\lambda_0^* + 1/|\lambda_1^*|}{\sqrt{\theta^*(1-\theta^*)}},
        \end{equation}
    \item \textbf{Non-adaptive limit} $K_{\mathrm{na}}$ (Lemma~B, using $\theta^*$ without $\eta$ adaptation):
        \begin{equation}
        K_{\mathrm{na}} = \left(\frac{1}{2|\lambda_1^*|} + \frac{1-\eta}{2\eta\lambda_0^*}\right) \frac{1}{\sqrt{\theta^*(1-\theta^*)}},
        \end{equation}
    \item \textbf{Adaptive limit} $K_{\mathrm{ad}}$ (Lemma~D, using $\eta$-aware threshold $\theta^\dagger$):
        \begin{equation}
        K_{\mathrm{ad}} = \left(\frac{1-\eta}{\eta}\right)^{s} \frac{1/\lambda_0^* + 1/|\lambda_1^*|}{2\sqrt{\theta^*(1-\theta^*)}}.
        \end{equation}
\end{enumerate}

The verification criterion is the machine-precision equality $K_{\mathrm{ad}} = C_{\min}/\eta$, which confirms that the adaptive threshold achieves the minimax lower bound exactly (up to the asymptotic limit).

\subsection{Finite-$M$ Convergence Analysis}

The Bahadur--Rao asymptotic expansion predicts that for finite $M$,
\begin{equation}
e^{M\kappa} \sqrt{2\pi M} \bigl(1 - \mathrm{F1}(M)\bigr) \longrightarrow \frac{C_{\min}}{\eta} \quad \text{as } M\to\infty.
\end{equation}

Table~\ref{tab:finiteM} shows the finite-$M$ convergence for Case~2 (the only case where $M=500$ produces non-zero $1-\mathrm{F1}$ within double precision; for the other cases, the error probabilities are below machine epsilon at $M=500$).

\paragraph{Key findings.}
\begin{enumerate}
    \item \textbf{Machine-precision equality} ($K_{\mathrm{ad}} = C_{\min}/\eta$) holds across all five cases, confirming the algebraic equivalence of the adaptive achievability limit and the minimax lower bound.
    \item \textbf{Chernoff information is always slower than the Hoeffding bound}: $2\Delta^2 / \kappa > 1$ in all cases (range $2.46$--$3.41$). This confirms that the Hoeffding bound overestimates the rate of convergence.
    \item \textbf{Non-adaptive penalty is case-dependent}: The naive threshold $\theta^*$ (ignoring $\eta$) is suboptimal by factors $1.00$--$2.41$. The penalty is largest when $\eta$ is far from $1/2$ (Cases~1,~3,~5) and disappears at $\eta = 1/2$ (Case~4).
    \item \textbf{All constants satisfy theoretical consistency checks}: $\mathrm{KL}(\theta^*\|p_0) = \mathrm{KL}(\theta^*\|p_1)$ holds to $\pm 10^{-16}$; $C_{\min} > 0$ in all cases; $K_{\mathrm{ad}} > 0$; the exponent fraction $s \in (0,1)$.
\end{enumerate}
