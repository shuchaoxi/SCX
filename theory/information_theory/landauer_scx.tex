\section{经典Landauer原理：严格回顾}
\label{sec:landauer_classic}
% ===========================================================================

\begin{definition}[Landauer原理，1961]
\label{def:landauer}
在温度$T$的热库中，\textbf{擦除}1比特的经典信息\textbf{必然}导致至少
\begin{equation}\label{eq:landauer_classic}
\Delta Q \geq \kB T \ln 2
\end{equation}
的热量耗散到环境中（等价于熵增$\Delta S \geq \kB \ln 2$）。
\end{definition}

\begin{remark}[Landauer原理的物理地位]
\strictness{严格：实验验证[Lutz+ 2012, Bérut+ 2012]，理论地位=热力学第二定律的直接推论}
Landauer原理是\textbf{物理定理}，不是工程约束。它来自Liouville定理+热力学第二定律：
信息擦除本质上是将相空间体积压缩再映射的过程，在Hamilton动力学下必须向外排放熵。
\end{remark}

\begin{theorem}[Landauer下界——信息处理的物理成本]
\label{thm:landauer_general}
对温度$T$下擦除$I$比特的经典信息，所需最小能量为：
\begin{equation}\label{eq:landauer_general}
E_{\text{erase}} \geq \kB T \cdot I \cdot \ln 2.
\end{equation}
等价地，\textbf{不擦除信息}（仅复制或转移）的物理操作可以\textbf{在原理上无能量消耗}。
\end{theorem}

\begin{proof}[直觉证明——相空间容积论证]
考虑存储1比特的双稳态势阱。逻辑0和逻辑1对应相空间中两个不相交的区域$\Gamma_0, \Gamma_1$，
总体积为$2V$。擦除操作将两区域映射到单区域（逻辑0），相空间体积从$2V$压缩到$V$。
由Liouville定理，保守动力学不改变相空间体积——因此压缩必须伴随向环境的体积“排放”。
排放的最小熵增由Boltzmann公式$\Delta S = \kB \ln(V_{\text{before}}/V_{\text{after}}) = \kB \ln 2$给出。
\strictness{严格：在经典统计力学框架内无可争议}
\end{proof}

\begin{honestbox}
\textbf{常见误解：Landauer原理已被“量子Landauer”推翻？}

不。量子信息擦除的Landauer下界\textbf{完全一致}——$\kB T \ln 2$每量子比特。
量子优势在于\textbf{可逆计算}（不擦除中间结果），而非绕开擦除的物理代价。
SCX的CEC设计恰是\textbf{可逆计算精神的制度化}——但使用的是经典信息。
\end{honestbox}

\section{CEC追加式存储 = Landauer最优}
\label{sec:cec_optimal}
% ===========================================================================

\begin{definition}[CEC追加式存储的形式化]
\label{def:cec_append}
SCX的CEC（Chain-Extending Commitment）存储范式规定：
\begin{itemize}
    \item 状态空间$\mathcal{S}$中的转移\textbf{仅允许追加}：$s_{t+1} = s_t \circ a_{t+1}$。
    \item \textbf{禁止操作}：覆盖、删除、原地修改。
    \item 旧状态$s_t$在追加后\textbf{物理上继续存在}（可索引但不可删除）。
\end{itemize}
\end{definition}

\begin{theorem}[CEC = Landauer最优性定理]
\label{thm:cec_landauer}
在CEC追加式存储下，SCX系统\textbf{在任意时间区间内}的Landauer热力学代价为：
\begin{equation}\label{eq:cec_cost}
E_{\text{Landauer}}^{\text{CEC}} = 0.
\end{equation}
\end{theorem}

\begin{proof}[证明——纯定义性]
Landauer代价仅发生于\textbf{信息擦除}时刻。
CEC规定永不可删除——$\mathcal{S}$上的转移不含覆盖/删除操作。
因此擦除事件集合为空集$\varnothing$。
零个擦除事件的代价之和为零。

\strictness{严格：定义性证明——若接受CEC确实永不删除，则Landauer代价为0}
\end{proof}

\begin{remark}[“作弊”还是“最优”？]
CEC的零Landauer代价\textbf{不是物理奇迹}——它是以\textbf{无限增长的存储空间}为代价换取的。
每比特永久保留的信息需要对应的物理存储介质。在有限宇宙中，
CEC最终受Bekenstein界（$I \leq A/(4\ell_P^2 \ln 2)$）和宇宙总熵容量的约束。
\textbf{这是一个工程权衡，不是物理原理的违反。}
\end{remark}

\begin{physicsbox}
\textbf{Landauer最优的三个层次：}
\begin{enumerate}[leftmargin=*]
    \item \textbf{逻辑层：}CEC = 从不擦除 = 从不支付Landauer代价。\quad\strictness{严格}
    \item \textbf{物理层：}实际存储介质（SSD/HDD）的写操作\textbf{实际上}伴随发热——
    但这来自电子器件的焦耳热和阻抗损耗，不是Landauer原理约束的擦除热。
    两者在数量级上差$\sim 10^3$（Landauer $\sim 10^{-21}$J/bit，SSD写 $\sim 10^{-15}$J/bit）。
    \quad\strictness{严格：但需区分物理层和原理层}
    \item \textbf{宇宙学层：}在有限宇宙中永远追加是不可能的。
    但SCX的有意义时间尺度（$\ll$宇宙热寂时间）内，CEC逼近Landauer最优。
    \quad\strictness{启发式}
\end{enumerate}
\end{physicsbox}

\section{省定理(Forgotten Theorem)的热力学解释}
\label{sec:forgotten}
% ===========================================================================

\begin{definition}[省定理——SCX的“遗忘”操作]
\label{def:forgotten}
\textbf{省定理(Forgotten Theorem)}陈述：
在SCX审计中，审计者$\mathcal{A}$可以“省去”（即不检查）
部分历史状态而不影响审计的\textbf{安全性下界}，当且仅当这些被省去的内容
在CEC链上\textbf{被子序列操作完全覆盖}（信息上呈Markov屏蔽）。
\end{definition}

\begin{theorem}[省定理的热力学对应——遗忘的物理成本]
\label{thm:forgotten_thermo}
设审计者选择“遗忘”（即不检查）状态子序列$\{s_{i_1}, \ldots, s_{i_k}\}$。
在热力学意义上：
\begin{enumerate}
    \item 若这些状态被CEC的\textbf{后续追加操作完全覆盖}（即Markov屏蔽成立），
    则遗忘它们的\textbf{审计有效性代价为零}（不引入新误差下界），
    物理上等价于\textbf{不需要额外的信息擦除能量}。

    \item 若Markov屏蔽不成立（即被省去内容影响可审计性），
    则遗忘引入了额外的\textbf{不可检测风险}$\Delta P_{\text{undetected}}$，
    其物理对应为：要“补偿”这一遗忘所需的额外审计能量下界为
    \begin{equation}\label{eq:forgotten_cost}
    E_{\text{compensate}} \geq \kB T \cdot \Delta P_{\text{undetected}}^{-1} \cdot \ln 2.
    \end{equation}
\end{enumerate}
\end{theorem}

\begin{proof}[证明概要——信息-能量等价的应用]
\strictness{概要：Markov屏蔽部分的论证严格，补偿能量部分为启发式下界}
\begin{enumerate}
    \item Markov屏蔽 $\Rightarrow$ 被覆盖状态的信息已\textbf{无冗余价值}。
    在信息论中，$I(X; Y \mid Z) = 0$意味着给定$Z$后$X$不提供关于$Y$的新信息。
    因此“遗忘”$X$不损失审计信息——审计的能量需求不变。

    \item 若屏蔽不成立，$\Delta I = I(\text{被省去}; \text{审计目标} \mid \text{剩余观测})$是非零的。
    这个$\Delta I$是审计者为弥补信息缺口所需\textbf{额外获取}的信息量。
    由Landauer，处理$\Delta I$的信息需要至少$\kB T \Delta I \ln 2$的能量
    （即使不擦除，获取信息的测量步骤也有物理代价——但这是\textbf{量子力学}约束，不等同于Landauer擦除代价）。

    \item \textbf{诚实声明：}式(\ref{eq:forgotten_cost})将不可检测风险$P_{\text{undetected}}$和信息量$\Delta I$
    通过Fano型不等式联系起来（$\Delta I \sim \log(1/P_{\text{undetected}})$），
    然后套用Landauer。这提供了\textbf{数量级估计}而非严格等式。
\end{enumerate}
\end{proof}

\begin{honestbox}
\textbf{省定理的热力学解释——一个危险的类比：}

“遗忘的信息有物理成本”是正确的（Landauer），但将审计的\textbf{逻辑遗忘}
与物理信息擦除\textbf{等同}是一个\textbf{范畴错误}：
\begin{itemize}
    \item 审计者“不检查某段历史”只是在她的\textbf{决策模型中忽略}了该数据——
    数据在CEC上仍然存在，未被物理擦除。
    \item 真正的物理擦除发生在存储介质主动释放空间时。SCX-CEC设计下这不会发生。
    \item 因此省定理的“热力学成本”是\textbf{机会成本}（放弃了利用已有信息的机会），
    而非Landauer的\textbf{物理必然成本}。
\end{itemize}

\textbf{两者之间的类比是有益的启发，但不应混为一谈。}
\end{honestbox}

\section{审计的最小能量下界}
\label{sec:audit_energy}
% ===========================================================================

\begin{theorem}[审计能量下界——Landauer+Brillouin的SCX推广]
\label{thm:audit_energy}
在温度$T$下运行SCX审计，设审计者需从观测$Y$中提取关于目标假设的
\textbf{净信息量}$\Delta I$（以比特计）。则审计操作的\textbf{理论最小能量}为：
\begin{equation}\label{eq:audit_energy_lb}
E_{\text{audit}} \geq \kB T \cdot \Delta I \cdot \ln 2.
\end{equation}
\end{theorem}

\begin{proof}[推导——三个物理原理的组合]
\strictness{概要：组合了Landauer, Brillouin, Szilard。$\Delta I$的精确定义是关键难点}
\begin{enumerate}
    \item \textbf{Szilard引擎论证(Szilard, 1929)：}获取1比特信息的最小代价是$\kB T \ln 2$的热量。
    这是Landauer原理的“逆”——获取信息同样有物理代价（在热力学循环中表现为功的提取能力变化）。

    \item \textbf{Brillouin原理(Brillouin, 1956)：}任何测量操作需要至少$\kB T \ln 2$的能量来
    克服热噪声以及产生与非门可辨的信号差。更精确：要在温度$T$下分辨两个状态，
    需要的“负熵”输入至少为$\kB \ln 2$每比特分辨。

    \item \textbf{递推：}审计操作分解为：(a) 读取日志$\to$消耗能量$\sim \kB T \times I(Y)$；
    (b) 推理$\to$信息处理的代价（对不可逆推理步骤，每擦除中间变量需$\kB T \ln 2$）；
    (c) 输出判定$\to$若输出确定了状态并擦除了其他可能性，支付对应Landauer代价。

    净效果为式(\ref{eq:audit_energy_lb})。
\end{enumerate}
\end{proof}

\begin{remark}[$\Delta I$的严格定义问题]
\strictness{核心困难：非严格}
“审计所需净信息量”$\Delta I$的正确定义应为：
\[
\Delta I = \MI(\text{审计判定}; \text{真实状态} \mid \text{先验知识}).
\]
但在SCX框架下，这个互信息量\textbf{依赖于审计策略}——不同的审计算法提取不同的$\Delta I$。
式(\ref{eq:audit_energy_lb})因此是\textbf{下界族}（对每个审计策略有一个对应下界），
而非统一的理论极限。\textbf{最优审计策略}不仅最小化错误概率，也应最小化能量-信息比。
\end{remark}

\begin{honestbox}[title=🩸 全文诚实暴击总结]
\begin{enumerate}[leftmargin=*]
    \item \textbf{CEC = Landauer最优}这个命题在逻辑层是正确的（定义性），
    但在物理层无实际意义——当前所有计算设备的能耗由焦耳热主导，
    Landauer极限在$10^{-21}$J/bit量级，工程能耗在$10^{-15}$J/bit量级，
    \textbf{差了百万倍}。

    \item \textbf{省定理 $\neq$ 物理遗忘。}
    审计者“不检查”不等于“信息被物理擦除”。
    省定理的热力学成本是\textbf{类比}，不是严格恒等式。

    \item \textbf{审计能量下界的核心难点}不在于Landauer原理本身，
    而在于\textbf{“审计净信息量”$\Delta I$的严格操作化定义}。
    没有这个定义，式(\ref{eq:audit_energy_lb})只是量纲分析。

    \item \textbf{SCX-CEC的真正物理优势}在于\textbf{减少工程层面的写放大}——
    不需要DELETE/UPDATE导致的写前日志、垃圾回收、vacuum操作——
    这些工程开销远大于Landauer原理的理论极限。
    这是一个\textbf{工程洞察}，不是物理突破。

    \item \textbf{Bekenstein界警告：}在宇宙学尺度上，CEC的永久追加不可持续。
    对于地球尺度的SCX部署，这不是问题（$\sim 10^{50}$ bits的Bekenstein界远未饱和）。
    但应将此列为SCX的\textbf{理论可扩展性边界条件}（$\S_\text{max} \leq A/(4\ell_P^2 \ln 2)$）。
\end{enumerate}
\end{honestbox}
