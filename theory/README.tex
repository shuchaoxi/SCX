\section{SCX Mathematical Framework Overview}\label{scx-mathematical-framework-overview}

\begin{quote}
State-Conditioned eXpertise Theory --- From reliability problems in multi-expert systems to data value definitions
\end{quote}

\textbf{Last updated}: 2026-06-27

\begin{center}\rule{0.5\linewidth}{0.5pt}\end{center}

\subsection{Problem}\label{problem}

Given M experts \{f\_1, \ldots, f\_M\}, input space X, label space Y.

\textbf{Traditional approach}: give each expert a global accuracy R\_m = E[\ell(f\_m(x), y)].

\textbf{The problem}: In heterogeneous input spaces, expert accuracy varies greatly across different regions. Global ranking cannot reflect this.

\begin{center}\rule{0.5\linewidth}{0.5pt}\end{center}

\subsection{Core Insight}\label{core-insight}

\begin{verbatim}
R_m is insufficient. The correct object is R_m(s).
\end{verbatim}

where s: X $\to$ S is the state mapping, assigning inputs to latent states.

\begin{center}\rule{0.5\linewidth}{0.5pt}\end{center}

\subsection{Three Core Definitions}\label{three-core-definitions}

\subsubsection{1. State-Conditioned Expert Risk}\label{state-conditioned-expert-risk}

\begin{verbatim}
R_m(s) = E_{x~P(·|s)} [ \ell(f_m(x), f*(x)) ]
\end{verbatim}

\subsubsection{2. SCX Reliability}\label{scx-reliability}

\begin{verbatim}
SCX_m(s) = P( \ell(f_m(x), y) < \tau | x \in s )
\end{verbatim}

\subsubsection{3. State Data Value (Deprecated)}\label{state-data-value-deprecated}

\begin{quote}
\textbf{2026-06-27}: V(s) = \={r}(s) $\cdot$ $\rho$(s) $\cdot$ L(s) $\cdot$ [1 - D(s)] $\cdot$ max\_m SCX\_m(s) has been deprecated. Its function has been replaced by more precise component decomposition: redundancy score D(s) handles compression decisions, consensus score \={C}(s) handles noise detection, and state-conditioned risk \^{R}\_m(s) handles expert routing.
\end{quote}

\subsection{Algorithm: State-Certified Active Learning}\label{algorithm-state-certified-active-learning}

\begin{verbatim}
Input: L (labeled), U (unlabeled), E (experts), \phi (representation map)

1. Construct states S in \phi(X)
2. Estimate \={C}(s), \rho(s), D(s)
3. Estimate \^{R}_m(s) for all experts m and states s
4. Classify each state: valuable / redundant / noisy / expert-dependent
5. Select action for each state:
   a*(s) = arg max_a U(a, s)
   where a \in {acquire, relabel, downweight, discard, route}
6. Execute actions, update model and state map
7. Repeat
\end{verbatim}

\begin{quote}
In Step 2, D(s) computation no longer requires ground truth labels (using Theorem 1's \={C}(s) to replace \={r}(s)). \={C}(s) comes directly from multi-expert consensus, enabling the algorithm to run in a completely unsupervised setting.
\end{quote}
