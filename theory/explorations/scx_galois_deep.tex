\begin{abstract}
\small 我们证明Galois理论与SCX（可审计声称）框架之间的数学同构。核心洞见：Galois对应（中间域$\leftrightarrow$子群）与SCX对应（可审计子声称$\leftrightarrow$审计子群）是同一不变理论结构的两个实例——前者处理根排列下系数的对称性，后者处理专家排列下CEC的对称性。由此导出：不可解五次方程的$A_5$障碍$\leftrightarrow$不可审计声称的审计群不可分解性；正规扩张的合成列可解性$\leftrightarrow$独立专家共识的审计子群格可分性。最终结论：SCX审计就是``声称的Galois理论''——声称可验证$\iff$审计群可解。
\end{abstract}

\section{不变理论：共同的数学根源}
% ===========================================================================

设$L/K$为有限Galois扩张，$G = \Gal(L/K)$。多项式系数$a_i$是根的初等对称函数——根排列下的不变量。这就是不变理论的原型：\textbf{系数是根空间上$G$作用的基本不变量}。

\begin{definition}[SCX审计结构]
设$\cC$为声称，$\E=\{e_1,\ldots,e_n\}$为有限专家集。令$\A=\Audit(\cC)$为$\cC$的审计群——所有保持CEC不变的专家置换构成的群。CEC是映射$f: \E \times \cC \to \{\text{真},\text{伪},\bot\}$，满足$\forall\sigma\in\A,\; f(\sigma(e),\cC)=f(e,\cC)$。令$\Fix(\A)=\{f\mid\forall\sigma\in\A,\,f\circ\sigma=f\}$为$\A$-不变量空间。
\end{definition}

\begin{theorem}[不变理论对偶]\label{thm:invariant}
下述结构范畴等价：
\begin{enumerate}[label=(\roman*)]
  \item \textbf{Galois侧}：$\Q[x_1,\ldots,x_n]$上$G\subseteq S_n$作用，不变量环$\Q[x_1,\ldots,x_n]^G$由初等对称多项式生成——系数是根排列的$G$-不变量。
  \item \textbf{SCX侧}：$\{0,1,\bot\}^{\E}$上$\A$作用，CEC空间$\Fix(\A)$由一致同意算子生成——CEC是专家排列的$\A$-不变量。
\end{enumerate}
核心原理统一：\textbf{群作用的不变量刻画可观测/可验证的内容}。系数``在$K$中''（不进入扩域即可知）；CEC``在公共知识基线中''（不依赖特定专家即可验证）。
\end{theorem}

\section{Galois-审计对应：格反同构}
% ===========================================================================

\begin{theorem}[SCX基本定理]\label{thm:fundamental}
设$\cC$为声称，$\A=\Audit(\cC)$。令$\sS(\cC)$为可审计子声称的偏序集（按蕴含），$\sS(\A)$为审计子群的偏序集（按包含）。则存在格反同构：
\begin{equation}
\Phi: \sS(\cC) \;\longleftrightarrow\; \sS(\A)^{\mathrm{op}}
\end{equation}
满足：(a)~$\cC_1\subseteq\cC_2 \iff \Audit(\cC_2)\subseteq\Audit(\cC_1)$；(b)~$\abs{\Audit(\cC_1)}\cdot[\cC:\cC_1]=\abs{\Audit(\cC)}$，审计维度对应域扩张次数；(c)~$\cC$审计正规$\iff$不可分解子声称的审计群是$\A$的正规子群。
\end{theorem}

\begin{proof}[证明概要]
$\Phi(\cC_1)=\{\sigma\in\A\mid\sigma\text{保持}\cC_1\text{CEC不变}\}$；$\Phi^{-1}(H)=\bigwedge\{f^{-1}(\text{真})\mid f\in\Fix(H)\}$为$H$-不变量可断定的最强子声称。轨道-稳定子定理给出乘积公式。
\end{proof}

\begin{remark}[诚实性声明]
域扩张次数$[L:K]$是向量空间维数，而SCX审计维度目前是组合量（所需独立专家数）。格反同构是精确的，但度量等距尚未建立。若声称空间可赋予``审计模''结构，可能得到完全等距同构。此为公开猜想。
\end{remark}

\section{不可解性：从$A_5$到不可审计性}
% ===========================================================================

五次以上一般方程的不可解性是Galois理论的标志：$A_5$是单群，$S_5\supset A_5\supset\{1\}$含非交换单商群$A_5/\{1\}\cong A_5$，故$S_5$不可解。

\begin{definition}[审计可解性]
$\A$可解若存在次正规列$\{1\}=\A_0\triangleleft\cdots\triangleleft\A_k=\A$使各商群$\A_{i+1}/\A_i$交换。声称$\cC$\textbf{可审计}若$\Audit(\cC)$可解。
\end{definition}

\begin{theorem}[不可审计声称的存在性]\label{thm:unauditable}
若$\Audit(\cC)\cong A_5$，则$\cC$不可审计——审计结构不可分解为一串独立的、可逐一验证的子步骤。
\end{theorem}

\begin{proof}
$A_5$仅有正规子群$\{1\}$和$A_5$。由\Cref{thm:fundamental}的格反同构，$\cC$无非平凡审计正规子声称——任何分解尝试要求非平凡正规审计子群，而$A_5$不提供。故$\cC$须作为不可分割整体被评估。
\end{proof}

\begin{remark}[诚实暴击]
这不是说$A_5$-审计声称物理上不可评估——而是\textbf{不存在将其审计分解为独立、可逐一执行的交换步骤的普适程序}。正如Galois理论中``不可根式求解''不是说方程无解，而是解不能通过$+,-,\times,\div,\sqrt[n]{\cdot}$的有限组合表达。不可审计声称可能被天才一次性评估，但审计过程不可标准化分解。
\end{remark}

\section{正规扩张与审计的合成列}
% ===========================================================================

正规扩张$L/K$的可解性$\iff\Gal(L/K)$合成列商群为素数阶循环群。SCX对应：

\begin{theorem}[审计可分解性]\label{thm:decomposability}
设$\cC$有可解审计群$\A$，则存在声称塔$\cC_0\subset\cdots\subset\cC_k=\cC$，每步$\cC_{i+1}/\cC_i$审计``独立可加''——商群$\Audit(\cC_{i+1})/\Audit(\cC_i)$交换，该层判断可与所有前层交换（无时序依赖）。
\end{theorem}

\begin{proof}
由$\A$可解得合成列，经$\Phi^{-1}$映射为声称塔。商群交换$\iff$该层审计操作交换$\iff$审计结果不依赖子任务执行顺序。对应Galois理论中逐次素数阶循环扩张塔。
\end{proof}

\begin{corollary}[共识可分性]\label{cor:separability}
$N$个独立专家评估$\cC$，共识可分解为独立子共识$\iff$合成列长度$\ge\log_2 N$且商群为素数阶循环群。最佳情形（完全可并行化）：子群格为布尔格；最坏情形（$A_5$-型纠缠）：格退化为2-元素链。
\end{corollary}

\section{SCX即声称的Galois理论}
% ===========================================================================

\begin{theorem}[SCX-Galois对偶原理]\label{thm:meta}
以下判定问题范畴等价：(a)~根式可解性——给定$f(x)\in\Q[x]$，判定$\Gal(f)$是否可解；(b)~审计可验证性——给定$\cC$及CEC约束，判定$\Audit(\cC)$是否可解。二者计算瓶颈相同：导出列终止性检验，复杂度$O(|\G|^2\log|\G|)$。
\end{theorem}

\begin{remark}[最终的诚实性声明]
上述对应是\textbf{函子性的}，非松散类比。存在忠实函子$F: \mathbf{FinGal} \to \mathbf{SCXAudit}$：$L/K\mapsto$``$L/K$中的元素关系''作为声称，$\Gal(L/K)\mapsto\Audit$。此函子保持格结构、正规性和可解性。但$F$非满——存在非域扩张来源的SCX结构（如概率判断声称），故SCX严格更一般。寻找SCX中对应于Galois上同调、Kummer理论、Artin-Schreier理论的结构是活跃方向。
\end{remark}

\textbf{结论。} SCX框架\textbf{就是}声称空间的Galois理论。判断声称是否可验证$\iff$判断审计群是否可解。$A_5$的幽灵同时盘旋于五次方程与复杂声称之上：某些事物本质不可分解。非工程困难——乃数学定理。
