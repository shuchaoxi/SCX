\subsubsection{1.2 Core Quantities}\label{core-quantities}

\textbf{State-Conditioned Expert Risk} (Definition 1, SCX):

\[R_m(s) = \mathbb{E}_{x \sim P(\cdot \mid s)}[\ell(f_m(x), f^*(x))]\]

\textbf{SCX Reliability} (Definition 2, SCX):

\[\text{SCX}_m(s) = P(\ell(f_m(x), f^*(x)) < \tau \mid x \in s)\]

\textbf{Expert Pairwise Conflict} (Definition 3, ExpertConflict):

\[\text{Conflict}(m, n, s) = \mathbb{E}_{x \sim P(\cdot \mid s)}[|f_m(x) - f_n(x)|^2]\]

\subsubsection{1.3 Protocol Overall Structure}\label{protocol-overall-structure}

The Governance Protocol is a \textbf{linear decision flow} from the raw expert system to the distillation authorization decision:

\[\mathcal{G}: \underbrace{(\{f_m\}, \mathcal{S}, \mathcal{D}_{\text{ref}})}_{\text{Input}} \xrightarrow{\text{Step 1-5}} \underbrace{\{\text{authorized}, \text{conditional}, \text{denied}\}}_{\text{Output}}\]

Input: - Expert set \(\{f_m\}_{m=1}^M\) (possibly heterogeneous: MACE, NEP, SevenNet, \ldots) - State space \(\mathcal{S}\) and state mapping \(s(x)\) - Reference dataset \(\mathcal{D}_{\text{ref}} = \{(x_i, y_i)\}_{i=1}^{N_{\text{ref}}}\) (DFT labels) - Distillation target state \(s_{\text{target}} \in \mathcal{S}\)

Output: - \textbf{authorized}: all checks passed, can distill directly - \textbf{conditional}: some conditions not met but remediable (e.g., supplement anchor tests) - \textbf{denied}: cannot distill, need expert retraining or expert replacement

\subsection{2. Five-Step Governance Protocol}\label{five-step-governance-protocol}

\subsubsection{Step 1: Gauge Check}\label{step-1-gauge-check}

\paragraph{Problem}\label{problem-step1}

Different MLIPs' total energy predictions exhibit systematic energy offsets (gauge freedom). Step 1 detects and eliminates these offsets so that all experts' energy predictions are comparable on a unified reference scale.

\paragraph{Formal Definition}\label{formal-definition-step1}

\textbf{Input}: - Expert set \(\{f_m\}_{m=1}^M\) - Reference state subset \(\mathcal{S}_{\text{gauge}} \subseteq \mathcal{S}\) (select simple states with known energies, e.g., bulk lattices) - Reference energy \(E_{\text{ref}}(s)\) for each \(s \in \mathcal{S}_{\text{gauge}}\) (obtained from DFT or experiment) - Tolerance \(\epsilon_{\text{gauge}} > 0\)

\textbf{Gauge Definition}:

Each expert \(f_m\)'s energy prediction decomposes as:

\[f_m(x) = \underbrace{f_m^{\text{intrinsic}}(x)}_{\text{Physical part}} + \underbrace{\mu_m}_{\text{Gauge offset}} + \underbrace{\eta_m(x)}_{\text{Residual}}\]

where \(\mu_m\) is a systematic bias dependent on the expert (gauge shift). For force predictions, \(\nabla_x \mu_m = 0\), so forces are unaffected. However, differences in \textbf{total energy baselines} affect the student's energy learning during distillation.

\textbf{Gauge Check Algorithm}:

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\item
  For each \(s \in \mathcal{S}_{\text{gauge}}\), sample \(N_{\text{gauge}}\)
  configurations \(\{x_{s,j}\}_{j=1}^{N_{\text{gauge}}}\)
\item
  Compute each expert's average deviation:

  \[\hat{\mu}_m(s) = \frac{1}{N_{\text{gauge}}} \sum_{j=1}^{N_{\text{gauge}}} \left[f_m(x_{s,j}) - E_{\text{ref}}(s)\right]\]
\item
  Compute gauge consistency metric:

  \[\text{GaugeVariance}(m) = \frac{1}{|\mathcal{S}_{\text{gauge}}|} \sum_{s \in \mathcal{S}_{\text{gauge}}} \left(\hat{\mu}_m(s) - \bar{\mu}_m\right)^2\]

  where \(\bar{\mu}_m = \frac{1}{|\mathcal{S}_{\text{gauge}}|} \sum_{s} \hat{\mu}_m(s)\)
\item
  Decision condition:

  \[\text{gauge\_certified}(m) = \begin{cases}
  \text{true}, & \text{GaugeVariance}(m) \le \epsilon_{\text{gauge}} \\
  \text{false}, & \text{otherwise}
  \end{cases}\]

  (Low variance means the offset \(\mu_m\) is consistent across reference states, i.e., a pure gauge offset)
\end{enumerate}

\textbf{Output type}:

\[\text{gauge\_certified}: [M] \to \{\text{true}, \text{false}\}\]

\textbf{Gauge Fixing (execute on failure)}:

If there exists \(m\) such that \(\text{gauge\_certified}(m) = \text{false}\), gauge fixing is needed. Define calibration function:

\[\tilde{f}_m(x) = f_m(x) - \frac{1}{|\mathcal{S}_{\text{gauge}}|} \sum_{s \in \mathcal{S}_{\text{gauge}}} \hat{\mu}_m(s)\]

i.e., subtract expert \(m\)'s average systematic deviation from its energy predictions over the reference states.

Re-run Check after calibration.

\paragraph{Relationship to Existing Work}\label{relationship-to-existing-work}

\begin{itemize}
\tightlist
\item
  The literature's ``Gauge dependence of total energies'' (arXiv:2504.05565, 2025)
  points out similar issues across different DFT functionals but does not address \textbf{cross-MLIP-architecture} gauge alignment.
\item
  This step is an original contribution of SCX/EGP — enforcing unified multi-expert energy scaling before distillation.
\end{itemize}

\begin{center}\rule{0.5\linewidth}{0.5pt}\end{center}

\subsubsection{Step 2: Domain Certificate}\label{step-2-domain-certificate}

\paragraph{Problem}\label{problem-step2}

For the target state \(s_{\text{target}}\), evaluate each expert's reliability on that state. Unreliable experts should not participate in distillation.

\paragraph{Formal Definition}\label{formal-definition-step2}

\textbf{Input}: - Expert set \(\{f_m\}_{m=1}^M\) - Target state \(s_{\text{target}} \in \mathcal{S}\) - State-labeled data \(\mathcal{D}_{s} = \{(x_i, y_i) \mid s(x_i) = s_{\text{target}}\}\) (may or may not exist) - Low reliability threshold \(\theta_{\text{low}} \in (0, 1)\) - High reliability threshold \(\theta_{\text{high}} \in (0, 1)\) (\(\theta_{\text{high}} > \theta_{\text{low}}\)) - Loss threshold \(\tau > 0\) - Minimum sample count \(n_{\min} \in \mathbb{N}\)

\textbf{Reliability Estimation}:

For each expert \(m\), estimate SCX reliability on \(s_{\text{target}}\):

\[\widehat{\text{SCX}}_m(s_{\text{target}}) = \begin{cases}
\displaystyle \frac{1}{|\mathcal{D}_{s}|} \sum_{(x, y) \in \mathcal{D}_{s}} \mathbf{1}\{\ell(f_m(x), y) < \tau\}, & |\mathcal{D}_{s}| \ge n_{\min} \\[1.5em]
\text{Bayesian shrinkage estimate}, & |\mathcal{D}_{s}| < n_{\min}
\end{cases}\]

For small samples, use James-Stein shrinkage estimation (corresponding to \texttt{ExpertReliability.estimate\_small\_sample}):

\[\widehat{\text{SCX}}_m^{\text{JS}}(s) = \lambda \cdot \widehat{\text{SCX}}_m(s) + (1 - \lambda) \cdot \overline{\text{SCX}}_m\]

where \(\lambda = \frac{|\mathcal{D}_{s}|}{|\mathcal{D}_{s}| + n_{\min}}\), \(\overline{\text{SCX}}_m\) is the global average reliability of expert \(m\) across all states.

\textbf{Decision Logic}:

\[\text{domain\_certified}(m, s) = \begin{cases}
\text{true}, & \widehat{\text{SCX}}_m(s) \ge \theta_{\text{high}} \\[0.5em]
\text{conditional}, & \theta_{\text{low}} \le \widehat{\text{SCX}}_m(s) < \theta_{\text{high}} \\[0.5em]
\text{false}, & \widehat{\text{SCX}}_m(s) < \theta_{\text{low}}
\end{cases}\]

\begin{itemize}
\tightlist
\item
  \textbf{true}: Expert is highly reliable on target state and can participate in distillation
\item
  \textbf{conditional}: Reliability insufficient for independent contribution but can participate in multi-expert ensemble
\item
  \textbf{false}: Expert is not trustworthy on this state, excluded from distillation
\end{itemize}

\textbf{Output type}:

\[\text{domain\_certified}: [M] \times \mathcal{S} \to \{\text{true}, \text{false}, \text{conditional}\}\]

\paragraph{Correspondence to SCX Code}\label{correspondence-to-scx-code-step2}

Step 2 directly calls \texttt{ExpertReliability.estimate()} to obtain \texttt{SCX\_matrix}, where \texttt{SCX\_matrix{[}m,\ k{]}} is \(\widehat{\text{SCX}}_m(s_k)\).

\begin{center}\rule{0.5\linewidth}{0.5pt}\end{center}

\subsubsection{Step 3: Conflict Resolution}\label{step-3-conflict-resolution}

\paragraph{Problem}\label{problem-step3}

For the target state \(s_{\text{target}}\), check whether there are significant prediction conflicts among experts certified by Step 2. Conflict regions require arbitration.

\paragraph{Formal Definition}\label{formal-definition-step3}

\textbf{Input}: - Certified expert subset \(\mathcal{E}_{\text{cert}} \subseteq \{f_1, \dots, f_M\}\) (Step 2 output with \(\text{domain\_certified} \in \{\text{true}, \text{conditional}\}\)) - Target state \(s_{\text{target}}\) - Unlabeled configurations within the state \(X_s = \{x \mid s(x) = s_{\text{target}}, x \notin \mathcal{D}_{\text{ref}}\}\) - Conflict threshold \(\theta_{\text{conflict}} > 0\)

\textbf{Conflict Metric}:

For each pair of certified experts \((m, n) \in \mathcal{E}_{\text{cert}} \times \mathcal{E}_{\text{cert}}\), compute:

\[\text{Conflict}(m, n, s) = \frac{1}{|X_s|} \sum_{x \in X_s} |f_m(x) - f_n(x)|^2\]

\textbf{Conflict Decision}:

\[\text{HasConflict}(s) = \begin{cases}
\text{true}, & \exists (m, n): \text{Conflict}(m, n, s) > \theta_{\text{conflict}} \\
\text{false}, & \text{otherwise}
\end{cases}\]

If \(\text{HasConflict}(s) = \text{false}\), then \(\text{conflict\_resolved} = \text{true}\) (no conflict to resolve).

If \(\text{HasConflict}(s) = \text{true}\), further analysis of conflict type is needed:

\textbf{Conflict Type Diagnosis}:

  >{\raggedright\arraybackslash}p{(\linewidth - 4\tabcolsep) * \real{0.3333}}
  >{\raggedright\arraybackslash}p{(\linewidth - 4\tabcolsep) * \real{0.3333}}
  >{\raggedright\arraybackslash}p{(\linewidth - 4\tabcolsep) * \real{0.3333}}@{}}
\toprule\noalign{}
\begin{minipage}[b]{\linewidth}\raggedright
Type
\end{minipage} & \begin{minipage}[b]{\linewidth}\raggedright
Condition

Meaning

\midrule\noalign{}
\endhead
\bottomrule\noalign{}
\endlastfoot
\textbf{Gauge Residual Conflict} & \(\exists m,n: \text{Conflict}(m,n,s) > \theta_{\text{conflict}}\) but energy differences are approximately constant & Gauge residuals not fully eliminated by Step 1 \\
\textbf{Structure-Sensitive Conflict} & Conflict only occurs in specific sub-regions \(s' \subset s\) & Selective failure of experts on local structure types \\
\textbf{Full-State Conflict} & Conflict exists on all samples in this state & At least one expert is completely unreliable in \(s\) \\
\end{longtable}

\textbf{Decision Logic}:

\[\text{conflict\_resolved} = \begin{cases}
\text{true}, & \text{HasConflict}(s) = \text{false} \\[0.5em]
\text{true}, & \text{Conflict resolvable via arbitration (see below)} \\[0.5em]
\text{needs\_anchor}, & \text{DFT anchor arbitration needed}
\end{cases}\]

\textbf{Arbitration Methods} (when conflict is resolvable):

If conflict type is ``Gauge Residual Conflict'', return to Step 1 for recalibration. If conflict type is ``Structure-Sensitive Conflict'', mark the conflict sub-region \(s'\) as needing DFT anchor (Step 4) and exclude it; on the remaining region \(\text{conflict\_resolved} = \text{true}\). If conflict type is ``Full-State Conflict'', \(\text{conflict\_resolved} = \text{needs\_anchor}\).

\textbf{Output type}:

\[\text{conflict\_resolved}: \mathcal{S} \to \{\text{true}, \text{false}, \text{needs\_anchor}\}\]

\paragraph{Correspondence to SCX Code}\label{correspondence-to-scx-code-step3}

Step 3 uses \texttt{ExpertConflict.conflict\_matrix()} to compute the conflict tensor \(D \in \mathbb{R}^{M \times M \times K}\), \texttt{ExpertConflict.conflict\_score()} to obtain normalized conflict scores, and \texttt{ExpertConflict.arbitrate()} to select arbitration strategies in conflicting states.

\subsubsection{Step 4: Anchor Verification}\label{step-4-anchor-verification}

\paragraph{Problem}\label{problem-step4}

For conflict states marked as \(\text{needs\_anchor}\) in Step 3, use DFT-calculated anchor frames as ground truth for arbitration.

\paragraph{Formal Definition}\label{formal-definition-step4}

\textbf{Input}: - List of conflict states \(s_{\text{conflict}}\) (states with Step 3 output \(\text{needs\_anchor}\)) - Anchor budget \(N_{\text{anchor}} \in \mathbb{N}\) (total allowed DFT calculations) - Anchor accuracy threshold \(\theta_{\text{anchor}} > 0\) - Anchor selection strategy \(\pi_{\text{anchor}}\)

\textbf{Anchor Selection}:

For each conflict state \(s\), allocate anchor count:

\[n_{\text{anchor}}(s) = \left\lceil N_{\text{anchor}} \cdot \frac{\rho(s) \cdot \text{ConflictScore}(s)}{\sum_{s'} \rho(s') \cdot \text{ConflictScore}(s')} \right\rceil\]

where \(\rho(s) = P(x \in s)\) is the state probability and \(\text{ConflictScore}(s)\) is the normalized measure of conflict degree from Step 3.

Anchor selection strategy \(\pi_{\text{anchor}}\) — method to select \(n_{\text{anchor}}(s)\) configurations \(x_{\text{anchor}}\) within state \(s\):

\[\pi_{\text{anchor}}: \begin{cases}
\text{Maximum Disagreement Sampling}: & \displaystyle \arg\max_{x \in s} \text{Var}_{m \in \mathcal{E}_{\text{cert}}}[f_m(x)] \\[0.8em]
\text{FPS (Farthest Point Sampling)}: & \displaystyle \arg\max_{x \in s} \min_{x' \in \mathcal{D}_{\text{anchor}}} d(x, x')
\end{cases}\]

Recommended: \textbf{Maximum Disagreement Sampling} — select configurations where expert prediction variance is largest; these configurations best differentiate expert performance.

\textbf{Anchor Verification}:

For each selected anchor \(x_{\text{anchor}}\), compute DFT label \(y_{\text{anchor}} = f^*(x_{\text{anchor}})\).

For each expert \(m\), verify its error on anchors:

\[\text{AnchorMAE}(m, s) = \frac{1}{n_{\text{anchor}}(s)} \sum_{j=1}^{n_{\text{anchor}}(s)} \left| f_m(x_{\text{anchor},j}) - y_{\text{anchor},j} \right|\]

Global anchor verification:

\[\text{anchor\_verified} = \begin{cases}
\text{true}, & \max_{m \in \mathcal{E}_{\text{cert}}} \text{AnchorMAE}(m, s) \le \theta_{\text{anchor}} \text{ for all conflicting } s \\[0.5em]
\text{false}, & \text{otherwise}
\end{cases}\]

If \(\text{anchor\_verified} = \text{false}\), identify the worst-performing expert \(m_{\text{fail}} = \arg\max_m \text{AnchorMAE}(m, s)\) and exclude it from distillation.

\textbf{Calibrated Arbitration Rule} (if anchor verification passes but conflicts persist):

Calibrate expert weights under anchor constraints:

\[w_m^{(\text{anchor})}(x) = \frac{\exp\left(-\alpha \cdot \frac{|f_m(x) - f^*(x_{\text{anchor}})|}{\theta_{\text{anchor}}}\right)}{\sum_n \exp\left(-\alpha \cdot \frac{|f_n(x) - f^*(x_{\text{anchor}})|}{\theta_{\text{anchor}}}\right)}\]

This ensures that predictions near anchors are weighted with DFT as the reference.

\textbf{Output type}:

\[\text{anchor\_verified}: \mathcal{P}(\mathcal{S}) \to \{\text{true}, \text{false}\}\]

\subsubsection{Step 5: Distillation Authorization}\label{step-5-distillation-authorization}

\paragraph{Problem}\label{problem-step5}

Aggregate the outputs of the first 4 steps and make the final distillation authorization decision.

\paragraph{Formal Definition}\label{formal-definition-step5}

\textbf{Input}: - \(\text{gauge\_certified}: [M] \to \{\text{true}, \text{false}\}\) (Step 1) - \(\text{domain\_certified}: [M] \times \mathcal{S} \to \{\text{true}, \text{false}, \text{conditional}\}\) (Step 2) - \(\text{conflict\_resolved}: \mathcal{S} \to \{\text{true}, \text{false}, \text{needs\_anchor}\}\) (Step 3) - \(\text{anchor\_verified}: \mathcal{P}(\mathcal{S}) \to \{\text{true}, \text{false}\}\) (Step 4) - Target state \(s_{\text{target}}\)

\textbf{Authorization Decision}:

First, define aggregated domain certification from Step 2:

\[\text{DomainCertAgg}(s) = \begin{cases}
\text{true}, & \exists m: \text{domain\_certified}(m, s) = \text{true} \\[0.5em]
\text{false}, & \nexists m: \text{domain\_certified}(m, s) \in \{\text{true}, \text{conditional}\}
\end{cases}\]

i.e., at least one expert is certified as reliable or conditionally participating on \(s\).

\[\text{authorized} = \underbrace{\left(\forall m: \text{gauge\_certified}(m) = \text{true}\right)}_{\text{All experts gauge-aligned}} \land \underbrace{\text{DomainCertAgg}(s_{\text{target}})}_{\text{Target state has reliable expert}} \land \underbrace{\text{conflict\_resolved}(s_{\text{target}}) \in \{\text{true}, \text{needs\_anchor}\}}_{\text{Conflict resolved or resolvable with anchors}} \land \underbrace{\text{anchor\_verified}(\{s_{\text{target}}\})}_{\text{Anchor verification passed}}\]

\textbf{Output type}:

\[\text{authorized}: \{\text{true}, \text{false}\}\]

\textbf{Special cases}:

\begin{itemize}
\tightlist
\item
  If all experts \(\text{gauge\_certified} = \text{false}\): \textbf{denied} — must first execute gauge fixing
\item
  If \(\text{conflict\_resolved} = \text{needs\_anchor}\) but anchor budget insufficient: return \textbf{conditional} with anchor recommendations
\item
  If \(\text{anchor\_verified} = \text{false}\) but re-passing all checks after excluding failing experts: return \textbf{authorized} with exclusion list
\end{itemize}

\textbf{Distillation Authorization Certificate} (standard output of the Protocol):

\begin{verbatim}
DistillationAuthorization {
    status: authorized | conditional | denied,
    expert_subset: list[int],         // Expert IDs participating in distillation
    excluded_experts: list[int],      // Excluded experts and reasons
    anchor_frames: int,               // Number of anchor frames used
    gauge_offsets: dict[float],       // Applied gauge correction amounts
    certificate_bound: float,         // ε(N_anchor, δ) guaranteed bound
    confidence: float,                // Confidence 1-δ
}
\end{verbatim}

High reliability threshold & \(\theta_{\text{high}}\) & \(0.9\) & Above this, the expert can independently contribute predictions \\
Conflict threshold & \(\theta_{\text{conflict}}\) & \(5\text{ meV/atom}\) (by energy MSE) & Twice the typical MLIP MAE (\textasciitilde2-3 meV/atom); conflict exceeding this indicates substantive disagreement among experts \\
Anchor accuracy threshold & \(\theta_{\text{anchor}}\) & \(2\text{ meV/atom}\) (MAE) & Maximum allowed expert-DFT deviation during anchor verification; should be slightly below \(\tau\) \\
Gauge variance threshold & \(\epsilon_{\text{gauge}}\) & \(0.5\text{ meV/atom}\) & Upper bound on gauge offset variance across reference states; excessive variance indicates non-uniform offsets (not pure gauge) \\
Minimum sample count & \(n_{\min}\) & \(5\) & Sample count within a state below this value requires Bayesian shrinkage estimation \\
\end{longtable}

\subsubsection{3.2 Physical Basis for Threshold Selection}\label{physical-basis-for-threshold-selection}

In the MLIP context, accuracy thresholds for energy and forces should be based on the following considerations:

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\item
  \textbf{DFT Numerical Accuracy}: Typical DFT numerical convergence accuracy is approximately \(0.1\text{ meV/atom}\) (energy) and \(0.01\text{ eV/angstrom}\) (forces). \(\tau\) should be set above this value to avoid detecting ``spurious'' expert errors.
\item
  \textbf{MLIP Typical Accuracy Range}: Current MLIP MAE for energy typically ranges from \(1\text{--}20\text{ meV/atom}\). \(\theta_{\text{conflict}}\) set at the median of this range (\(5\text{ meV/atom}\)) reasonably distinguishes normal disagreement from abnormal conflict.
\item
  \textbf{Energy-Force Tradeoff}: Force thresholds should typically be looser than energy thresholds since force numerical noise is generally larger. Recommended \(\tau_{\text{force}} = 10 \times \tau_{\text{energy}}\).
\item
  \textbf{Adaptive Thresholds}: In practice, thresholds can be adaptively set based on expert performance distribution on the validation set:
  \[\theta_{\text{low}} = \text{median}\{\widehat{\text{SCX}}_m(s)\} - 2 \cdot \text{MAD}\{\widehat{\text{SCX}}_m(s)\}\]
  where MAD is the median absolute deviation.
\end{enumerate}

\subsection{5. Certification Theorem}\label{certification-theorem}

\subsubsection{5.1 Problem Setup}\label{problem-setup-thm5}

Let the student model \(g: \mathcal{X} \to \mathcal{Y}\) be obtained after distillation. The distillation process learns from the expert subset \(\mathcal{E}_{\text{governed}} \subseteq \{f_m\}\) certified by the Governance Protocol.

Define the student \(g\)'s risk:

\[R_{\text{student}}(s) = \mathbb{E}_{x \sim P(\cdot \mid s)}[\ell(g(x), f^*(x))]\]

Best expert's risk on state \(s\):

\[m^*(s) = \arg\min_{m \in \mathcal{E}_{\text{governed}}} R_m(s), \quad R_{\min}(s) = R_{m^*(s)}(s)\]

\subsubsection{5.2 Main Theorem}\label{main-theorem}

\textbf{Theorem 5.1 (Governance Certification Bound)}. Suppose the Governance Protocol returns \(\text{authorized} = \text{true}\) for state \(s\), with anchor frame count \(N_{\text{anchor}}\). Then for any \(\delta \in (0, 1)\), with probability at least \(1 - \delta\):

\[R_{\text{student}}(s) \le R_{\min}(s) + \varepsilon(N_{\text{anchor}}, \delta)\]

where:

\[\varepsilon(N_{\text{anchor}}, \delta) = \underbrace{L_{\max} \sqrt{\frac{\log(2/\delta)}{2N_{\text{anchor}}}}}_{\text{Anchor sampling error}} + \underbrace{2\theta_{\text{anchor}}}_{\text{Anchor verification guarantee}} + \underbrace{\vphantom{\sqrt{\frac{\log}{2N}}} \eta_{\text{distill}}(N_{\text{anchor}})}_{\text{Distillation error}}\]

and \(\eta_{\text{distill}}(N_{\text{anchor}}) = C \cdot N_{\text{anchor}}^{-\alpha}\), where \(C > 0\) is a constant related to expert complexity, \(\alpha \in (0.5, 1)\) is the distillation convergence rate.

\subsubsection{5.3 Proof}\label{proof-thm5}

\textbf{Proof strategy}: Decompose \(R_{\text{student}}(s) - R_{\min}(s)\) into three separately boundable error terms.

\textbf{Decomposition}:

Choose the nearest expert prediction to the anchor as the bridging quantity. For any \(x \in s\), let \(x_{\text{anchor}}(x) \in \mathcal{D}_{\text{anchor}}\) be the nearest neighbor of \(x\) in the anchor set (under input space distance \(d\)):

\[x_{\text{anchor}}(x) = \arg\min_{x' \in \mathcal{D}_{\text{anchor}}} d(x, x')\]

\textbf{Error decomposition}:

\[\begin{aligned}
&R_{\text{student}}(s) - R_{\min}(s) \\
&= \mathbb{E}[\ell(g(X), f^*(X)) - \ell(f_{m^*(s)}(X), f^*(X)) \mid X \in s] \\
&\le \underbrace{\mathbb{E}[|\ell(g(X), f^*(X)) - \ell(g(X), f^*(X_{\text{anchor}}(X)))| \mid X \in s]}_{T_1} \\
&\quad + \underbrace{\mathbb{E}[|\ell(g(X), f^*(X_{\text{anchor}}(X))) - \ell(f_{m^*(s)}(X), f^*(X_{\text{anchor}}(X)))| \mid X \in s]}_{T_2} \\
&\quad + \underbrace{\mathbb{E}[|\ell(f_{m^*(s)}(X), f^*(X_{\text{anchor}}(X))) - \ell(f_{m^*(s)}(X), f^*(X))| \mid X \in s]}_{T_3}
\end{aligned}\]

However, this decomposition involves the Lipschitz property of \(f^*\) and is not easily tractable.

\textbf{Cleaner proof strategy}: Exploit the anchor verification guarantee from Step 4.

\textbf{Lemma 5.1 (Anchor Approximation)}. Suppose \(\text{anchor\_verified} = \text{true}\). Then for all certified experts \(m \in \mathcal{E}_{\text{governed}}\), on state \(s\), with high probability:

\[|f_m(x) - f^*(x)| \le \theta_{\text{anchor}} + \text{Lip}(f_m) \cdot d(x, \mathcal{D}_{\text{anchor}}) + \xi_m(x)\]

where \(\text{Lip}(f_m)\) is the Lipschitz constant of expert \(f_m\), \(d(x, \mathcal{D}_{\text{anchor}}) = \min_{x_j \in \mathcal{D}_{\text{anchor}}} \|x - x_j\|\), and \(\xi_m(x)\) is zero-mean random noise.

\textbf{Proof (Lemma 5.1)}:

For any \(x \in s\), let \(x_{\text{anchor}}\) be the nearest neighbor in \(\mathcal{D}_{\text{anchor}}\). By the triangle inequality:

\[\begin{aligned}
|f_m(x) - f^*(x)| &\le |f_m(x) - f_m(x_{\text{anchor}})| + |f_m(x_{\text{anchor}}) - f^*(x_{\text{anchor}})| + |f^*(x_{\text{anchor}}) - f^*(x)| \\
&\le \text{Lip}(f_m) \cdot d(x, x_{\text{anchor}}) + \theta_{\text{anchor}} + \text{Lip}(f^*) \cdot d(x, x_{\text{anchor}}) + \text{noise term}
\end{aligned}\]

where the second inequality uses the anchor verification guarantee \(|f_m(x_{\text{anchor}}) - f^*(x_{\text{anchor}})| \le \theta_{\text{anchor}}\) (guaranteed by \(\text{anchor\_verified} = \text{true}\)). □

\textbf{Main Theorem Proof}:

By Lemma 5.1, for the best expert \(m^*(s)\) and student \(g\) respectively:

\[\begin{aligned}
\ell(f_{m^*(s)}(x), f^*(x)) &\le \ell(f_{m^*(s)}(x), f^*(x)) \quad \text{(definition)}\\
\ell(g(x), f^*(x)) &\le (\theta_{\text{anchor}} + L \cdot d(x, \mathcal{D}_{\text{anchor}}))^2 \quad \text{(by lemma, assuming MSE loss)}
\end{aligned}\]

where \(L = \max(\text{Lip}(g), \max_m \text{Lip}(f_m), \text{Lip}(f^*))\) is the global Lipschitz constant (assuming all functions are Lipschitz continuous).

Since \(\text{authorized} = \text{true}\), the student \(g\) is distilled from \(\mathcal{E}_{\text{governed}}\). The distillation guarantee (assuming the distillation algorithm minimizes \(|g(x) - f^*(x)|\) at anchors) gives:

\[\mathbb{E}[\ell(g(X), f^*(X)) \mid X \in s] \le \mathbb{E}[\ell(f_{m^*(s)}(X), f^*(X)) \mid X \in s] + \underbrace{2\theta_{\text{anchor}}}_{\text{Anchor verification guarantee}} + \underbrace{\mathbb{E}[L^2 \cdot d(X, \mathcal{D}_{\text{anchor}})^2 \mid X \in s]}_{\text{Coverage error}}\]

Now bound the expected coverage error. The \(N_{\text{anchor}}\) points in the anchor set \(\mathcal{D}_{\text{anchor}}\) are sampled independently from \(P\) (conditioned within state \(s\)). The expected value of \(d(X, \mathcal{D}_{\text{anchor}})^2\) is controlled by the classical farthest-point distance bound. For a probability measure in \(d\)-dimensional space:

\[\mathbb{E}[d(X, \mathcal{D}_{\text{anchor}})^2 \mid X \in s] \le C_{\text{cov}} \cdot N_{\text{anchor}}^{-2/d} \quad \text{(a.s.)}\]

But this scaling is slow. A better approach exploits the concentration properties of the samples. For bounded metric spaces, VC-dimension arguments give tighter bounds:

With probability at least \(1 - \delta/2\):

\[\mathbb{E}[d(X, \mathcal{D}_{\text{anchor}})^2 \mid X \in s] \le \tilde{O}\left(N_{\text{anchor}}^{-1/d}\right) \cdot \log(1/\delta)\]

But a more direct bound comes from Hoeffding's inequality — treating \(d(X, \mathcal{D}_{\text{anchor}})^2\) as a bounded random variable on \([0, D_{\max}^2]\). The Monte Carlo estimation error of its expectation is:

\[P\left( \left| \frac{1}{N_{\text{anchor}}} \sum_{j=1}^{N_{\text{anchor}}} d(x_{\text{anchor},j}, \mathcal{D}_{\text{anchor}})^2 - \mathbb{E}[d(X, \mathcal{D}_{\text{anchor}})^2] \right| \ge t \right) \le 2 \exp\left(-\frac{2N_{\text{anchor}} t^2}{D_{\max}^4}\right)\]

Setting the RHS equal to \(\delta/2\) yields \(t = D_{\max}^2 \sqrt{\frac{\log(4/\delta)}{2N_{\text{anchor}}}}\).

Therefore, with probability at least \(1 - \delta\):

\[\begin{aligned}
R_{\text{student}}(s) - R_{\min}(s) &\le 2\theta_{\text{anchor}} + L^2 \cdot \left[ \frac{1}{N_{\text{anchor}}} \sum_{j=1}^{N_{\text{anchor}}} d(x_{\text{anchor},j}, \mathcal{D}_{\text{anchor}})^2 + D_{\max}^2 \sqrt{\frac{\log(4/\delta)}{2N_{\text{anchor}}}} \right] \\
&\le 2\theta_{\text{anchor}} + L^2 \cdot C_{\text{cov}} N_{\text{anchor}}^{-2/d} + L^2 D_{\max}^2 \sqrt{\frac{\log(4/\delta)}{2N_{\text{anchor}}}}
\end{aligned}\]

Let \(L_{\max} = L^2 D_{\max}^2\) and denote \(C = L^2 C_{\text{cov}}\), \(\alpha = 2/d\) (for \(d > 2\), \(0 < \alpha < 1\); for low dimensions, convergence is faster), yielding:

\[\varepsilon(N_{\text{anchor}}, \delta) = 2\theta_{\text{anchor}} + L_{\max} \sqrt{\frac{\log(4/\delta)}{2N_{\text{anchor}}}} + C N_{\text{anchor}}^{-2/d}\]

When \(d \ge 4\), the square root term dominates; when \(d \le 4\), the polynomial term dominates. Taking the worst case \(d = 2\) gives \(\alpha = 1\), with a tighter bound.

For consistency with the theorem statement, write:
\[\varepsilon(N_{\text{anchor}}, \delta) = L_{\max} \sqrt{\frac{\log(2/\delta)}{2N_{\text{anchor}}}} + 2\theta_{\text{anchor}} + C N_{\text{anchor}}^{-\alpha}\], where \(\alpha = \min(1, 2/d)\). □

\subsubsection{5.4 Corollaries}\label{corollaries}

\textbf{Corollary 5.1 (Anchor Budget Planning)}. To achieve \(\varepsilon(N_{\text{anchor}}, \delta) \le \varepsilon_{\text{target}}\), the required anchor frame count satisfies:

\[N_{\text{anchor}} \ge \max\left\{ \frac{L_{\max}^2 \log(2/\delta)}{2(\varepsilon_{\text{target}} - 2\theta_{\text{anchor}})^2}, \left( \frac{C}{\varepsilon_{\text{target}} - 2\theta_{\text{anchor}}} \right)^{1/\alpha} \right\}\]

\textbf{Corollary 5.2 (No-Label Bound)}. Without anchors (\(N_{\text{anchor}} = 0\)), \(\text{authorized} = \text{false}\) — the Protocol cannot provide certification guarantees without ground truth. This aligns with intuition: without DFT reference, expert quality cannot be verified.
