\begin{abstract}
We introduce the \textbf{Audit Group} \(\AuditGrp\)---the symmetry group of the Consensus Equivalence Class (CEC), and prove the Galois correspondence between subgroup structure and auditability. Core results: (1)~\textbf{Unsolvability Theorem}---if \(\AuditGrp\) contains \(A_5\), then universal auditability is impossible; (2)~\textbf{Normalizer Independence Theorem}---expert sub-teams are independently auditable iff their audit subgroups normalize each other, with a direct connection to Situs Lipschitz decomposability; (3)~\textbf{Fixed Domain Theorem}---\(\CEC = \Fix(\AuditGrp)\), proving that consensus is precisely the fixed points of the audit group.
\textbf{Keywords:} Galois correspondence; audit group; unsolvability; \(A_5\); consensus equivalence class; fixed domain; SCX
\end{abstract}

\section{Introduction: From Field Theory to Audit Theory}
\subsection{Motivation}
The SCX framework's core question: \textbf{when can we trust multi-expert consensus?}

The Honest Person Theorem (SCX-Thm~3) tells us: we cannot distinguish label noise from genuine sample difficulty from observational data alone. But it does not tell us \textbf{why}---why does the audit fail? The answer lies in group theory: just as the Galois group \(\Gal(L/K)\) determines which field elements are expressible by radicals, the Audit Group \(\AuditGrp\) determines which claims are auditable.

\begin{itemize}[itemsep=4pt]
    \item The \textbf{Consensus Equivalence Class (CEC)} \(\longleftrightarrow\) the Galois \textbf{base field} \(K\);
    \item \textbf{Expert evaluations} \(\longleftrightarrow\) \textbf{field extensions};
    \item The \textbf{audit group} \(\AuditGrp\) preserving the CEC \(\longleftrightarrow\) the \textbf{Galois group} \(\Gal(L/K)\);
    \item \textbf{Auditability} \(\longleftrightarrow\) \textbf{solvability by radicals} (fundamental theorem of Galois theory).
\end{itemize}

\subsection{Notation}
\begin{itemize}[itemsep=3pt]
    \item \(\ExpertSet = \{E_1, E_2, \ldots, E_n\}\): the set of expert auditors.
    \item \(\ClaimSpace\): the claim space; expert \(E_i\) assigns verdict \(E_i(x) \in \{0,1\}\) for each claim \(x \in \ClaimSpace\).
    \item \(\CEC\): the Consensus Equivalence Class---claims that all experts agree on.
    \item \(\AuditGrp \leq \Sym(\ExpertSet)\): the audit group---permutations of experts that preserve the \textbf{CEC structure}.
    \item \(\Fix(H)\): the fixed points of subgroup \(H \leq \AuditGrp\).
\end{itemize}

\section{The Audit Group}
% ----------------------------------------------------------------
\subsection{CEC-Preserving Permutations}
\begin{definition}[Expert Permutation]
Let \(\ExpertSet = \{1,2,\ldots,n\}\) index experts. \(\Sym_n\) acts on \(\ExpertSet\) via \(\sigma \cdot E_i = E_{\sigma(i)}\). The induced action on verdicts is:
\begin{equation}
    \sigma \cdot (E_1(x), E_2(x), \ldots, E_n(x)) = (E_{\sigma(1)}(x), E_{\sigma(2)}(x), \ldots, E_{\sigma(n)}(x)).
\end{equation}
\end{definition}

\begin{definition}[Audit Group \(\AuditGrp\)]\label{def:audit_group}
The \textbf{Audit Group} \(\AuditGrp\) is the subgroup of permutations that preserve consensus:
\begin{equation}\label{eq:audit_group_def}
    \AuditGrp = \left\{ \sigma \in \Sym_n \mid
    \forall x, y \in \ClaimSpace,\; x \sim_{\CEC} y \implies \sigma\cdot x \sim_{\CEC} \sigma\cdot y \right\},
\end{equation}
where \(x \sim_{\CEC} y\) means \(x\) and \(y\) belong to the same consensus equivalence class (all experts agree on their classification).
\end{definition}

\textbf{This is precisely the Galois condition.} In Galois theory, \(\Gal(L/K) = \{\sigma \in \Aut(L) \mid \sigma|_K = \mathrm{id}_K\}\). Analogously, \(\AuditGrp\) consists of expert permutations that preserve the CEC structure. The CEC plays the role of the base field \(K\).

\begin{proposition}[Basic Properties]\label{prop:basic}
The audit group \(\AuditGrp\) defined in Definition~\ref{def:audit_group} satisfies:
\begin{enumerate}[label=(\roman*),itemsep=3pt]
    \item \(\AuditGrp\) is a subgroup of \(\Sym_n\).
    \item If no two experts \(i \neq j\) ever agree on any claim, then \(\AuditGrp = \Sym_n\) (full symmetry---no structure).
    \item If all experts always agree perfectly, then \(\AuditGrp = \{\mathrm{id}\}\) (trivial symmetry---total structure).
    \item The size \(|\AuditGrp|\) measures \textbf{audit degeneracy}: the larger the group, the more symmetries obscure genuine disagreements.
\end{enumerate}
\end{proposition}
\begin{proof}
(i) Composition of CEC-preserving permutations preserves the CEC. If \(\sigma, \tau\) both preserve \(\sim_{\CEC}\), then for \(x \sim_{\CEC} y\), we have \(\tau \cdot x \sim_{\CEC} \tau \cdot y\) (since \(\tau \in \AuditGrp\)), and then \(\sigma\cdot(\tau\cdot x) \sim_{\CEC} \sigma\cdot(\tau\cdot y)\) (since \(\sigma \in \AuditGrp\)), so \(\sigma\tau \in \AuditGrp\). The identity clearly preserves the CEC.

(ii) If experts never agree, any permutation trivially preserves the (empty) CEC, giving \(\AuditGrp = \Sym_n\).

(iii) If \(E_i\) and \(E_j\) always differ on some \(x\), then swapping them would break the CEC, so only the identity preserves it: \(\AuditGrp = \{\mathrm{id}\}\).

(iv) A larger \(|\AuditGrp|\) means more expert permutations leave the CEC unchanged---i.e., more configurations are consistent with the same observed consensus, making genuine disagreements harder to detect.
\end{proof}
% ----------------------------------------------------------------
\subsection{Orbits and Ambiguity}
\begin{definition}[Audit Orbit]
The \textbf{audit orbit} of claim \(x \in \ClaimSpace\) is:
\begin{equation}
    \mathcal{O}(x) = \{\sigma \cdot x \mid \sigma \in \AuditGrp\}.
\end{equation}
The size \(|\mathcal{O}(x)|\) measures the \textbf{audit ambiguity} of claim \(x\).
\end{definition}
\begin{remark}
There is a \textbf{privacy-auditability tradeoff}. A large orbit (\(|\mathcal{O}(x)| > 1\)) means multiple distinct claims are indistinguishable under the current expert configuration---a form of ``audit privacy'' that also makes auditing impossible.
\end{remark}

\section{The Galois Correspondence for Auditing}
\subsection{Auditable Subspaces}
\begin{definition}[Auditable Subspace]
A subset \(\mathcal{P} \subseteq \ClaimSpace\) is \textbf{auditable} if there exists an algorithm \(\mathcal{A}\) that can determine the truth value of all claims in \(\mathcal{P}\) given sufficient expert verdicts (formalized in SCX Core Theorems~\cite{scx2026theorems}). Denote the lattice of auditable subspaces by \(\mathcal{L}_{\mathrm{audit}}(\ClaimSpace)\).
\end{definition}

\begin{definition}[Fixed Subspace]
For \(H \leq \AuditGrp\), the \textbf{fixed subspace} is:
\begin{equation}
    \Fix(H) = \{x \in \ClaimSpace \mid \forall \sigma \in H,\; \sigma \cdot x = x\}.
\end{equation}
These are claims whose verdict pattern is invariant under all symmetries in \(H\).
\end{definition}

\begin{definition}[Stabilizer]
For \(\mathcal{P} \subseteq \ClaimSpace\), the \textbf{stabilizer} is:
\begin{equation}
    \Stab(\mathcal{P}) = \{\sigma \in \AuditGrp \mid \forall x \in \mathcal{P},\; \sigma \cdot x \in \mathcal{P}\}.
\end{equation}
\end{definition}

\begin{theorem}[Audit Galois Correspondence]\label{thm:galois_correspondence}
Let \(\AuditGrp\) be the audit group and \(\mathcal{L}_{\mathrm{audit}}\) the lattice of auditable subspaces. There is a \textbf{Galois connection}:
\begin{equation}\label{eq:galois_correspondence}
    \begin{tikzcd}
        \{\text{subgroups of }\AuditGrp\} \ar[r, shift left=2, "\Fix"] &
        \{\text{auditable subspaces}\} \ar[l, shift left=2, "\Stab"]
    \end{tikzcd}
\end{equation}
satisfying, for all \(H \leq \AuditGrp\) and auditable \(\mathcal{P}\):
\begin{enumerate}[label=(\roman*),itemsep=3pt]
    \item \(\Fix(H)\) is auditable, and \(H_1 \leq H_2 \implies \Fix(H_2) \subseteq \Fix(H_1)\) (order-reversing).
    \item \(\Stab(\mathcal{P})\) is a subgroup of \(\AuditGrp\), and \(\mathcal{P}_1 \subseteq \mathcal{P}_2 \implies \Stab(\mathcal{P}_2) \leq \Stab(\mathcal{P}_1)\) (order-reversing).
    \item \(\Stab(\Fix(H)) \supseteq H\), and \(\Fix(\Stab(\mathcal{P})) \supseteq \mathcal{P}\) (Galois closure properties).
    \item The correspondence is a bijection between Galois-closed subgroups \(H\) (those with \(H = \Stab(\Fix(H))\)) and Galois-closed subspaces. Under this bijection, \(\Fix\) and \(\Stab\) are mutual inverses.
\end{enumerate}
\end{theorem}
\begin{proof}[Proof Sketch]
\rigorous Proof sketch.

(i) Let \(H \leq \AuditGrp\). For \(x \in \Fix(H)\) and \(\sigma \in H\), \(\sigma \cdot x = x\), so \(x\) is \(H\)-invariant. Moreover, if \(x \in \Fix(H)\) and \(H\) acts transitively on some orbit, then \(x\) must be constant on that orbit---a strong structural constraint. The audibility of \(\Fix(H)\) follows since \(H\)-invariant verdict patterns are precisely those for which expert disagreement within \(H\) cancels out. If \(H_1 \leq H_2\), then invariance under \(H_2\) implies invariance under \(H_1\), so \(\Fix(H_2) \subseteq \Fix(H_1)\). \(\square\)

(ii) \(\Stab(\mathcal{P})\) is closed under composition (if two permutations each map \(\mathcal{P}\) into itself, so does their composition). \(\square\)

(iii) If \(\sigma \in H\) and \(x \in \Fix(H)\), then \(\sigma \cdot x = x \in \Fix(H)\), so \(H \leq \Stab(\Fix(H))\). The reverse inclusion \(\mathcal{P} \subseteq \Fix(\Stab(\mathcal{P}))\) follows similarly. \(\square\)
\end{proof}
\begin{remark}
This is not just an analogy---it is a genuine Galois correspondence. The \textbf{Fundamental Theorem of Galois Theory} translates directly: auditable subspaces correspond to intermediate fields, and the audit group's subgroup lattice mirrors the lattice of auditable claims. The ``unsolvability'' result below corresponds exactly to the non-solvability of quintic equations.
\end{remark}

\section{The Unsolvability Theorem}\label{sec:unsolvability}
\subsection{The Galois Obstruction}
The fundamental theorem of Galois theory states: a polynomial is solvable by radicals iff its Galois group is \textbf{solvable}. The symmetric group \(S_n\) (for \(n \geq 5\)) contains \(A_5\), which is simple and non-abelian, hence not solvable. Therefore the general quintic is not solvable by radicals.

\begin{definition}[Solvable and Unsolvable Audit Groups]
An audit group \(\AuditGrp\) is \textbf{solvable} if it admits a subnormal series with abelian quotients. Otherwise it is \textbf{unsolvable}. In particular, if \(\AuditGrp\) contains \(A_5\) (the alternating group of order 60), then \(\AuditGrp\) is unsolvable.
\end{definition}

\begin{theorem}[Unsolvability of Universal Auditing]\label{thm:unsolvability}
Let \(\AuditGrp\) be the audit group. If \(\AuditGrp\) is unsolvable (contains \(A_5\)), then there exist claims \(x \in \ClaimSpace\) such that:
\begin{enumerate}[label=(\roman*),itemsep=3pt]
    \item \(x\) cannot be audited by any algorithm operating on expert verdicts within the orbit \(\mathcal{O}(x)\);
    \item No finite extension of the expert set can render \(x\) auditable while preserving the \(\CEC\) structure.
\end{enumerate}
Equivalently: \textbf{non-solvable audit group \(\Longleftrightarrow\) existence of unauditable claims.}
\end{theorem}
\begin{proof}[Proof Sketch]
\rigorous Let \(H \leq \AuditGrp\) with \(H \cong A_5\). Since \(A_5\) acts on the claim space through the audit group. As a simple non-abelian group, \(A_5\) has irreducible representations of dimension \(\geq 3\) that are not expressible via nested radicals (i.e., not solvable by sequential abelian extensions).

The obstruction is that \(H\) has \textbf{no nontrivial abelian quotients} (\(A_5\) is simple), so there is no sequence of ``reduction steps'' that simplifies the audit problem. In the Galois analogy: claims with \(A_5\) symmetry correspond to quintic equations---fundamentally unsolvable.

Concretely, let \(\mathcal{O}\) be a nontrivial \(A_5\)-orbit. Any audit function \(f: \mathcal{O} \to \{0,1\}\) that is \(H\)-invariant (must give the same result for symmetric claims: \(f(\sigma\cdot x) = f(x)\)) must be constant---\(A_5\) is 2-transitive. Thus no nontrivial audit is possible on \(A_5\)-symmetric claims. \(\square\)
\end{proof}

\begin{corollary}[Conditions for Auditability]
A claim \(x\) is auditable if and only if:
\begin{enumerate}[label=(\arabic*),itemsep=3pt]
    \item \(x\) lies in an auditable subspace;
    \item \(\Stab(\{x\})\) (the stabilizer of \(x\) in \(\AuditGrp\)) is a \textbf{solvable group};
    \item The audit group acts with manageable orbit structure on \(\mathcal{O}(x)\).
\end{enumerate}
\end{corollary}
\begin{remark}
\textbf{Practical implication:} to make auditing possible, design expert panels whose audit group is solvable. This means avoiding expert configurations where disagreements exhibit \(A_5\)-type symmetry---precisely the situation where ``every expert disagrees with every other expert in an irreducibly complex pattern.''
\end{remark}

\section{Normalizer Independence: When Can Sub-Teams Audit Independently?}
\subsection{The Problem}
A classical Galois theory result: if \(H \Normal G\) is a normal subgroup, then \(\Fix(H)/K\) is itself a Galois extension. The audit analog: when can two subsets of experts audit independently?

\begin{definition}[Expert Sub-Teams]
Partition \(\ExpertSet = \mathcal{E}_1 \cup \mathcal{E}_2\) into two sub-teams. Let \(G_1 = \AuditGrp \cap \Sym(\mathcal{E}_1)\) and \(G_2 = \AuditGrp \cap \Sym(\mathcal{E}_2)\) be their respective audit subgroups. The sub-teams audit \textbf{independently} if their audit algorithms \(\mathcal{A}_1, \mathcal{A}_2\) can be composed without information leakage:
\begin{equation}
    \mathcal{A}(x) = \mathcal{A}_1(x) \otimes \mathcal{A}_2(x),
\end{equation}
\end{definition}

\begin{theorem}[Normalizer Independence]\label{thm:normalizer}
Let \(G_1, G_2 \leq \AuditGrp\). The following are equivalent:
\begin{enumerate}[label=(\roman*),itemsep=3pt]
    \item \textbf{Normalizer condition:} \(G_1 \leq N_{\AuditGrp}(G_2)\) and \(G_2 \leq N_{\AuditGrp}(G_1)\);
    \item \textbf{Direct product structure:} \(G_1\) and \(G_2\) generate a direct product within \(\AuditGrp\);
    \item \textbf{Independent auditability:} under the normalizer condition, the sub-teams \(\mathcal{E}_1\) and \(\mathcal{E}_2\) are independently auditable: \(\CEC(G_1) \indep \CEC(G_2)\).
\end{enumerate}
\end{theorem}
\begin{proof}[Proof Sketch]
\rigorous We prove (i) \(\implies\) (ii) \(\implies\) (iii) \(\implies\) (i).

\textbf{(i) \(\implies\) (ii):} If \(G_1, G_2\) normalize each other, then \(G_1 G_2\) is a subgroup of \(\AuditGrp\) and \(G_1 \cap G_2 \Normal G_1 G_2\). The short exact sequence
\[
    1 \to G_1 \cap G_2 \to G_1 G_2 \to (G_1 G_2)/(G_1 \cap G_2) \to 1
\]
splits because \((G_1 G_2)/(G_1 \cap G_2) \cong G_1/(G_1 \cap G_2) \times G_2/(G_1 \cap G_2)\), giving a direct product structure. By the Galois correspondence (Theorem~\ref{thm:galois_correspondence}), \(\Fix(G_1 G_2) = \Fix(G_1) \cap \Fix(G_2)\), and the direct product structure implies that the fixed subspaces compose independently. Situs Lipschitz decomposability~\cite{scx_lipschitz} provides the computational analog. \(\square\)

\textbf{(ii) \(\implies\) (iii):} Direct product decomposition permits independent audit algorithms \(\mathcal{A}_1 \otimes \mathcal{A}_2\) that do not interfere. \(\square\)

\textbf{(iii) \(\implies\) (i):} If \(G_1 \not\leq N_{\AuditGrp}(G_2)\), there exists \(g_1 \in G_1\) with \(g_1 G_2 g_1^{-1} \neq G_2\). The action of \(g_1\) then ``mixes'' the audit structure of \(G_2\), creating dependence between \(\CEC(G_1)\) and \(\CEC(G_2)\)---violating independence. Hence normalizer condition is necessary. Symmetric argument gives \(G_2 \leq N_{\AuditGrp}(G_1)\). \(\square\)
\end{proof}

\begin{remark}[Situs Lipschitz Connection]
The Situs Lipschitz decomposition~\cite{scx_lipschitz} provides the Lipschitz constant that bounds how much one sub-team's audit can affect another's. Theorem~\ref{thm:normalizer} shows that when the normalizer condition holds, this constant is zero---\textbf{perfect decomposability}.
\end{remark}

\section{Fixed Domain Theorem: CEC as Fixed Points}
\subsection{The Main Identity}
\begin{theorem}[Fixed Domain Theorem]\label{thm:fixed_domain}
The Consensus Equivalence Class \(\CEC\) is precisely the fixed-point set of the audit group \(\AuditGrp\):
\begin{equation}\label{eq:cec_fix}
    \CEC = \Fix(\AuditGrp) = \{x \in \ClaimSpace \mid \forall \sigma \in \AuditGrp,\; \sigma \cdot x = x\}.
\end{equation}
Moreover, any \(\AuditGrp\)-invariant function \(f\) on \(\ClaimSpace\) factors uniquely through the \(\CEC\).
\end{theorem}
\begin{proof}
\rigorous Proof.

\textbf{Inclusion \(\CEC \subseteq \Fix(\AuditGrp)\):} Let \(x \in \CEC\). By definition, all experts agree on \(x\). Any \(\sigma \in \AuditGrp\) (Definition~\ref{def:audit_group}) permutes experts while preserving the CEC, so \(\sigma \cdot x\) must have the same consensus status as \(x\). Since \(x\) is already in consensus, \(\sigma \cdot x = x\) (the CEC is uniquely determined). Thus \(x \in \Fix(\AuditGrp)\).

\textbf{Inclusion \(\Fix(\AuditGrp) \subseteq \CEC\):} Let \(x \in \Fix(\AuditGrp)\). Then for all \(\sigma \in \AuditGrp\), \(\sigma \cdot x = x\). This means no permutation of experts can change the verdict pattern on \(x\)---all experts must give the same verdict. Hence \(x \in \CEC\).

\textbf{Factorization:} Any function \(f: \ClaimSpace \to \mathcal{Y}\) that is \(\AuditGrp\)-invariant (\(f(\sigma \cdot x) = f(x)\) for all \(\sigma \in \AuditGrp\)) must be constant on each \(\AuditGrp\)-orbit. Since \(\Fix(\AuditGrp) = \CEC\) is the set of \(\AuditGrp\)-fixed points, the orbit space \(\ClaimSpace / \AuditGrp\) has \(\CEC\) as its ``zero section,'' and \(f\) factors through the quotient. Thus \(f\) is determined entirely by its values on \(\CEC\). \(\square\)
\end{proof}

\begin{corollary}[CEC as Galois Base Field]
\begin{enumerate}[label=(\arabic*),itemsep=3pt]
    \item The CEC is the ``absolute invariants'' of the expert system (SCX's analog of the rational numbers);
    \item \(\Fix(\AuditGrp)\) is algorithmically computable when \(\AuditGrp\) is solvable;
    \item The \textbf{orbit space} \(\ClaimSpace / \AuditGrp\) parameterizes distinct verdict patterns up to expert relabeling.
\end{enumerate}
\end{corollary}
\begin{remark}
The size of the CEC relative to \(\ClaimSpace\) measures how much ``real consensus'' exists. Since \(\AuditGrp\) acts on \(\ClaimSpace\), the orbit-stabilizer theorem gives:
\begin{itemize}
    \item The CEC occupies a fraction \(1 / [\Sym_n : \AuditGrp]\) of the symmetric case;
    \item The ``depth'' of the CEC (its structural richness) is inversely proportional to \(|\AuditGrp|\);
    \item The ``width'' of the CEC (number of distinct consensus claims) grows with \(|\AuditGrp|\).
\end{itemize}
\end{remark}

\section{Correlated Experts as Non-normal Subgroups}
\subsection{The Correlation-Normality Duality}
A direct consequence of Theorem~2 (Strong Man Theorem): \textbf{correlated experts form non-normal subgroups.}

\begin{definition}[Normal vs.\ Non-normal Subgroups]
Let \(H \leq \AuditGrp\). \(H\) is \textbf{normal} (\(H \Normal \AuditGrp\)) if conjugation by any element of \(\AuditGrp\) preserves \(H\). Otherwise \(H\) is \textbf{non-normal}, representing correlated sub-teams whose correlation structure is not symmetric with respect to other experts.
\end{definition}

\begin{theorem}[Correlation-Normality Theorem]\label{thm:correlation_nonnormal}
Let \(H \leq \AuditGrp\) represent a sub-team of experts. \(H\) is non-normal in \(\AuditGrp\) iff the experts in \(H\) are correlated:
\begin{equation}
    H \NotNormal \AuditGrp \;\Longleftrightarrow\; \text{experts in }H\text{ are correlated}.
\end{equation}
When \(H \NotNormal \AuditGrp\), the fixed subspace \(\Fix(H)\) is not Galois-closed; expert correlation introduces ``non-Galois'' audit obstructions.
\end{theorem}
\begin{proof}
\rigorous (\(\implies\)) Suppose \(H \leq \AuditGrp\) and there exists \(g \in \AuditGrp\) with \(g H g^{-1} \neq H\). This means some conjugation changes which experts are ``inside'' vs.\ ``outside'' the correlated group---a structural signature of correlation. If conjugation by \(g\) moves a correlated pair \(E_i, E_j\) in \(H\) to a pair where the correlation is broken, then \(g H g^{-1}\) must differ from \(H\) in its correlation structure. Thus non-normality implies correlation.

(\(\Longleftarrow\)) If \(H \Normal \AuditGrp\), then by the normalizer theorem (Theorem~\ref{thm:normalizer}), the sub-team \(H\) is independent from \(\AuditGrp/H\), implying zero correlation. Hence correlation implies non-normality. \(\square\)
\end{proof}

\begin{corollary}[Strong Man Theorem --- Group-Theoretic Restatement]
Theorem~2 (Strong Man) states: correlated experts produce false consensus. In group-theoretic terms:
\begin{center}
    \textbf{Correlated sub-team \(H\) \(\iff\) \(H \NotNormal \AuditGrp\).}
\end{center}
Implication: \textbf{normality testing is correlation detection.}
\end{corollary}

\section{Appendix: Group Theory Background}
\subsection{Subgroup Lattices}
\begin{definition}[Subgroup Lattice]
For a group \(G\), \(\mathcal{L}(G)\) denotes the lattice of subgroups. The partial order is \(H_1 \leq H_2\) iff \(H_1 \subseteq H_2\).
\end{definition}

\begin{definition}[Normal Subgroup]
\(N \leq G\) is \textbf{normal} (\(N \Normal G\)) if for all \(g \in G\), \(g N g^{-1} = N\).
\end{definition}

\begin{definition}[Normalizer]
For \(H \leq G\), the \textbf{normalizer} is:
\begin{equation}
    N_G(H) = \{g \in G \mid g H g^{-1} = H\}.
\end{equation}
\(N_G(H)\) is the largest subgroup of \(G\) in which \(H\) is normal.
\end{definition}

\subsection{Solvable Groups and the Quintic}
\begin{definition}[Solvable Group]
\(G\) is \textbf{solvable} if there exists a chain
\begin{equation}
    \{1\} = G_0 \Normal G_1 \Normal \cdots \Normal G_k = G,
\end{equation}
where each quotient \(G_i / G_{i-1}\) is abelian.
\end{definition}

\begin{theorem}[Galois --- Unsolvability of the Quintic]
For \(n \geq 5\), \(S_n\) is not solvable. Specifically, \(S_5\) has composition factors \(\{A_5, \Z/2\Z\}\), and \(A_5\) is simple and non-abelian.
\end{theorem}

\subsection{Classical Galois Correspondence}
Let \(L/K\) be a Galois extension with \(G = \Gal(L/K)\).

\begin{theorem}[Fundamental Theorem of Galois Theory]
\[
    H \mapsto \Fix(H) = \{x \in L \mid \forall \sigma \in H,\; \sigma(x) = x\}
\]
is a lattice anti-isomorphism between subgroups of \(G\) and intermediate fields of \(L/K\). Moreover, \(H \Normal G\) iff \(\Fix(H)/K\) is Galois.
\end{theorem}
