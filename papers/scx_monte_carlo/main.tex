\begin{abstract}
\noindent
\textbf{English Abstract:} We present a unified Monte Carlo framework for SCX
(Supercomputing Claim eXamination) audit problems, encompassing five complementary
Markov Chain Monte Carlo (MCMC) methodologies: (1) \textbf{Hamiltonian Monte Carlo
on the Situs manifold} — we equip the expert-configuration space with the Situs
metric as a mass matrix, enabling efficient HMC sampling from the Boltzmann
distribution $P(E) \propto \exp(-\beta \cdot \Cercis(E))$, and prove ergodicity
under mild conditions; (2) \textbf{Sequential Monte Carlo for online audit} —
as new data arrives, particle filters update the posterior over expert
reliability via importance resampling, with a rigorous effective sample size
bound; (3) \textbf{Thermodynamic integration for Cercis calibration} — we
anneal from high-SNR (easy) to low-SNR (hard) audit problems, extracting the
Cercis score as a free-energy difference; (4) \textbf{Replica exchange for
multi-expert consensus} — parallel tempering with replicas at different
temperatures explores distinct modes of the expert-configuration space,
escaping local optima via replica swaps; (5) \textbf{Convergence diagnostics}
— we formalize audit convergence via the Gelman-Rubin $\Rhat$ statistic and
effective sample size (ESS), establishing that an audit is ``converged'' when
$\Rhat < 1.01$ for all gauge parameters $g_i$.

\vspace{6pt}
\noindent
\textbf{中文摘要：} 本文提出 SCX（超级计算声明审查）审计问题的统一蒙特卡洛框架，
涵盖五种互补的马尔可夫链蒙特卡洛（MCMC）方法：(1) \textbf{Situs 流形上的哈密顿
蒙特卡洛}——以 Situs 度规为质量矩阵配置专家空间，从 Boltzmann 分布
$P(E) \propto \exp(-\beta \cdot \Cercis(E))$ 中高效采样，并在温和条件下证明
遍历性；(2) \textbf{在线审计的序贯蒙特卡洛}——随新数据到达，粒子滤波器通过
重要性重采样更新专家可靠性的后验分布，导出严格的有效样本量界；
(3) \textbf{Cercis 校准的热力学积分}——从高信噪比（易）退火至低信噪比（难）
审计问题，以自由能差的形式提取 Cercis 分数；(4) \textbf{多专家共识的副本交换}
——不同温度副本的并行回火探索专家配置空间的不同模态，通过副本交换逃离局部
最优；(5) \textbf{收敛诊断}——通过 Gelman-Rubin $\Rhat$ 统计量和有效样本量
（ESS）形式化审计收敛，建立所有规范参数 $g_i$ 满足 $\Rhat < 1.01$ 的审计
收敛标准。

\vspace{6pt}
\noindent
\textbf{Keywords:} Hamiltonian Monte Carlo, Sequential Monte Carlo, Thermodynamic
Integration, Replica Exchange, Convergence Diagnostics, Situs Manifold, Cercis,
SCX, MCMC, Audit

\noindent

Situs流形，Cercis，SCX，MCMC，审计
\end{abstract}

\section{预备知识：SCX 审计的概率框架}
\section*{Preliminaries: Probabilistic Framework of SCX Audit}
\addcontentsline{toc}{section}{0. Preliminaries: Probabilistic Framework of SCX Audit}

\subsection{核心结构 Core Structures}

SCX 审计的核心对象包括以下数学结构：

\begin{definition}[专家配置空间 Expert Configuration Space]
令 $\Situs$ 为 Situs 流形，即所有可能专家配置 (expert configuration) 的黎曼流形。
对于 $n$ 个专家评估 $m$ 个声明，每个专家 $j$ 对声明 $i$ 给出规范姿态
$g_{ij} \in \R$，满足 $\sum_i g_{ij} = 0$（规范约束）。则一个配置为向量

\begin{equation}\label{eq:config}
    E = (g_{11}, \ldots, g_{m1}, g_{12}, \ldots, g_{m2}, \ldots, g_{mn}) \in \R^{mn-n}
\end{equation}

$\Situs$ 配备 Situs 度规 $h_{\Situs}$，由势能 Hessian 定义：
$h_{\Situs} = \nabla^2 \mathcal{S}_{\text{total}}$，其中
$\mathcal{S}_{\text{total}}$ 为总势能面。
\end{definition}

\begin{definition}[Cercis 分数 Cercis Score]
Cercis 分数 $\Cercis: \Situs \to \R$ 是规范不变的可观测量，定义为

\begin{equation}\label{eq:cercis}
    \Cercis(E) = \frac{1}{2} \sum_{i,j} g_{ij}^2 - \lambda \sum_{i} \left( \sum_{j} g_{ij} \right)^2
\end{equation}

其中 $\lambda > 0$ 为规范固定参数。Cercis 分数衡量专家配置的"不协调程度"：
$\Cercis(E) = 0$ 对应于完美的 $\sum g = 0$ 共识；$\Cercis(E)$ 越大，配置越偏离
共识。
\end{definition}

\begin{definition}[Boltzmann 分布 Boltzmann Distribution]
在逆温度 $\beta = 1/T$ 下，Situs 流形上的 Boltzmann 分布为

\begin{equation}\label{eq:boltzmann}
    \pi_\beta(E) = \frac{1}{Z(\beta)} \exp\!\big(-\beta \cdot \Cercis(E)\big),\quad
    Z(\beta) = \int_{\Situs} e^{-\beta \cdot \Cercis(E)} \, d\Vol_{\Situs}(E)
\end{equation}

其中 $Z(\beta)$ 为配分函数，$d\Vol_{\Situs}(E)$ 为 Situs 度规诱导的体积元。
\end{definition}

\begin{remark}
$\pi_\beta$ 是 SCX 审计的\textbf{核心采样目标}。低 $\beta$（高温）对应弱约束，
配置空间广泛探索；高 $\beta$（低温）对应强约束，采样集中在 $\Cercis$ 的低谷
（高共识区域）。极限 $\beta \to \infty$ 恢复确定性审计结果。
\end{remark}

\subsection{蒙特卡洛审计原理 The Monte Carlo Audit Principle}

\begin{principle}[蒙特卡洛审计原理 MC Audit Principle]
审计即从 Boltzmann 分布 $\pi_\beta(E)$ 中采样。给定 $N$ 个独立样本
$\{E^{(k)}\}_{k=1}^N \sim \pi_\beta$：
\begin{itemize}
    \item \textbf{点估计：} $\hat{g}_{ij} = \frac{1}{N} \sum_{k=1}^N g_{ij}^{(k)}$
    \item \textbf{不确定度：} $\widehat{\Var}[g_{ij}] = \frac{1}{N-1} \sum_{k=1}^N (g_{ij}^{(k)} - \hat{g}_{ij})^2$
    \item \textbf{共识概率：} $\widehat{\Pbb}[\sum_j g_{ij} \approx 0] = \frac{1}{N} \sum_{k=1}^N \mathbb{1}[|\sum_j g_{ij}^{(k)}| < \epsilon]$
\end{itemize}
当 $N \to \infty$ 时，这些估计以概率 1 收敛于真实后验量（大数定律）。
\end{principle}

\subsection{从欧几里得到 Situs：为什么需要流形上的 HMC？}

标准 HMC 假设状态空间为 $\R^d$ 配备欧几里得度规 $M = I$。在 SCX 审计中，
专家配置空间 $\Situs$ 并非平坦：Situs 度规 $h_{\Situs} = \nabla^2 \mathcal{S}$
在势能面陡峭处收缩，在平缓处扩张。忽略此几何结构导致以下问题：

\begin{enumerate}[label=(\arabic*)]
    \item \textbf{采样效率低下：} 在高度弯曲区域，欧几里得 HMC 需要极小步长
    \item \textbf{混合缓慢：} 无法利用 Situs 度规提供的自然坐标
    \item \textbf{遍历性丧失风险：} 欧几里得步长可能导致链被"困在"高弯曲区域
\end{enumerate}

\textbf{解决方案：} 将 Situs 度规作为 HMC 的质量矩阵，即 $M \equiv h_{\Situs}$。

\subsection{黎曼流形上的哈密顿动力学}

令 $q \in \Situs$ 为配置空间坐标（对应 $E = (g_{ij})$），$p \in T_q^*\Situs$ 为
共轭动量。定义哈密顿量：

\begin{equation}\label{eq:hamiltonian}
    H(q, p) = \frac{1}{2} p^\top M(q)^{-1} p + U(q),\quad
    U(q) = \beta \cdot \Cercis(q)
\end{equation}

其中 $M(q) = h_{\Situs}(q)$ 为位置依赖的质量矩阵（即 Situs 度规）。

\begin{keyeq}
\textbf{黎曼 HMC 的运动方程：}

\begin{align}
    \dv{q}{t} &= M(q)^{-1} p \label{eq:eom_q} \\
    \dv{p}{t} &= -\nabla_q U(q) + \frac{1}{2} \nabla_q \big[ p^\top M(q)^{-1} p \big] \label{eq:eom_p}
\end{align}

方程 \eqref{eq:eom_p} 中的第二项是\textbf{度量修正项}，补偿位置依赖质量矩阵
对流形的弯曲。这是黎曼 HMC 与欧几里得 HMC 的核心区别。
\end{keyeq}

\subsection{数值积分：广义蛙跳法}

\begin{algorithm-env}[广义蛙跳积分器 Generalized Leapfrog Integrator]
给定步长 $\epsilon$ 和步数 $L$，黎曼流形上的蛙跳积分：

\begin{algorithmic}[1]
\State \textbf{输入:} 当前位置 $q$，质量矩阵 $M(q)$
\State \textbf{采样动量:} $p \sim \Ncal(0, M(q))$
\State \textbf{半步动量更新:}
    $p \leftarrow p - \frac{\epsilon}{2} \nabla_q U(q)
        + \frac{\epsilon}{4} \nabla_q \big[ p^\top M(q)^{-1} p \big]$
\For{$l = 1$ \textbf{to} $L-1$}
    \State \textbf{全步位置更新:}
        $q \leftarrow q + \epsilon \cdot M(q)^{-1} p$
    \State \textbf{全步动量更新:}
        $p \leftarrow p - \epsilon \nabla_q U(q)
            + \frac{\epsilon}{2} \nabla_q \big[ p^\top M(q)^{-1} p \big]$
\EndFor
\State \textbf{最后全步位置更新:}
    $q \leftarrow q + \epsilon \cdot M(q)^{-1} p$
\State \textbf{最后半步动量更新:}
    $p \leftarrow p - \frac{\epsilon}{2} \nabla_q U(q)
        + \frac{\epsilon}{4} \nabla_q \big[ p^\top M(q)^{-1} p \big]$
\State \textbf{Metropolis 接受:}
    $A(q'|q) = \min(1, \exp(H(q,p) - H(q',p')))$
\end{algorithmic}
\end{algorithm-env}

\begin{remark}
当 $M(q) \equiv I$（单位矩阵，即 $\Situs$ 平坦），上述算法退化为标准欧几里得
HMC。因此，Situs-HMC 是标准 HMC 的\textbf{严格推广}。
\end{remark}

\subsection{遍历性证明}

\begin{theorem}[Situs-HMC 的遍历性 Ergodicity of Situs-HMC]
\label{thm:ergodicity}
设 $\Situs$ 为紧致光滑黎曼流形，Cercis 函数 $\Cercis: \Situs \to \R$ 为
$C^2$ 光滑，且 Situs 度规 $h_{\Situs} = \nabla^2 \mathcal{S}$ 在 $\Situs$ 上
一致正定（即存在 $\lambda_{\min} > 0$ 使得对所有 $q \in \Situs$ 有
$h_{\Situs}(q) \succeq \lambda_{\min} I$）。则具有广义蛙跳积分器和非零步长
$\epsilon > 0$ 的 Situs-HMC 链是

\begin{enumerate}[label=(\roman*)]
    \item \textbf{$\pi_\beta$-不变的：} $\pi_\beta$ 是转移核 $T_\epsilon$ 的平稳分布。
    \item \textbf{不可约的：} 对任意开集 $A \subset \Situs$ 且 $\pi_\beta(A) > 0$，
        从几乎任意起点 $q_0$ 出发，存在 $n > 0$ 使得链访问 $A$。
    \item \textbf{非周期的：} 不存在对状态空间的非平凡周期划分。
    \item \textbf{遍历的：} 对任意 $\pi_\beta$-可积函数 $f$，
        \begin{equation}
            \lim_{N \to \infty} \frac{1}{N} \sum_{k=1}^N f(q_k) = \int_{\Situs} f(q) \, d\pi_\beta(q) \quad \text{a.s.}
        \end{equation}
\end{enumerate}
\end{theorem}

\begin{proof}[证明概要 Proof Sketch]
(i) 由哈密顿动力学的辛结构和蛙跳积分器的保辛性质，结合 Metropolis 校正步骤的
细致平衡条件直接可得。蛙跳积分器的 Jacobi 行列式为 1 保证了体积守恒。

(ii) 紧致流形上，正定度规条件保证存在 $\delta > 0$ 使得在任意点 $q \in \Situs$，
单步蛙跳更新的可达集包含半径为 $\delta$ 的开球。由 $\Situs$ 的紧致性和连通性，
有限步内可达整个流形。

(iii) 非周期性来自 Metropolis 拒绝步骤引入的非零拒绝概率，打破任何确定性周期。

(iv) 由 (i)-(iii) 和标准遍历定理（Birkhoff 个体遍历定理在紧致状态空间上的形式）
直接可得。详细的测度论构造参见附录。
\end{proof}

\begin{corollary}[遍历性仅需温和条件]
遍历性的充要条件仅为：(a) $\Situs$ 紧致，(b) $\Cercis \in C^2$，以及
(c) $h_{\Situs}$ 一致正定。对 SCX 审计而言，这些条件均在标准设定下自动满足：
$\Situs$ 的紧致性来自规范约束 $\sum_j g_{ij} = 0$ 和有限势能范围；
$\Cercis$ 作为二次型自然 $C^\infty$ 光滑；$h_{\Situs}$ 的正定性由审计势能面
的凸性保证。
\end{corollary}

\subsection{实践考虑：自适应步长}

在 Situs 流形上，单一全局步长 $\epsilon$ 最优性不足。我们提出自适应方案：

\begin{equation}\label{eq:adaptive}
    \epsilon_k = \epsilon_0 \cdot \min\!\left(1, \frac{\bar{\lambda}}{\lambda_{\max}(h_{\Situs}(q_k))}\right)
\end{equation}

其中 $\bar{\lambda} = \frac{1}{|\Situs|} \int_{\Situs} \Tr(h_{\Situs}(q)) \, dq$ 为
平均特征值。在高度弯曲区域（$\lambda_{\max}$ 大），步长自动减小；在平坦区域，
步长恢复至 $\epsilon_0$。

\subsection{在线审计的序贯性质}

实际 SCX 审计中，新数据持续到达：新声明发布，新专家加入评审，现有专家更新
判断。这要求审计模型能够\textbf{在线更新}——无需从头重采样，而是利用已有后验
近似高效吸收新证据。

形式化地，设 $t = 1, 2, \ldots, T$ 为审计时间步，$\D_t$ 为截止时间 $t$ 的累积
数据。目标后验分布序列为：

\begin{equation}\label{eq:posterior_sequence}
    \pi_t(E) = \Pbb(E \mid \D_t) \propto \pi_{t-1}(E) \cdot \Pbb(\D_t \setminus \D_{t-1} \mid E)
\end{equation}

序贯蒙特卡洛（SMC，也称粒子滤波）为这一序贯推断问题提供了自然框架。

\subsection{粒子滤波算法}

\begin{definition}[SMC 粒子集合 SMC Particle Ensemble]
在时间 $t$，SMC 维持一组 $K$ 个加权粒子：
\begin{equation}
    \{E_t^{(k)}, w_t^{(k)}\}_{k=1}^K,\quad \sum_{k=1}^K w_t^{(k)} = 1
\end{equation}
近似后验 $\pi_t(E) \approx \sum_{k=1}^K w_t^{(k)} \delta_{E_t^{(k)}}(E)$。
\end{definition}

\begin{algorithm-env}[SCX 在线审计 SMC 算法]
\begin{algorithmic}[1]
\State \textbf{初始化 ($t=0$):}
    $E_0^{(k)} \sim \pi_0$ (先验)，$w_0^{(k)} \leftarrow 1/K$，对 $k=1,\ldots,K$
\For{$t = 1$ \textbf{to} $T$}
    \State \textbf{重要性采样:}
        根据提议分布 $q_t(E \mid E_{t-1}^{(k)}, \D_t)$ 采样
        $\tilde{E}_t^{(k)}$
    \State \textbf{权重更新:}
        $\tilde{w}_t^{(k)} \leftarrow w_{t-1}^{(k)} \cdot
        \dfrac{\Pbb(\D_t \setminus \D_{t-1} \mid \tilde{E}_t^{(k)})
              \cdot \Pbb(\tilde{E}_t^{(k)} \mid E_{t-1}^{(k)})}
             {q_t(\tilde{E}_t^{(k)} \mid E_{t-1}^{(k)}, \D_t)}$
    \State \textbf{权重归一化:}
        $w_t^{(k)} \leftarrow \tilde{w}_t^{(k)} / \sum_{j=1}^K \tilde{w}_t^{(j)}$
    \State \textbf{计算 ESS:}
        $\ESS_t \leftarrow \big(\sum_{k=1}^K (w_t^{(k)})^2\big)^{-1}$
    \If{$\ESS_t < \alpha K$（如 $\alpha = 0.5$）}
        \State \textbf{重采样:}
            $E_t^{(k)} \sim \sum_{j=1}^K w_t^{(j)} \delta_{\tilde{E}_t^{(j)}(E)}$，
            $w_t^{(k)} \leftarrow 1/K$
    \EndIf
\EndFor
\end{algorithmic}
\end{algorithm-env}

\subsection{有效样本量下界}

\begin{theorem}[SMC 有效样本量界 ESS Bound for SCX-SMC]
\label{thm:ess_bound}
设 $\D_t$ 满足有界似然比条件：存在 $B < \infty$ 使得对所有 $E$，
\begin{equation}
    \frac{\Pbb(\D_t \setminus \D_{t-1} \mid E)}{\max_{E'} \Pbb(\D_t \setminus \D_{t-1} \mid E')} \geq \frac{1}{B}
\end{equation}
且提议分布满足 $q_t(\cdot) = \pi_{t-1}(\cdot)$（bootstrap 滤波）。则有效样本量的
期望下界为：

\begin{equation}\label{eq:ess_bound}
    \E[\ESS_t] \geq \frac{K}{B^2}
\end{equation}

特别地，$\ESS_t \geq \frac{K}{B^2 \log(1/\delta)}$ 以概率至少 $1-\delta$ 成立。
\end{theorem}

\begin{proof}
在 bootstrap 滤波 $q_t = \pi_{t-1}$ 下，非归一化权重简化为
$\tilde{w}_t^{(k)} = \Pbb(\D_t \setminus \D_{t-1} \mid \tilde{E}_t^{(k)})$。
有界似然比条件给出 $\tilde{w}_{\max} / \tilde{w}_{\min} \leq B$。
由 Cauchy-Schwarz，
$\ESS_t = \frac{(\sum w_k)^2}{\sum w_k^2} \geq \frac{K \cdot w_{\min}^2}{w_{\max}^2} \geq \frac{K}{B^2}$。
概率版本由 Markov 不等式给出。
\end{proof}

\begin{corollary}[SCX 审计的 ESS 条件]
SCX 在线审计的 ESS 退化由似然比界 $B$ 控制。当新数据与新粒子状态高度一致时
$B \approx 1$，ESS 接近 $K$；当新数据与新粒子状态剧烈冲突时 $B$ 很大，
需要增加粒子数或降低重采样阈值 $\alpha$。
\end{corollary}

\subsection{审计数据到达的粒子自适应}

SCX 审计中的关键挑战：新数据 $\D_t$ 可能来自三类源：

\begin{enumerate}[label=\textbf{类型 \arabic*:}]
    \item \textbf{新声明 (new claims):} 新声明引入新维度 $m \to m+1$，粒子空间
        扩张。处理：新维度的先验采样 + 重新加权。
    \item \textbf{新专家评审 (new reviews):} 新专家引入新规范姿态 $g_{\cdot, n+1}$。
        处理：类似类型 1，但更新 $n \to n+1$。
    \item \textbf{修正评审 (revised reviews):} 现有专家更新判断。处理：仅需
        重新加权，粒子状态不变。
\end{enumerate}

\begin{algobox}[SCX 审计数据的粒子自适应 Particle Adaptation for SCX Audit Data]
\textbf{新声明：}
\begin{enumerate}
    \item 扩展粒子维度：$E_{\text{new}} = [E_{\text{old}}, g_{\text{new},1}, \ldots, g_{\text{new},n}]$
    \item 从先验采样新维度：$g_{\text{new},j} \sim \Ncal(0, \sigma^2)$
    \item 保持旧权重不变（未涉及旧状态）
\end{enumerate}

\textbf{新专家评审：}
\begin{enumerate}
    \item 扩展粒子维度：$E_{\text{new}} = [E_{\text{old}}, g_{1,\text{new}}, \ldots, g_{m,\text{new}}]$
    \item 从先验或基于已有专家的相关性采样
    \item 重新权重：考虑新专家与已有专家的一致性
\end{enumerate}

\textbf{修正评审：}
\begin{enumerate}
    \item 粒子状态不变
    \item 仅重新计算权重：$w_k \propto w_k^{\text{old}} \cdot \Pbb(\text{修正} \mid E^{(k)})$
    \item 如 ESS 不足，在受影响的维度子集上重采样
\end{enumerate}
\end{algobox}

\subsection{收敛性分析}

\begin{theorem}[SCX-SMC 均方误差界 MSE Bound]
设 $\pi_t$ 为 $t$ 时刻的真实后验，$\hat{\pi}_t^K$ 为 $K$ 个粒子的 SMC 近似。
则在温和正则条件下，对任意有界检验函数 $\phi$：

\begin{equation}\label{eq:mse_bound}
    \E\!\left[ \Big( \int \phi(E) \hat{\pi}_t^K(dE) - \int \phi(E) \pi_t(dE) \Big)^2 \right] \leq \frac{C_t \|\phi\|_\infty^2}{K}
\end{equation}

其中 $C_t$ 依赖于 $B$（似然比界）和对数时间 $t$：
$C_t = \Ocal(B^{2t} \log t)$。对于 SCX 审计，$t$ 通常为 $\Ocal(1)$ 到
$\Ocal(10^2)$，$B \approx 2$–$5$，此时 $C_t$ 在可行范围内。
\end{theorem}

\subsection{Cercis 校准问题}

Cercis 分数 $\Cercis(E)$ 作为审计"不协调度"度量，其\textbf{校准}（calibration）
至关重要：对给定审计问题，$\Cercis(E)$ 的绝对数值应能够区分"良好共识"与
"系统性偏差"。校准的核心是确定配分函数 $Z(\beta)$ 及其导数——

\begin{equation}\label{eq:free_energy}
    F(\beta) = -\frac{1}{\beta} \log Z(\beta),\quad
    \frac{dF}{d\beta} = \E_{\pi_\beta}[\Cercis(E)]
\end{equation}

从高信噪比（$\beta_{\text{high}}$，容易审计）到低信噪比（$\beta_{\text{low}}$，
困难审计）的\textbf{自由能差} $\Delta F = F(\beta_{\text{low}}) - F(\beta_{\text{high}})$
提供了 Cercis 的校准基准。

\subsection{热力学积分的数学框架}

自由能差通过热力学积分（Thermodynamic Integration, TI）计算：

\begin{equation}\label{eq:ti}
    \Delta F = \int_{\beta_{\text{high}}}^{\beta_{\text{low}}} \E_{\pi_\beta}[\Cercis(E)] \, d\beta
\end{equation}

在实际计算中，在离散温度网格 $\beta_0, \beta_1, \ldots, \beta_R$ 上
（$\beta_0 = \beta_{\text{high}}, \beta_R = \beta_{\text{low}}$），每个
$\beta_r$ 运行 MCMC 获得样本 $\{E^{(r,k)}\}_{k=1}^{N_r}$，然后数值积分：

\begin{equation}\label{eq:ti_discrete}
    \Delta F \approx \sum_{r=0}^{R-1} \frac{(\beta_{r+1} - \beta_r)}{2}
    \Big( \bar{\Cercis}_r + \bar{\Cercis}_{r+1} \Big)
\end{equation}

其中 $\bar{\Cercis}_r = \frac{1}{N_r} \sum_{k=1}^{N_r} \Cercis(E^{(r,k)})$ 为
温度 $\beta_r$ 下的样本平均。

\subsection{温度调度：从易到难}

\begin{definition}[信噪比-温度对应 SNR-Temperature Correspondence]
审计问题的信噪比与逆温度 $\beta$ 存在单调对应关系：

\begin{itemize}
    \item \textbf{高 SNR (高 $\beta$):} 审计信号强，$\Cercis$ 表面深谷明确。
        少量样本即可准确定位共识点。例如：事实性声明（物理定律、数学定理）。
    \item \textbf{低 SNR (低 $\beta$):} 审计信号弱，$\Cercis$ 表面浅而平坦。
        需要大量样本探索。例如：价值判断（美学、伦理）、高度不确定预测。
\end{itemize}

温度调度 $\beta_0 > \beta_1 > \cdots > \beta_R$ 实现从"易"到"难"的审计
退火。
\end{definition}

\begin{keyeq}
\textbf{SCX 审计的热力学恒等式：}

\begin{align}
    \Cercis_{\text{cal}} &\equiv \Delta F = F(\beta_{\text{low}}) - F(\beta_{\text{high}})
        \label{eq:cercis_cal} \\
    &= \int_{\beta_{\text{high}}}^{\beta_{\text{low}}} \E_{\pi_\beta}[\Cercis(E)] \, d\beta
        \label{eq:cercis_ti} \\
    &= \int_{\beta_{\text{high}}}^{\beta_{\text{low}}} \frac{\int_{\Situs} \Cercis(E) e^{-\beta \Cercis(E)} dE}
        {\int_{\Situs} e^{-\beta \Cercis(E)} dE} \, d\beta \label{eq:cercis_explicit}
\end{align}

$\Cercis_{\text{cal}}$ 是 Cercis 分数的校准常量：审计问题的"内在难度"。
\end{keyeq}

\subsection{温度网格优化}

温度点的选择对 TI 精度至关重要。我们采用两种优化策略：

\begin{theorem}[最优温度网格 Optimal Temperature Grid]
\label{thm:temp_grid}
在总计算预算 $B = \sum_r N_r$ 下，最小化 $\Delta F$ 的均方误差要求温度点按
\emph{热容密度} 分布：

\begin{equation}\label{eq:temp_spacing}
    \Delta\beta_r \propto \frac{1}{\sqrt{\Var_{\pi_{\beta_r}}[\Cercis(E)]}}
\end{equation}

其中 $\Var_{\pi_{\beta_r}}[\Cercis(E)]$ 为 $\Cercis$ 在温度 $\beta_r$ 下的方差
（热容 $C_V(\beta_r) = \beta_r^2 \Var_{\pi_\beta}[\Cercis]$）。在相变区域，
热容激增，网格应自动加密。
\end{theorem}

\begin{proof}
由 $\delta$-方法，$\hat{F}$ 的方差近似为
$\Var[\hat{F}] \approx \sum_r (\Delta\beta_r)^2 \cdot \Var[\bar{\Cercis}_r]$。
在约束 $\sum_r N_r = B$ 和 $\sum_r \Delta\beta_r = \beta_{\text{high}} - \beta_{\text{low}}$
下，利用拉格朗日乘数法最小化方差，得最优分布
$\Delta\beta_r \propto 1/\sqrt{\Var[\Cercis]_r}$。
\end{proof}

\subsection{SCX 审计中的相变现象}

\begin{conjecture}[审计相变 Audit Phase Transition]
在临界温度 $\beta_c = \frac{1}{T_c}$ 处，SCX 审计经历一次\textbf{二级相变}：

\begin{itemize}
    \item $\beta > \beta_c$（低温）：系统呈现有序相——专家实质上达成共识
        ($\sum_j g_{ij} \approx 0$)。Cercis 热容 $C_V$ 小。
    \item $\beta < \beta_c$（高温）：系统呈现无序相——专家判断随机分散。
        Cercis 热容 $C_V$ 大。
    \item $\beta \approx \beta_c$：热容在热力学极限下发散 $C_V \sim |\beta - \beta_c|^{-\alpha}$。
\end{itemize}

$\beta_c$ 由审计问题的性质决定：专家组规模、声明的模糊性、数据的信噪比。
临界指数 $\alpha$ 反映了审计问题的"维数"。
\end{conjecture}

\begin{remark}
审计相变的存在意味着：对于足够困难的审计问题（$\beta < \beta_c$），
即使调用无限计算资源，也无法唯一确定专家共识——审计结果\textbf{本征地}不确定。
这是 SCX 审计的\textbf{不确定性原理}的统计力学形式。
\end{remark}

\subsection{TI 的实用实现}

\begin{algorithm-env}[SCX 审计的热力学积分 TI for SCX Audit]
\begin{algorithmic}[1]
\State \textbf{输入:} 温度范围 $[\beta_{\text{low}}, \beta_{\text{high}}]$，
        初始网格点数 $R_0$，每温度样本数 $N$，自适应公差 $\tau$
\State \textbf{初始化:} 设置温度网格 $\beta_0 = \beta_{\text{high}}, \ldots, \beta_{R_0} = \beta_{\text{low}}$
\For{$r = 0$ \textbf{to} $R_0$}
    \State 在温度 $\beta_r$ 运行 Situs-HMC（算法 1.1），获得 $N$ 个样本
    \State 计算 $\bar{\Cercis}_r$ 和 $\sigma_r^2 = \Var[\Cercis]_r$
\EndFor
\State \textbf{自适应加密:}
\While{$\max_r \sigma_r^2 \cdot (\Delta\beta_r)^2 > \tau$}
    \State 在 $\sigma_r^2$ 最大的区间中点插入新温度点
    \State 在新温度点运行 Situs-HMC
    \State 更新 $\bar{\Cercis}_r$ 和 $\sigma_r^2$
\EndWhile
\State \textbf{Trapezoidal 积分:}
    $\Cercis_{\text{cal}} \leftarrow \sum_{r=0}^{R-1} \frac{\beta_{r+1} - \beta_r}{2} (\bar{\Cercis}_r + \bar{\Cercis}_{r+1})$
\State \textbf{返回:} $\Cercis_{\text{cal}}$ 及估计标准误差
\end{algorithmic}
\end{algorithm-env}

\section{副本交换理论}

\subsection{多模态挑战与副本交换}

SCX 审计中，多专家共识面临一个根本挑战：专家配置空间 $\Situs$ 可能存在多个
局部最优（多种可能的共识配置）。例如：

\begin{enumerate}[label=(\arabic*)]
    \item \textbf{模棱两可的声明：} 两种互斥解读均有优秀专家支持，导致两个分离
        的 $\Cercis$ 低谷
    \item \textbf{专家派系：} 不同方法论阵营的专家形成聚类，各聚类内部高度一致
        但聚类之间不一致
    \item \textbf{时间演化中的亚稳态：} 共识随时间演化，链可能陷入过去的共识
        模式（亚稳态）
\end{enumerate}

单一温度下的 HMC/SMC 链可能被困在某个 $\Cercis$ 低谷中，无法探索整个配置空间。
\textbf{副本交换蒙特卡洛}（Replica Exchange Monte Carlo, REX，也称并行回火
Parallel Tempering）通过在不同温度运行多个副本（replica）并周期性交换状态来
解决多模态问题。

\subsection{REX 的数学框架}

设 $R$ 个副本，温度 $T_1 < T_2 < \cdots < T_R$（逆温度
$\beta_1 > \beta_2 > \cdots > \beta_R$）。副本 $r$ 在温度 $T_r$ 下独立演化，
目标分布 $\pi_{\beta_r}(E) \propto \exp(-\beta_r \Cercis(E))$。

\textbf{联合分布：} 副本之间统计独立，因此 $R$ 个副本的整体分布为乘积：

\begin{equation}\label{eq:rex_joint}
    \Pi(E^{(1)}, \ldots, E^{(R)}) = \prod_{r=1}^R \pi_{\beta_r}(E^{(r)})
        \propto \prod_{r=1}^R \exp(-\beta_r \Cercis(E^{(r)}))
\end{equation}

\textbf{交换操作：} 相邻副本 $r$ 和 $r+1$ 的交换以 Metropolis 概率接受：

\begin{equation}\label{eq:swap_prob}
    A_{\text{swap}} = \min\!\Bigg(1,
        \frac{\pi_{\beta_r}(E^{(r+1)}) \cdot \pi_{\beta_{r+1}}(E^{(r)})}
             {\pi_{\beta_r}(E^{(r)}) \cdot \pi_{\beta_{r+1}}(E^{(r+1)})}
    \Bigg)
\end{equation}

化简得：

\begin{equation}\label{eq:swap_simplified}
    A_{\text{swap}} = \min\!\Big(1,
        \exp\!\big( (\beta_r - \beta_{r+1}) \cdot [\Cercis(E^{(r)}) - \Cercis(E^{(r+1)})] \big)
    \Big)
\end{equation}

\begin{keyeq}
\textbf{副本交换接受率：}
\begin{equation}
    A_{\text{swap}}(r \leftrightarrow r+1) =
    \min\!\Bigg(1, \exp\!\Big(
        \underbrace{(\beta_r - \beta_{r+1})}_{\Delta\beta > 0}
        \cdot
        \underbrace{[\Cercis(E^{(r)}) - \Cercis(E^{(r+1)})]}_{\Delta \Cercis}
    \Big)\Bigg)
\end{equation}

当高温副本（更高的 $\Cercis$）与低温副本（更低的 $\Cercis$）交换时，
$\Delta\Cercis > 0$，接受率高。这实现了\textbf{低温副本的全局探索}：
高温副本广泛探索配置空间，通过交换将发现的模式传递给低温副本。
\end{keyeq}

\subsection{温度阶梯设计}

温度阶梯的设计决定 REX 的效率。核心权衡：

\begin{definition}[温度阶梯的最优条件 Optimal Temperature Ladder]
设 $R$ 个副本，温度 $T_1 < \cdots < T_R$。最优阶梯满足：

\begin{enumerate}[label=(\roman*)]
    \item \textbf{均匀交换率：} 所有相邻对的交换接受率近似相等
        $\bar{A}_{r,r+1} \approx \text{const}$
    \item \textbf{充分覆盖：} $T_R$ 足够高，使得副本 $R$ 能够自由遍历配置空间
    \item \textbf{精度保持：} $T_1$ 足够低，使得副本 1 的采样集中于 $\Cercis$ 低谷
\end{enumerate}
\end{definition}

\begin{proposition}[最优温度间隔 Optimal Spacing]
在假设 $\Cercis(E)$ 在温度 $T$ 下近似正态分布的前提下，期望交换接受率约为
$\bar{A} \approx 2\Phi\!\big(-\frac{\Delta\beta \cdot \sigma_\Cercis}{2}\big)$
（其中 $\Phi$ 为标准正态 CDF）。为维持均匀接受率 $\bar{A}$，温度点应满足：

\begin{equation}\label{eq:temp_ladder}
    T_{r+1} \approx T_r \cdot \left(1 + \frac{c}{\sqrt{n_{\text{eff}}}}\right)
\end{equation}

其中 $c = -2\Phi^{-1}(\bar{A}/2)$，$n_{\text{eff}}$ 为系统的有效自由度。
对于典型 SCX 审计 ($n_{\text{eff}} \sim 10$–$100$，$\bar{A} \approx 0.3$)，
$T_{r+1}/T_r \approx 1.15$–$1.5$。
\end{proposition}

\subsection{SCX 多专家共识的 REX 算法}

\begin{algorithm-env}[SCX 多专家共识的副本交换 REX for SCX Multi-Expert Consensus]
\begin{algorithmic}[1]
\State \textbf{输入:} 副本数 $R$，温度阶梯 $\{T_r\}_{r=1}^R$，
        每温度 HMC 步数 $L$，交换间隔 $S$，总迭代数 $N_{\text{iter}}$
\State \textbf{初始化:} 随机初始化 $E_0^{(r)} \sim \pi_{\beta_r}$ 对 $r=1,\ldots,R$
\For{$n = 1$ \textbf{to} $N_{\text{iter}}$}
    \State \textbf{本地演化:}
    \For{$r = 1$ \textbf{to} $R$} \Comment{各副本独立演化}
        \State $E_n^{(r)} \leftarrow$ 对 $E_{n-1}^{(r)}$ 运行 $L$ 步 Situs-HMC
            (温度 $T_r$)
    \EndFor
    \If{$n \bmod S = 0$} \Comment{每隔 $S$ 步交换}
        \For{$r = 1$ \textbf{to} $R-1$}
            \State 计算 $\Delta = (\beta_r - \beta_{r+1}) \cdot
                [\Cercis(E_n^{(r)}) - \Cercis(E_n^{(r+1)})]$
            \State $A \leftarrow \min(1, \exp(\Delta))$
            \If{$\text{Uniform}(0,1) < A$}
                \State 交换 $E_n^{(r)} \leftrightarrow E_n^{(r+1)}$
            \EndIf
        \EndFor
    \EndIf
\EndFor
\State \textbf{共识诊断（对低温副本 $r=1$ 的样本）:}
\State 计算各声明 $i$ 的 $\hat{g}_{ij} = \text{mean}(\{E^{(1)}_n\})$ 和
    $\hat{\sigma}_{ij} = \text{std}(\{E^{(1)}_n\})$
\State 识别多模态：对各声明 $i$，检验 $\{g_{ij}\}$ 的分布是否为多峰的
    （如 Hartigan dip test）
\State \textbf{返回:} 共识估计、不确定度、多模态标记
\end{algorithmic}
\end{algorithm-env}

\subsection{REX 的遍历性加速}

\begin{theorem}[REX 的混合加速 Mixing Acceleration by REX]
\label{thm:rex_mixing}
设单一温度 $T$ 的 HMC 链的混合时间为 $\tau_{\text{mix}}(T)$。则在包含 $R$ 个副本
且温度范围 $[T_{\min}, T_{\max}]$ 的 REX 方案下，低温副本（$T_{\min}$）的
有效混合时间为：

\begin{equation}\label{eq:rex_mixing_time}
    \tau_{\text{mix}}^{\text{REX}}(T_{\min}) \leq
    \tau_{\text{mix}}(T_{\min}) \cdot \min_{r} \left\{
        \frac{1}{\bar{A}_{r,r+1}} \cdot
        \frac{\tau_{\text{mix}}(T_r)}{\tau_{\text{mix}}(T_{\min})}
    \right\}
\end{equation}

其中 $\bar{A}_{r,r+1}$ 为相邻副本的平均交换接受率。当 $T_R \gg T_{\text{crit}}$
且 $\bar{A}_{r,r+1} \approx 0.2$–$0.5$ 时，加速因子可达 $\Ocal(10^2)$–
$\Ocal(10^4)$。
\end{theorem}

\begin{remark}
REX 的加速本质上来自\textbf{温度加速的隧道效应}：高温副本迅速穿越能量壁垒，
通过交换将"知识"传递至低温副本。对于 SCX 审计，这对应于：用宽松标准（高 $T$）
快速扫描可能的共识配置，然后用严格标准（低 $T$）精确校准最有希望的配置。
\end{remark}

\subsection{为什么收敛诊断对审计至关重要}

SCX 审计的蒙特卡洛方法涉及一个根本问题：\textbf{我们怎么知道链已经"足够好"地
收敛了？} 未收敛的链给出的审计结果可能是：

\begin{itemize}
    \item \textbf{系统性偏差：} 链仍被初始条件影响，未探索整个后验
    \item \textbf{过度自信：} 尚未访问的低概率区域可能导致低估不确定度
    \item \textbf{多模态遗漏：} 单一链可能仅发现多模态后验中的一个模态
\end{itemize}

\textbf{审计收敛标准：} 一个 SCX 审计是\textbf{收敛的}，当且仅当对所有规范参数
$g_i$（即声明 $i$ 的 $\sum_j g_{ij}$，或等价地，各专家的 $g_{ij}$ 质量加权的
综合规范姿态），Gelman-Rubin 统计量 $\Rhat < 1.01$。

\subsection{Gelman-Rubin 诊断}

\begin{definition}[Gelman-Rubin $\Rhat$ 统计量]
对于 $M$ 条链（$M \geq 2$），每条链 $N$ 个后燃烧期样本。对参数 $\theta$，
定义：

\vspace{4pt}
\noindent\textbf{链内方差 (within-chain variance):}
\begin{equation}\label{eq:W}
    W = \frac{1}{M} \sum_{m=1}^M s_m^2,\quad
    s_m^2 = \frac{1}{N-1} \sum_{n=1}^N (\theta_{mn} - \bar{\theta}_m)^2
\end{equation}

\noindent\textbf{链间方差 (between-chain variance):}
\begin{equation}\label{eq:B}
    B = \frac{N}{M-1} \sum_{m=1}^M (\bar{\theta}_m - \bar{\theta})^2,\quad
    \bar{\theta} = \frac{1}{M} \sum_{m=1}^M \bar{\theta}_m
\end{equation}

\noindent\textbf{边际后验方差估计:}
\begin{equation}\label{eq:var_plus}
    \widehat{\Var}^+(\theta) = \frac{N-1}{N} W + \frac{1}{N} B
\end{equation}

\noindent\textbf{$\Rhat$ 统计量:}
\begin{equation}\label{eq:Rhat}
    \Rhat = \sqrt{\frac{\widehat{\Var}^+(\theta)}{W}}
\end{equation}
\end{definition}

\begin{theorem}[Gelman-Rubin 收敛判据的审计形式]
\label{thm:rhat_criterion}
如果 $M$ 条链均已充分混合（即 $\widehat{\Var}^+$ 和 $W$ 均已稳定），则
$\Rhat \approx 1$。正式的审计收敛判据为：

\begin{equation}\label{eq:audit_converged}
    \text{审计收敛} \quad \Longleftrightarrow \quad
    \max_{i} \Rhat(g_i) < 1.01
\end{equation}

其中 $\max_{i}$ 取遍所有独立规范参数。
\end{theorem}

\begin{remark}
$\Rhat < 1.01$ 是比标准贝叶斯分析中常用的 $\Rhat < 1.1$ 更严格的标准。
SCX 审计要求更高的精度，因为：(a) 审计结果可能影响重大决策；
(b) 规范参数 $g_i$ 之间高度相关，单一参数的微小偏差可能通过约束
$\sum g = 0$ 传播。
\end{remark}

\subsection{有效样本量}

\begin{definition}[有效样本量 Effective Sample Size]
对于具有自相关 $\rho_k = \Corr(\theta_t, \theta_{t+k})$ 的单条链，
有效样本量为：

\begin{equation}\label{eq:ess_def}
    \ESS = \frac{N}{1 + 2 \sum_{k=1}^\infty \rho_k}
\end{equation}

在实际中，使用截断自相关和：
$\ESS \approx N / (1 + 2 \sum_{k=1}^{K_{\max}} \hat{\rho}_k)$，
其中 $K_{\max}$ 取 $\hat{\rho}_k$ 降为低于噪声水平的最小 $k$。
\end{definition}

\begin{proposition}[SCX 审计的最小 ESS 要求]
对于可靠的 SCX 审计推断，我们要求：
\begin{enumerate}[label=(\arabic*)]
    \item 对于点估计（$\hat{g}_{ij}$ 的期望）：$\ESS \geq 100$
    \item 对于不确定度估计（$\widehat{\Var}[g_{ij}]$）：$\ESS \geq 400$
    \item 对于尾部概率（如异常值检测）：$\ESS \geq 1000$
\end{enumerate}
这些阈值基于：后验均值估计精度 $\Ocal(\ESS^{-1/2})$，
方差估计精度 $\Ocal(\ESS^{-1/2})$，以及尾部分位数的更高方差。
\end{proposition}

\subsection{多链初始化策略}

\begin{algobox}[SCX 审计多链初始化 Multi-Chain Initialization for SCX Audit]
为确保链的独立性（从而 $\Rhat$ 的有效性），我们采用\textbf{过度离散初始化}：

\begin{enumerate}
    \item \textbf{链 1（共识先验）：} $g_{ij} \sim \Ncal(0, 0.1)$——预期共识
    \item \textbf{链 2（分歧先验）：} $g_{ij} \sim \Ncal(0, 1.0)$——大分歧可能
    \item \textbf{链 3（偏斜先验）：} $g_{ij} \sim \text{SkewNormal}(0, 0.5, \alpha=3)$
        ——系统性正偏
    \item \textbf{链 4（反偏斜先验）：} $g_{ij} \sim \text{SkewNormal}(0, 0.5, \alpha=-3)$
        ——系统性负偏
    \item \textbf{链 5-8（随机先验）：} $g_{ij} \sim \Ncal(0, \sigma_j^2)$ 其中
        $\sigma_j \sim \text{Exp}(1)$——随机方差
\end{enumerate}

所有 $M$ 条链独立运行，燃烧期后比较 $\Rhat$。
\end{algobox}

\subsection{Rank 归一化与分位点收敛}

除了 $\Rhat$ 和 ESS，我们使用\textbf{Rank 归一化}诊断：

\begin{definition}[Rank 归一化诊断 Rank-Normalized Diagnostic]
将 $M$ 条链的后燃烧期样本合并，对参数 $\theta$ 进行 Rank 变换：
$r_{mn} = \text{rank}(\theta_{mn})$（范围 $1$ 到 $MN$），然后标准化到
$[0,1]$：$z_{mn} = (r_{mn} - 0.5) / (MN)$。

如果各链混合良好，$z_{mn}$ 应在 $[0,1]$ 上均匀分布，且链间分布无显著差异
（通过 Kolmogorov-Smirnov 检验）。
\end{definition}

\begin{theorem}[审计收敛的综合诊断综合指标 Composite Diagnostic]
一个 SCX 审计是\textbf{全面收敛的}（fully converged）当且仅当：

\begin{equation}\label{eq:composite_diag}
    \begin{cases}
        \max_i \Rhat(g_i) < 1.01 & \text{(Gelman-Rubin)} \\[4pt]
        \min_i \ESS(g_i) > 400 & \text{(有效样本量)} \\[4pt]
        \max_i \text{KS-}p\text{-value} > 0.05 & \text{(Rank 均匀性)} \\[4pt]
        \text{无趋势} & \text{(迹线图目视检查)}
    \end{cases}
\end{equation}

同时满足以上四项条件的审计结果才可被视为可靠。
\end{theorem}

\subsection{迹线图诊断与早期停止}

\begin{algorithm-env}[自适应早期停止 Adaptive Early Stopping]
\begin{algorithmic}[1]
\State \textbf{输入:} 目标参数 $\theta$，最小迭代 $N_{\min}$，窗口大小 $w$
\State 初始化：$n \leftarrow 0$，历史 $\Rhat$ 序列 $\mathcal{R} \leftarrow []$
\While{$n < N_{\min}$ \textbf{或} $\text{未收敛}$}
    \State 对所有 $M$ 条链运行附加的 $L$ 步
    \State $n \leftarrow n + L$
    \State 计算最后 $w$ 个样本的 $\Rhat(\theta)$
    \State $\mathcal{R}.\text{append}(\Rhat(\theta))$
    \If{$|\mathcal{R}| \geq 3$}
        \State 拟合 $\Rhat_t = a \cdot t^{-b} + 1$ 对最后 3 个点
        \State 如果 $a < 0.01$ 且 $b > 0$（即 $\Rhat$ 已趋于 1），
            \textbf{break}
    \EndIf
\EndWhile
\State \textbf{返回:} 收敛状态，$\Rhat$ 值，ESS
\end{algorithmic}
\end{algorithm-env}

\subsection{完整审计流水线}

以上五种方法构成一个完整的 SCX 审计流水线：

\begin{enumerate}[label=\textbf{阶段 \arabic*:},leftmargin=*]
    \item \textbf{校准 (TI):} 使用热力学积分确定 Cercis 校准常量
        $\Cercis_{\text{cal}}$，评估审计问题的内在难度。
    \item \textbf{初始化 (REX):} 使用副本交换在广泛温度范围探索配置空间，
        识别所有显著模态。
    \item \textbf{精确采样 (HMC):} 利用 Situs 度规进行黎曼 HMC 精确采样，
        以目标逆温度 $\beta_{\text{target}}$ 从 Boltzmann 后验中生成样本。
    \item \textbf{在线更新 (SMC):} 随新数据到达，序贯更新粒子，维持近似后验。
        使用 HMC 样本作为 SMC 的提议分布。
    \item \textbf{收敛验证 ($\Rhat$/ESS):} 持续监测 $\Rhat$、ESS 和 Rank
        均匀性，确保审计结果可靠。
\end{enumerate}
