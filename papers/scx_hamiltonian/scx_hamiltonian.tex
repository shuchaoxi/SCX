\begin{abstract}
This paper establishes a deep correspondence between the SCX multi-expert auditing framework and the statistical mechanics of disordered systems. Core insight: each expert corresponds to a sample from a Gibbs distribution defined by the Hamiltonian \(H(\theta) = \loss(\theta) + \log Z\) where \(Z = \int \exp(-H(\theta)) d\theta\) is the partition function. We prove: (1) In the SCX framework, consensus emerges as \(M \to \infty\) at a well-defined critical temperature; (2) The \(n\)-replica overlap matrix \(Q_{ab}\) maps to expert-expert correlations, establishing the isomorphism; (3) The Parisi hierarchical RSB scheme classifies expert disagreement structures---RS (consensus), 1RSB (clustered disagreement), and fRSB (full hierarchical disagreement); (4) Central result: the minimum number of experts \(M_{\min}\) required for \(\varepsilon\)-complete coverage is \(M_{\min} \propto \exp(\Sigma(e_c))\), where \(\Sigma(e)\) is the complexity of the energy landscape. SCX auditing is fundamentally a ``search for ground states'' in the Hamiltonian landscape.
\end{abstract}

\section{Introduction}
% ======================================================================
A central question in AI auditing: when multiple experts evaluate the same data, how do we know whether their consensus is genuine or merely the byproduct of shared biases? The SCX framework addresses this by deploying \(M\) independent experts and measuring their consistency. But how many experts are needed? What is the ``minimum number of experts'' for reliable auditing? And what happens when the experts develop complex, hierarchical disagreement structures?

This paper adopts a statistical mechanics perspective. We show that the SCX multi-expert auditing protocol has a direct mathematical correspondence with spin glass theory, where the core organizing quantity is the \textbf{Hamiltonian}.

\begin{definition}[Hamiltonian]
For a neural network parameterized by \(\theta \in \R^N\) with loss function \(\loss(\theta)\) (the energy), the Hamiltonian is:
\begin{equation}\label{eq:hamiltonian}
H(\theta) = \loss(\theta) + \frac{1}{\beta} \log Z(\beta)
\end{equation}
\begin{equation}\label{eq:partition}
Z(\beta) = \int_{\R^N} \exp\pqty{-\beta \loss(\theta)} d\theta
\end{equation}
where \(\beta = 1/T\) is the inverse temperature, controlling the ``sharpness'' of the distribution.
\end{definition}

The loss \(\loss(\theta)\) defines the \textbf{energy landscape}, and \(Z(\beta)\) counts the accessible states. As \(\beta \to \infty\) (zero temperature), the distribution collapses to global minima; at high temperature, it explores broadly.

\begin{enumerate}[label=(\arabic*)]
    \item The SCX multi-expert protocol: \(M\) experts each sample from the Gibbs distribution at their own temperatures (reflecting different training conditions).
    \item We prove that SCX consensus corresponds to \textbf{spontaneous magnetization} at low temperature, with a phase transition.
    \item The replica overlap matrix \(Q_{ab}\) maps to expert-expert correlations, providing a rigorous isomorphism, and we prove the Parisi RSB classification for expert disagreement.
    \item \textbf{Central Theorem (Minimum Number of Experts):} Using the complexity \(\Sigma(e)\) from the Kac-Rice formula, we derive \(M_{\min} \geq C \cdot \exp(\Sigma(e_c))\). This is the ``search for a needle in a haystack'' complexity.
\end{enumerate}

\section{The Hamiltonian Formalism}
% ======================================================================
\subsection{Gibbs Distribution and Free Energy}
Consider a loss landscape \(\loss: \R^N \to \R\) over parameter space. The dimension \(N\) is typically very large. The Gibbs distribution at inverse temperature \(\beta\) is:
\begin{equation}\label{eq:gibbs}
P_\beta(\theta) = \frac{1}{Z(\beta)} \exp\pqty{-\beta \loss(\theta)}
\end{equation}
where \(\beta > 0\) controls the concentration around minima.

\begin{definition}[Free Energy]
\begin{equation}\label{eq:free_energy}
F(\beta) = -\frac{1}{\beta} \log Z(\beta) = -\frac{1}{\beta} \log \int_{\R^N} \exp\pqty{-\beta \loss(\theta)} d\theta
\end{equation}
As \(N \to \infty\), the free energy density is \(f(\beta) = \lim_{N\to\infty} F(\beta)/N\).
\end{definition}

The derivatives of \(\beta F\) with respect to \(\beta\) yield thermodynamic averages:
\begin{equation}
\avg{\loss}_\beta = \frac{\partial (\beta F)}{\partial \beta}, \quad \avg{(\loss - \avg{\loss})^2}_\beta = -\frac{\partial^2 (\beta F)}{\partial \beta^2}
\end{equation}

\subsection{Critical Points and the Complexity}
Critical points satisfy \(\nabla_\theta \loss(\theta) = 0\). Their stability is determined by the Hessian:
\begin{equation}
\mathcal{H}_{ij}(\theta) = \frac{\partial^2 \loss(\theta)}{\partial \theta_i \partial \theta_j}
\end{equation}

\begin{definition}[Complexity]
Let \(\mathcal{N}_N(e)\) be the number of critical points at energy density \(e\). The complexity is:
\begin{equation}\label{eq:complexity}
\Sigma(e) = \lim_{N \to \infty} \frac{1}{N} \log \mathcal{N}_N(e)
\end{equation}
Similarly, \(\Sigma_0(e) = \lim_{N\to\infty} \frac{1}{N}\log \mathcal{N}_N^{(0)}(e)\) counts only stable minima (positive-definite Hessian).
\end{definition}

\begin{theorem}[Kac-Rice Formula]\label{thm:kac_rice}
For random Gaussian landscapes, the expected number of critical points at energy density \(e\) is given by the Kac-Rice formula:
\begin{equation}\label{eq:kac_rice}
\Sigma(e) = \lim_{N\to\infty} \frac{1}{N} \log \E_\theta\bqty{|\det \mathcal{H}(\theta)| \cdot \delta(\nabla \loss(\theta)) \cdot \delta(\loss(\theta) - N e)}
\end{equation}
The stable-minima complexity \(\Sigma_0(e)\) additionally requires \(\mathcal{H}(\theta) \succeq 0\) (positive-definiteness).
\end{theorem}

\subsection{Quenched Disorder}
The training dataset \(\D = \{(x_i, y_i)\}_{i=1}^{|\D|}\) acts as \textbf{quenched disorder}---frozen randomness that defines the energy landscape. This is the core analogy with spin glasses.

\begin{definition}[Quenched and Annealed Free Energies]
Given dataset \(\D\), the quenched free energy is:
\begin{equation}\label{eq:quenched}
F_q(\beta) = -\frac{1}{\beta} \E_\D\bqty{\log Z_\D(\beta)}
\end{equation}
where \(\E_\D\) averages over datasets. The annealed free energy is:
\begin{equation}
F_a(\beta) = -\frac{1}{\beta} \log \E_\D\bqty{Z_\D(\beta)}
\end{equation}
By Jensen's inequality, \(F_q \geq F_a\), and the difference \(F_q - F_a\) measures the complexity of the landscape.
\end{definition}

\section{SCX Experts as Gibbs Samplers}
% ======================================================================
\subsection{Experts as Thermal Samples}
\begin{definition}[SCX Expert as Gibbs Sample]
In the SCX framework, expert \(\mathcal{E}_\alpha\) (with \(\alpha = 1, 2, \ldots, M\)) produces a configuration \(\theta_\alpha\) sampled from:
\begin{equation}\label{eq:expert_sample}
\theta_\alpha \sim P_{\beta_\alpha}(\theta) = \frac{1}{Z(\beta_\alpha)} \exp\pqty{-\beta_\alpha \loss_\alpha(\theta)}
\end{equation}
\begin{itemize}
    \item \(\beta_\alpha = 1/T_\alpha\) is the expert's inverse temperature, reflecting ``rigor'' or ``sharpness'';
    \item \(\loss_\alpha(\theta)\) is the expert-specific loss---training data, architecture, initialization, all encoded as ``disorder.''
\end{itemize}
\end{definition}

\textbf{Key intuition:} Experts at low temperature (high \(\beta\)) converge to ``sharp'' minima---they are confident but may overfit. Experts at high temperature (low \(\beta\)) explore broadly---they are diverse but less precise. This is \textbf{parallel tempering} applied to auditing.

\subsection{Coupled Experts and the Joint Distribution}
In practice, experts are trained on the same task. Let the shared dataset be \(\D\):
\begin{equation}\label{eq:coupled_loss}
\loss_\alpha(\theta) = \loss_0(\theta) + \delta \loss_\alpha(\theta)
\end{equation}
where \(\loss_0(\theta)\) is the common loss (from shared data \(\D\)), and \(\delta \loss_\alpha(\theta)\) is expert-specific variation (different initializations, architectures, hyperparameters). As \(\delta \loss_\alpha \to 0\) with equal \(\beta_\alpha = \beta\), experts become \textbf{perfectly correlated}.

\begin{proposition}[Joint Expert Distribution]\label{prop:joint}
Under the coupled-loss model, the joint distribution of \(M\) experts is:
\begin{equation}\label{eq:joint_dist}
P^{(M)}(\theta_1, \ldots, \theta_M) = \frac{1}{Z_M} \exp\pqty{-\sum_{\alpha=1}^{M} \beta_\alpha \loss_0(\theta_\alpha) + \ldots}
\end{equation}
where \(Z_M = \int \exp(-\sum_\alpha \beta_\alpha \loss_0(\theta_\alpha)) \prod_\alpha d\theta_\alpha\). This is the \(n \to 0\) replica limit with \(M\) replicas.
\end{proposition}

\subsection{The Expert Spin Variable and Order Parameter}
For a given input \(x\), expert \(\alpha\) produces a binary prediction (agree/disagree):
\begin{equation}\label{eq:expert_spin}
\sigma_\alpha(x) = \sgn(f_\alpha(x)) \in \{-1, +1\}
\end{equation}

\begin{definition}[SCX Magnetization --- Order Parameter]
The magnetization of \(M\) experts on input \(x\) is:
\begin{equation}\label{eq:order_param}
m_M(x) = \frac{1}{M} \sum_{\alpha=1}^{M} \sigma_\alpha(x)
\end{equation}
\(m_M(x)\) ranges in \([-1, 1]\). \(|m_M(x)|\) measures ``consensus'': \(|m_M| = 1\) means unanimous agreement, \(|m_M| \approx 0\) means maximal disagreement.

Analogously, the \textbf{Edwards-Anderson order parameter} for expert consensus is:
\begin{equation}\label{eq:ea_order}
q_{EA} = \lim_{M\to\infty} \E\bqty{m_M^2(x)} = \lim_{M\to\infty} \frac{1}{M(M-1)} \sum_{\alpha \neq \beta} \E\bqty{\sigma_\alpha(x) \sigma_\beta(x)}
\end{equation}
\end{definition}

\section{SCX Consensus: The Ferromagnetic Phase Transition}
% ======================================================================
\subsection{The Mean-Field Model}
We model expert-expert interactions via a mean-field Hamiltonian (Sherrington-Kirkpatrick type):
\begin{equation}\label{eq:sk_analog}
H_M(\{\sigma_\alpha\}) = -\frac{J}{M} \sum_{\alpha < \beta} \sigma_\alpha \sigma_\beta - h \sum_{\alpha=1}^{M} \sigma_\alpha
\end{equation}
where \(J\) is the coupling strength (positive \(J\) means experts tend to agree), and \(h\) is the ``external field'' (task difficulty).

\begin{proposition}[Self-Consistent Magnetization]\label{prop:phase_trans}
\begin{equation}\label{eq:self_consistent}
m = \tanh\pqty{\beta J m + \beta h}
\end{equation}
There is a critical temperature \(T_c = J\) (i.e., \(\beta_c = 1/J\)):
\begin{itemize}
    \item For \(T > T_c\) (high temperature/weak coupling): the only solution is \(m = \tanh(\beta h)\), and \(m=0\) when \(h=0\) (no spontaneous consensus);
    \item For \(T < T_c\) (low temperature/strong coupling): a nonzero solution \(m \neq 0\) exists even at \(h=0\) (spontaneous consensus emerges).
\end{itemize}
\end{proposition}

\subsection{Finite-Size Scaling and the Minimum Number of Experts}
In practice, \(M\) is finite, and we care about ``how many experts are needed.'' The question: to detect consensus at signal strength \(\varepsilon\), how large must \(M\) be?

\begin{theorem}[Finite-Size Scaling]\label{thm:finite_M}
\begin{equation}\label{eq:fss}
m_M(T) = M^{-\beta/\nu} \, \mathcal{F}\pqty{M^{1/\nu}(T - T_c)}
\end{equation}
with mean-field exponents \(\beta = 1/2\) and \(\nu = 1\), where \(\mathcal{F}\) is a universal scaling function. Consequently:
\begin{equation}\label{eq:M_min_scaling}
M_{\min} \propto |T - T_c|^{-\nu} \propto |T - T_c|^{-1}
\end{equation}
Thus, \(M_{\min}\) diverges as the temperature approaches criticality---the ``hardest'' regime for consensus detection.
\end{theorem}

\begin{proof}[Proof Sketch]
Expand \(m = \tanh(\beta J m)\) near \(T_c\). As \(T \to T_c^-\), \(m\) grows as \(m \approx \sqrt{3(1 - T/T_c)}\). The finite-\(M\) fluctuation \(\delta m \sim 1/\sqrt{M}\) must be smaller than the signal for reliable detection, yielding the scaling. Full derivation in Appendix.
\end{proof}

\subsection{Large Deviations and Reliability}
The magnetization \(m_M\) is itself a random variable whose fluctuations determine audit reliability.

\begin{proposition}[Large Deviation Principle]\label{prop:large_dev}
As \(M \to \infty\), \(m_M\) satisfies:
\begin{equation}\label{eq:ldp}
\Pbb(m_M = m) \asymp \exp\pqty{-M \, I(m)}
\end{equation}
where \(I(m) = \sup_{\lambda} \{ \lambda m - \log \E[e^{\lambda \sigma}] \}\) is the rate function. The audit reliability bound follows:
\begin{equation}\label{eq:reliability}
\Pbb(\text{audit error}) \leq \exp\pqty{-M \cdot \inf_{m \in \mathcal{A}_{\text{err}}} I(m)}
\end{equation}
where \(\mathcal{A}_{\text{err}}\) is the set of magnetization values that would lead to incorrect audit conclusions.
\end{proposition}

\section{The Replica Method and Expert Overlaps}
% ======================================================================
\subsection{The Replica Trick}
\begin{equation}\label{eq:replica_trick}
\log Z = \lim_{n \to 0} \frac{Z^n - 1}{n}
\end{equation}
This allows computation of the quenched average \(\E_\D[\log Z]\) by analytically continuing from integer \(n\).

\begin{definition}[Replica Partition Function]
\begin{equation}\label{eq:replica_partition}
Z^n(\beta) = \int \prod_{a=1}^{n} d\theta_a \, \exp\pqty{-\beta \sum_{a=1}^{n} \loss_\D(\theta_a)}
\end{equation}
After averaging over \(\D\), the replicas become coupled.
\end{definition}

\subsection{Overlap Matrix and Expert Correlations}
\begin{definition}[Replica Overlap Matrix]
For \(n\) replicas, define the overlap matrix \(Q \in \R^{n \times n}\):
\begin{equation}\label{eq:overlap_matrix}
Q_{ab} = \frac{1}{N} \theta_a \cdot \theta_b = \frac{1}{N} \sum_{i=1}^{N} \theta_{a,i} \, \theta_{b,i}, \quad a,b = 1,\ldots,n
\end{equation}
with \(Q_{aa} = \frac{1}{N}\|\theta_a\|^2\) as self-overlap.

Analogously, the \textbf{expert correlation matrix} is:
\begin{equation}\label{eq:expert_corr}
C_{\alpha\beta} = \frac{1}{|\mathcal{X}|} \sum_{x \in \mathcal{X}} \sigma_\alpha(x) \sigma_\beta(x) = \E_x\bqty{\sigma_\alpha(x) \sigma_\beta(x)}
\end{equation}
\end{definition}

\begin{theorem}[Replica-Expert Isomorphism]\label{thm:overlap_corr}
In the infinite-width limit, the expert correlation \(C_{\alpha\beta}\) and the replica overlap \(Q_{ab}\) are spectrally isomorphic:
\begin{equation}\label{eq:spectral_corr}
\spec(C) \leftrightarrow \spec(Q) \quad \text{with the mapping} \quad M \leftrightarrow n, \; N \to \infty \;\text{(thermodynamic limit)}
\end{equation}
That is, the eigenvalue spectra of both matrices coincide when \(N, |\mathcal{X}| \to \infty\).
\end{theorem}

\begin{proof}
In the linear (NTK) regime, \(\sigma_\alpha(x) = \sgn(\theta_\alpha \cdot \phi(x))\). The correlation becomes:
\begin{equation}
C_{\alpha\beta} \approx \E_x\bqty{\sgn(\theta_\alpha \cdot \phi(x)) \, \sgn(\theta_\beta \cdot \phi(x))}
\end{equation}
Using \(\sgn(u)\sgn(v) = \frac{2}{\pi} \arcsin\pqty{\frac{u \cdot v}{\|u\|\|v\|}}\) for Gaussian \(u, v\), and assuming isotropic \(\phi(x)\), the result follows. Full derivation in Appendix.
\end{proof}

\subsection{Irreducible Expert Subspaces}
\begin{theorem}[Spectral Decomposition of Expert Space]\label{thm:irreducible}
Let \(\{v_k \in \R^M\}_{k=1}^{r}\) be the eigenvectors of \(C\) with \(r = \rank(C)\). The expert space decomposes into \(r\) irreducible subspaces:
\begin{equation}\label{eq:irreducible_dec}
\mathsf{ExpertSpace} = \bigoplus_{k=1}^{r} \mathcal{V}_k, \quad \dim(\mathcal{V}_k) = \text{mult}(\lambda_k)
\end{equation}
where \(\lambda_k\) is the \(k\)-th eigenvalue of \(C\) and \(\mathcal{V}_k\) is its eigenspace. Subspaces with \(\lambda_k \approx 1\) represent strongly correlated expert clusters (shared opinions); those with \(\lambda_k \approx 0\) represent independent disagreements.

Interpretation: among \(M\) experts, only \(r\) truly independent ``meta-opinions'' exist. When \(r\) is small, experts are highly redundant; when \(r \approx M\), they are maximally diverse.
\end{theorem}

\section{Parisi RSB and Expert Disagreement Hierarchies}
% ======================================================================
\subsection{The Replica Symmetry Ansatz}
The simplest approximation is \textbf{Replica Symmetry} (\(\RS\)):
\begin{equation}\label{eq:rs_ansatz}
Q_{ab} = q_0 \delta_{ab} + q (1 - \delta_{ab})
\end{equation}
All off-diagonal overlaps are identical, parameterized by a single number \(q\). The \(\RS\) solution is correct at high temperature, but fails at low temperature.

Parisi discovered that replica symmetry must be broken: the \(\RS\) solution becomes unstable because it yields negative entropy (a contradiction). The correct solution requires \textbf{Replica Symmetry Breaking} (\(\RSB\)).

\begin{definition}[Parisi Order Parameter]
In the \(\RSB\) scheme, the overlap \(Q_{ab}\) is parameterized by a function \(q(x)\) on \([0,1]\)---the \textbf{Parisi order parameter}. The function \(q(x)\) encodes the hierarchical organization: \(x\) indexes the ultrametric tree depth.
\begin{itemize}
    \item \(\RS\) corresponds to \(q(x) = q\) (constant function).
    \item One-step \(\RSB\) (\(\oneRSB\)): \(q(x) = q_0\) for \(x < m\), \(q(x) = q_1\) for \(x > m\).
    \item Full \(\RSB\) (\(\fRSB\)): \(q(x)\) is a continuous, non-decreasing function.
\end{itemize}
\end{definition}

\subsection{Parisi Classification of Expert Structures}
\begin{theorem}[Parisi-Expert Correspondence]\label{thm:parisi_expert}
As \(M \to \infty\) (equivalently \(n \to 0\)), the Parisi function \(q(x)\) classifies expert disagreement into three regimes:
\begin{enumerate}[label=(\arabic*)]
    \item \textbf{\(\RS\) (Replica Symmetric) --- Consensus:} \(Q_{ab} = q\) for all \(a \neq b\). Experts form a single homogeneous cluster. All pairwise disagreements are statistically identical;
    \item \textbf{\(\oneRSB\) (One-Step RSB) --- Clustered Disagreement:} Experts form distinct clusters. Within each cluster (overlap \(q_1\)), experts agree; between clusters (overlap \(q_0\)), they disagree. Cluster size \(\propto 1/m\), where \(m\) is the Parisi breakpoint;
    \item \textbf{\(\fRSB\) (Full RSB) --- Hierarchical Disagreement:} Experts exhibit a continuous hierarchy of disagreement, with ``fractal'' structure at all scales. This is the most complex and most common regime.
\end{enumerate}
\end{theorem}

\begin{proof}
The \(M \times M\) expert correlation matrix \(C_{\alpha\beta}\) is analyzed via its eigenvalue spectrum. Parisi theory dictates that \(q(x)\) parameterizes the distribution of overlaps. For each pair \((\alpha, \beta)\), there is an ultrametric ``distance'' \(x_{\alpha\beta} \in [0,1]\) such that \(C_{\alpha\beta} = q(x_{\alpha\beta})\). The Parisi function \(q(x)\) is determined by free energy minimization (Guerra-Tononelli bound). In the \(\fRSB\) regime, \(q(x)\) varies continuously, reflecting fractal cluster structure.
\end{proof}

\begin{theorem}[Unauditability in the fRSB Regime]\label{thm:unauditable}
When the expert system is in the \(\fRSB\) regime, reliable auditing is impossible because: (1) there is no ``ground truth'' consensus---every expert inhabits a distinct free energy valley; (2) the number of metastable states is extensive (exponential in \(N\)), so no finite \(M\) samples them all; (3) audit conclusions are inherently non-robust---small changes in the expert pool produce qualitatively different conclusions.
\end{theorem}

\section{Central Theorem: The Minimum Number of Experts}
% ======================================================================
\subsection{Problem Statement}
The core question: given the Hamiltonian \(H(\theta)\), what is \(M_{\min}\), the minimum number of experts needed for \(\varepsilon\)-complete coverage of the loss landscape?

In complex landscapes, experts (sampled via SGD) tend to cluster in a few large basins, leaving many minima unexplored. This ``incomplete sampling'' is the fundamental limitation. \(M_{\min}\) is governed by the complexity \(\Sigma(e)\).

\subsection{Kac-Rice and the Landscape Geometry}
From Section 2.2, the Kac-Rice formula gives \(\Sigma(e)\). For SCX auditing, we care about two energies:
\begin{equation}
e^* = \argmin_e \{ \Sigma_0(e) > 0 \} = \text{ground state energy}
\end{equation}
\begin{equation}
e_c = \text{energy of maximum complexity}, \quad \Sigma_0(e_c) = \Sigma_{\max}
\end{equation}

\begin{lemma}[Basin Volume]\label{lemma:golf}
The volume of the basin of attraction around minimum \(\theta_k^*\) is:
\begin{equation}
\mathcal{B}_k = \{\theta : \text{gradient flow from } \theta \text{ converges to } \theta_k^*\}
\end{equation}
\begin{equation}
\operatorname{Vol}(\mathcal{B}_k) \propto \frac{1}{\sqrt{\det \mathcal{H}(\theta_k^*)}}
\end{equation}
\end{lemma}

\subsection{The Main Theorem}
With \(M\) experts sampled independently (via SGD from random initializations), each expert lands in basin \(k\) with probability \(p_k\):
\begin{equation}\label{eq:basin_prob}
p_k = \frac{\exp(-\beta \loss(\theta_k^*)) / \sqrt{\det \mathcal{H}(\theta_k^*)}}{\sum_j \exp(-\beta \loss(\theta_j^*)) / \sqrt{\det \mathcal{H}(\theta_j^*)}}
\end{equation}

\begin{definition}[\(\varepsilon\)-Complete Coverage]
\(M\) experts achieve \textbf{\(\varepsilon\)-complete coverage} if every basin with \(p_k > \varepsilon\) is sampled at least once, with high probability:
\begin{equation}
\min_{k : p_k > \varepsilon} (\text{probability that basin } k \text{ is sampled}) \geq 1 - \delta
\end{equation}
for \(\delta > 0\) small.
\end{definition}

\begin{theorem}[Central Theorem: Minimum Number of Experts]\label{thm:central}
For a Hamiltonian \(H(\theta)\) with energy-landscape complexity \(\Sigma_0(e) = \lim_{N\to\infty} \frac{1}{N} \log \mathcal{N}_N^{(0)}(e)\), the minimum number of experts required for \(\varepsilon\)-complete coverage is:
\begin{equation}\label{eq:central_thm}
\boxed{M_{\min}(\beta, \varepsilon) \geq C \cdot \frac{\exp\pqty{N \, \Sigma_0(e_\beta)}}{-\log(1 - \varepsilon)} \cdot \pqty{1 + \mathcal{O}\pqty{\frac{1}{\sqrt{N}}}}}
\end{equation}
where \(C\) is a constant and \(e_\beta = \argmax_e [\Sigma_0(e) - \beta e]\) is the dominant energy at inverse temperature \(\beta\).

Equivalently, in more intuitive form:
\begin{equation}\label{eq:central_simple}
\boxed{M_{\min} \gtrsim \frac{\mathcal{N}_{\text{eff}}}{-\log(1 - p_{\text{cover}})}}
\end{equation}
where \(\mathcal{N}_{\text{eff}} = \exp(N \Sigma_0(e_\beta))\) is the effective number of basins and \(p_{\text{cover}}\) is the desired coverage probability.
\end{theorem}

\begin{proof}
Let \(\mathcal{K} = \{k : p_k > \varepsilon\}\) be the set of significant basins. Then \(|\mathcal{K}| \approx \exp(N \Sigma_0(e_\beta))\). The probability that basin \(k\) is missed by all \(M\) experts is \((1-p_k)^M\). By union bound, the probability that any significant basin is missed is:
\begin{equation}
\Pbb(\exists k \in \mathcal{K} : \text{basin } k \text{ missed}) \leq \sum_{k \in \mathcal{K}} (1-p_k)^M \leq |\mathcal{K}| \cdot (1 - \varepsilon)^M
\end{equation}
Setting this \(\leq \delta\):
\begin{equation}
M \geq \frac{\log |\mathcal{K}| - \log \delta}{-\log(1-\varepsilon)} \approx \frac{N \Sigma_0(e_\beta) - \log \delta}{-\log(1-\varepsilon)}
\end{equation}
When \(|\mathcal{K}| \gg 1\) (large \(N\) with \(\Sigma_0 > 0\)), the \(\log |\mathcal{K}| = N \Sigma_0(e_\beta)\) term dominates, yielding Equation~\eqref{eq:central_thm}.

For non-uniform \(p_k\), Bennett or Talagrand concentration inequalities refine the bound. If \(p_k\) follows a power-law distribution \(p_{(k)} \propto k^{-\alpha}\), the minimum shifts by at most a constant factor.
\end{proof}

\subsection{Consequences}
\begin{corollary}[Unauditability Criterion]\label{cor:unauditable}
\begin{enumerate}[label=(\arabic*)]
    \item \textbf{Exponential barrier:} If \(\Sigma_0(e_\beta) > 0\) and \(N \Sigma_0(e_\beta) \gg \log M_{\text{feasible}}\), where \(M_{\text{feasible}}\) is the practically deployable number of experts, then reliable auditing is impossible;
    \item \textbf{\(\fRSB\) obstruction:} The Parisi function \(q(x)\) varying continuously on \([0,1]\) signals the \(\fRSB\) regime---the landscape is so complex that no finite \(M\) suffices;
    \item \textbf{Fundamental limit:} In the worst case, audit conclusions are inherently non-robust.
\end{enumerate}
\end{corollary}

\begin{corollary}[Minimum Accessible Temperature]\label{cor:min_temp}
Given a maximum feasible expert count \(M_{\max}\), the minimum accessible temperature is:
\begin{equation}
T_{\min} = \argmin_T \{ \Sigma_0(e_{1/T}) \leq \frac{1}{N} \log (M_{\max} \cdot (-\log(1-\varepsilon))) \}
\end{equation}
Below \(T < T_{\min}\) (sharper/more confident experts), the landscape is too complex to cover---this is ``Overfitting'' in the auditing sense.
\end{corollary}

\section{SCX Audit Protocol via Spectral Analysis}
% ======================================================================
\subsection{Decision Decomposition}
The theory from Sections 5-6 directly yields a practical SCX audit protocol based on the spectral decomposition of \(Q_{ab}\) (equivalently, the expert correlation \(C_{\alpha\beta}\)). This replaces ``majority vote'' heuristics.

\begin{definition}[Audit Decision Axes]\label{def:audit_axis}
Let \(C = \sum_{k=1}^{r} \lambda_k v_k v_k^\top\) be the spectral decomposition, with \(\lambda_1 \geq \lambda_2 \geq \cdots \geq \lambda_r > 0\). The \(k\)-th eigenvector \(v_k \in \R^M\) defines an audit ``axis,'' where \(v_{k,\alpha}\) is expert \(\alpha\)'s loading. The expert decision vector for input \(x\) decomposes as:
\begin{equation}\label{eq:decision_decomp}
\boldsymbol{\sigma}(x) = \sum_{k=1}^{r} c_k(x) \cdot v_k + \boldsymbol{\eta}(x)
\end{equation}
where \(c_k(x) \in \R\) is the ``consensus score'' along axis \(k\), and \(\boldsymbol{\eta}(x)\) is the residual (individual expert noise).
\end{definition}

\begin{proposition}[Effective Degrees of Freedom]\label{prop:eff_dof}
The effective number of independent expert opinions for input \(x\) is:
\begin{equation}\label{eq:eff_dof_eq}
d_{\text{eff}}(x) = \sum_{k=1}^{r} \mathbb{1}\bqty{|c_k(x)| > \tau}
\end{equation}
where \(\tau\) is a significance threshold. A low \(d_{\text{eff}}\) indicates strong consensus; a high \(d_{\text{eff}}\) indicates genuine multi-perspective disagreement.
\end{proposition}

\subsection{Parisi Regime Detection}
\begin{enumerate}[label=\textbf{Regime \Roman*}, leftmargin=*]
    \item \textbf{Consensus (\(\RS\)):} One dominant eigenvalue: \(\lambda_1 \gg \lambda_2 \approx 0\). All experts agree---the audit is trivial.
    \item \textbf{Clustered (\(\oneRSB\)):} \(K\) significant eigenvalues. Experts form \(K\) clusters. The audit must report cluster-level conclusions: ``within each of the \(K\) camps, experts agree; between camps, they disagree.''
    \item \textbf{Hierarchical (\(\fRSB\)):} No clear eigenvalue gap. The audit is inherently ambiguous. The report must state: ``expert opinions exhibit continuous, fractal disagreement structure; no single consensus exists.''
\end{enumerate}

\subsection{Algorithm}
\begin{algorithm}[H]
\caption{SCX Hamiltonian Audit Protocol}
\label{alg:audit}
\begin{algorithmic}[1]
\Require \(M\) experts \(\{\mathcal{E}_\alpha\}_{\alpha=1}^{M}\), task \(\mathcal{T}\), threshold \(\delta\)
\Ensure Audit conclusion with reliability score
\State Compute \(C_{\alpha\beta} \gets \E_{x \in \mathcal{T}}[\sigma_\alpha(x) \sigma_\beta(x)]\)
\State Eigendecomposition: \(\{(\lambda_k, v_k)\}_{k=1}^{r} \gets \operatorname{eig}(C)\)
\State Determine Parisi regime:
\If{\(\frac{\lambda_1}{\sum_k \lambda_k} > 1 - \delta\)}
    \State \textbf{RS:} Consensus; report majority conclusion
\ElsIf{\(\exists K \text{ s.t. } \frac{\sum_{k=1}^{K} \lambda_k}{\sum_k \lambda_k} > 1 - \delta \text{ and } \frac{\lambda_K}{\lambda_{K+1}} \gg 1\)}
    \State \textbf{1RSB:} \(K\) clusters detected; report cluster-level conclusions \(\to\) flag for human review
\Else
    \State \textbf{fRSB:} Hierarchical disagreement; report ``inconclusive''; recommend increasing \(M\)
\EndIf
\State Reliability: \(R \gets 1 - \exp(-M \cdot I(m_{\text{obs}}))\) (from large deviation theory)
\State \Return conclusion, reliability \(R\), Parisi regime
\end{algorithmic}
\end{algorithm}

\section{Limitations and Open Problems}
% ======================================================================
\subsection{Limitations}
\begin{enumerate}[label=\textbf{L\arabic*}, leftmargin=*]
    \item \textbf{Exponential lower bound:} \(\Sigma_0(e_\beta) > 0\) implies \(\exp(N\Sigma_0)\) grows super-polynomially in \(N\). In practice, the landscape may be simpler---but the bound is a worst-case guarantee. \textbf{Implication:} auditing large models may require infeasibly many experts.
    \item \textbf{Non-Gaussian landscapes:} \(\Sigma_0(e_\beta) > 0\) is the generic case---most loss landscapes are glassy. \textbf{Implication:} the exponential barrier is the rule, not the exception.
    \item \textbf{Fragility of \(\fRSB\):} In the fRSB regime, audit conclusions are sensitive to the exact composition of the expert pool. \textbf{Implication:} reproducibility requires documenting the full expert pool---a ``which experts'' specification.
\end{enumerate}

\subsection{Open Problems}
In the \(\fRSB\) regime, the following fundamental question arises:
\begin{proposition}[Chaos in Expert Selection]\label{prop:chaos}
In the \(\fRSB\) regime, replacing a single expert \(\mathcal{E}\) with a different expert \(\mathcal{E}'\) can cause the consensus magnetization to change by an \(O(1)\) amount (even as \(M \to \infty\)):
\begin{equation}\label{eq:chaos}
\lim_{M \to \infty} \E\bqty{\abs{m_M(\mathcal{E}) - m_M(\mathcal{E}')}} > 0
\end{equation}
\end{proposition}

\subsection{Connection to Neural Scaling}
\begin{equation}
\lim_{\text{width} \to \infty} \Sigma_0(e) = 0 \quad \text{for all } e > e_{\min}
\end{equation}
In the infinite-width (NTK) regime, the landscape becomes convex (\(\RS\) solution), and the exponential barrier vanishes. \textbf{Implication:} auditing wide networks is easier---but real networks are not infinitely wide. The effective complexity \(N_{\text{eff}}\) interpolates between these extremes, governed by \(\Sigma_0(e)\).

\section{Conclusion}
% ======================================================================
\begin{enumerate}
    \item We established a rigorous correspondence between the SCX multi-expert auditing framework and the statistical mechanics of disordered systems, with the Hamiltonian \(H(\theta) = \loss(\theta) + \log Z\) as the central organizing quantity.
    \item Each SCX expert corresponds to a Gibbs sample from the Hamiltonian landscape at effective temperature \(T_\alpha\), with coupling between experts mediated by shared training data (the ``quenched disorder'').
    \item In the SCX consensus model, spontaneous order emerges at a critical temperature \(T_c = J\) (where \(J\) is expert coupling strength), with a sharp phase transition as \(M \to \infty\).
    \item The replica overlap matrix \(Q_{ab}\) maps isomorphically to expert-expert correlations \(C_{\alpha\beta}\), with identical spectral structure.
    \item The Parisi RSB scheme classifies disagreement structures: \(\RS\) (consensus), \(\oneRSB\) (clustered disagreement), and \(\fRSB\) (hierarchical, unauditable disagreement).
    \item \textbf{Central result:} \(M_{\min} \propto \exp(N \Sigma_0(e_\beta))\)---the minimum number of experts grows exponentially with the landscape complexity.
\end{enumerate}

\subsection{Proof of Theorem~\ref{thm:finite_M} (Finite-Size Scaling)}
\begin{equation}
m = \tanh(\beta J m + \beta h)
\end{equation}
Set \(h = 0\) (no external field). For \(m \ll 1\), Taylor expand:
\begin{equation}
m = \beta J m - \frac{(\beta J m)^3}{3} + \mathcal{O}(m^5)
\end{equation}
\begin{equation}
m\bqty{1 - \beta J + \frac{(\beta J)^3 m^2}{3}} = 0
\end{equation}
For \(1 - \beta J < 0\), i.e., \(T < J\) (or \(\beta > 1/J\)), the nonzero solution is:
\begin{equation}
m = \pm \sqrt{\frac{3(\beta J - 1)}{(\beta J)^3}} \approx \pm \sqrt{3\pqty{1 - \frac{T}{T_c}}}, \quad T \to T_c^-
\end{equation}
Writing \(\beta = \beta_c(1 + \varepsilon)\) with \(\varepsilon = (T_c - T)/T \ll 1\). For finite \(M\), the empirical magnetization \(m_M = \frac{1}{M}\sum_\alpha \sigma_\alpha\) has \(\operatorname{Var}(m_M) = \frac{1 - m^2}{M}\) (binomial). The signal exceeds noise when \(m\) is at least \(\sqrt{\operatorname{Var}(m_M)}\):
\begin{equation}
\sqrt{3\pqty{1 - \frac{T}{T_c}}} \gtrsim \sqrt{\frac{1}{M}} \implies M \gtrsim \frac{1}{3(1 - T/T_c)}
\end{equation}
Thus \(M_{\min} \propto |T - T_c|^{-1}\), confirming \(\nu = 1\). \(\hfill \square\)

\subsection{Proof of Theorem~\ref{thm:overlap_corr} (Replica-Expert Isomorphism)}
Consider the linear (NTK) regime: \(f(x; \theta) = \sum_{i=1}^{N} \theta_i \phi_i(x)\), where \(\phi_i(x)\) are features. For random \(x\):
\begin{equation}
\Pbb(\sigma_\alpha(x) = \sigma_\beta(x)) = \Pbb(\sgn(\theta_\alpha \cdot \phi(x)) = \sgn(\theta_\beta \cdot \phi(x)))
\end{equation}
With \(\theta_\alpha, \theta_\beta\) fixed and \(\phi(x)\) isotropic Gaussian, the angle between projections yields:
\begin{equation}
\Pbb(\sigma_\alpha = \sigma_\beta) = 1 - \frac{1}{\pi} \arccos\pqty{\frac{\theta_\alpha \cdot \theta_\beta}{\|\theta_\alpha\| \|\theta_\beta\|}}
\end{equation}
\begin{equation}
C_{\alpha\beta} = \E[\sigma_\alpha \sigma_\beta] = \frac{2}{\pi} \arcsin\pqty{\frac{\theta_\alpha \cdot \theta_\beta}{\|\theta_\alpha\| \|\theta_\beta\|}}
\end{equation}
Define \(\tilde{Q}_{\alpha\beta} = \frac{\theta_\alpha \cdot \theta_\beta}{\|\theta_\alpha\| \|\theta_\beta\|}\). Then \(C_{\alpha\beta} = g(\tilde{Q}_{\alpha\beta})\) where \(g(z) = \frac{2}{\pi}\arcsin(z)\) is monotone on \([-1,1]\). As \(|\mathcal{X}| \to \infty\), the empirical \(C\) converges to its expectation. The monotonicity of \(g\) and the random matrix concentration of \(\tilde{Q}\) to \(Q_{ab} = \frac{1}{N}\theta_a \cdot \theta_b\) (in the \(N \to \infty\) limit) imply spectral isomorphism. \(\hfill \square\)

\subsection{Refined Bound for Theorem~\ref{thm:central}}
Let \(\mathcal{N} = \exp(N \Sigma_0(e_\beta))\) be the number of significant basins, indexed \(k = 1, \ldots, \mathcal{N}\). Let \(p_k\) be the probability that expert \(\alpha\) lands in basin \(k\). Define \(I_k^{(M)} = \mathbb{1}\{\text{basin } k \text{ is missed by all } M \text{ experts}\}\). Then \(\Pbb(I_k^{(M)} = 1) = (1 - p_k)^M\). The expected number of missed basins is:
\begin{equation}
\E\bqty{\sum_{k=1}^{\mathcal{N}} I_k^{(M)}} = \sum_{k=1}^{\mathcal{N}} (1 - p_k)^M
\end{equation}
Uniform case (\(p_k = 1/\mathcal{N}\) for all \(k\)):
\begin{equation}
\E[\text{missed}] = \mathcal{N} \cdot \pqty{1 - \frac{1}{\mathcal{N}}}^M \approx \mathcal{N} \cdot \exp\pqty{-\frac{M}{\mathcal{N}}}
\end{equation}
Requiring \(\E[\text{missed}] \leq \delta\):
\begin{equation}
\mathcal{N} \, e^{-M / \mathcal{N}} \leq \delta \implies M \geq \mathcal{N} \log \frac{\mathcal{N}}{\delta}
\end{equation}
Since \(\mathcal{N} \gg 1\) and \(\mathcal{N} = \exp(N \Sigma_0)\), this confirms the exponential scaling.

For non-uniform \(p_k\), Bennett's inequality refines the bound. Let \(p_{(1)} \leq p_{(2)} \leq \cdots \leq p_{(\mathcal{N})}\). Then:
\begin{equation}
\frac{\log \mathcal{N} - \log \delta}{-\log(1 - p_{(1)})} \leq M_{\min} \leq \frac{\log \mathcal{N} - \log \delta}{-\log(1 - p_{(\mathcal{N})})}
\end{equation}
If \(p_{(k)} \propto k^{-\alpha}\) (power-law from Hessian determinant variation), the bounds remain within constant factors. \(\hfill \square\)
