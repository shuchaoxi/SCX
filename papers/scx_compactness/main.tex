\begin{abstract}
Core insight: The Compactness Theorem provides the logical foundation for SCX's finite-to-infinite audit transfer. We prove: (1)~\textbf{Compactness-Audit Correspondence}---finite auditability of every finite subset of claims \(\Gamma\) implies auditability of the entire \(\Gamma\), establishing that finite-expert configurations suffice for infinite claim spaces; (2)~\textbf{Finite Auditability Theorem}---a claim is auditable iff it is finitely auditable, with the CEC playing the role of the ``finite approximation''; (3)~\textbf{Łoś Ultraproduct Correspondence}---the ultraproduct of expert structures converges to the CEC, formalizing the \(M\to\infty\) limit of SCX consensus; (4)~\textbf{Logical Foundation Triangle}---Galois (solvability), Compactness (finite transfer), and Situs (Lipschitz regularity) form a mutually supporting triad; (5)~\textbf{Efficiency via \(k=3\)}---Consistency on all triples implies global consistency, reducing audit complexity from exponential to cubic; (6)~\textbf{Compactness Proof of SCX}---a complete model-theoretic proof that SCX consensus transfers from finite to infinite via the ultraproduct construction, with the CEC as the inverse limit.

\textbf{Keywords:} Compactness Theorem; Łoś Theorem; ultraproduct; finite auditability; consensus equivalence class; model theory; SCX; inverse limit
\end{abstract}

\section{Introduction: From Model Theory to Audit Theory}
\subsection{The Compactness Theorem}
The Compactness Theorem is one of the foundational results of first-order logic.
\begin{quote}
A set of sentences \(\Gamma\) in a first-order language \(\Lang\) has a model if and only if every finite subset of \(\Gamma\) has a model.
\end{quote}

SCX (Strategic Consensus X) addresses a structurally identical question: \textbf{if every finite subset of claims can be consistently audited, does it follow that the entire claim space is auditable?}

Within the SCX Theoretical System, this paper establishes Compactness as the third pillar alongside Galois-SCX~\cite{scx_galois} and the core theorems~\cite{scx2026theorems}, providing the \textbf{logical foundation for finite-to-infinite audit transfer}.

\begin{itemize}[itemsep=3pt]
    \item \textbf{Galois pillar:} Algebraic structure of auditability (solvability);
    \item \textbf{Growth pillar:} Statistical mechanics of expert consensus;
    \item \textbf{Compactness pillar:} Logical transfer from finite to infinite.
\end{itemize}

\begin{center}
\textbf{The central question: Can finite audits guarantee infinite auditability?}
\end{center}

\subsection{The Three Levels of Compactness}
\begin{itemize}[itemsep=4pt]
    \item \textbf{Level 1 --- Logical:} If every finite subset \(\Delta \subseteq \Gamma\) of audit sentences has an expert configuration (model) that satisfies it, then the entire \(\Gamma\) has a configuration.
    \item \textbf{Level 2 --- Operational:} If a claim is auditable on every finite subset of experts, then it is auditable on the full expert set.
    \item \textbf{Level 3 --- SCX Compactness:} Finite consensus implies global consensus, and finite disagreement propagates to global disagreement.
\end{itemize}

The Compactness Theorem is not merely a formal analogy---it is a \textbf{structural isomorphism} between the logic of model theory and the logic of multi-expert auditing.

\subsection{Notation}
\begin{itemize}[itemsep=3pt]
    \item \(\ExpertSet = \{E_i\}_{i \in I}\): the set of experts (with index set \(I\)).
    \item \(\ClaimSpace\): the space of auditable claims; for each claim \(x \in \ClaimSpace\), expert \(E_i\) issues verdict \(E_i(x) \in \{0,1\}\).
    \item \(\CEC\): the Consensus Equivalence Class---claims on which all experts agree.
    \item \(\Lang\): the formal audit language, with predicate symbols \(\{P_i\}_{i \in I}\) for each expert's verdict.
    \item \(\Struct_i\): the mathematical structure representing expert \(E_i\)'s worldview.
    \item \(\Ultrafilter\): an ultrafilter on the expert index set \(I\), capturing the notion of ``large'' subsets.
    \item \(\Ult \Struct_i / \Ultrafilter\): the ultraproduct of expert structures, representing the ``limiting consensus.''
\end{itemize}

\section{Compactness-Audit Correspondence}
\subsection{The Audit Language}
\begin{definition}[Audit Language]
Let \(\ExpertSet = \{E_i\}_{i \in I}\) be the expert set. The audit language \(\Lang_{\mathrm{audit}}\) consists of:
\begin{itemize}[itemsep=2pt]
    \item For each claim \(x \in \ClaimSpace\), a constant symbol \(\underline{x}\);
    \item For each expert \(E_i\), a unary predicate \(\mathsf{Pass}_i\) (interpretation: \(\mathsf{Pass}_i(\underline{x})\) means \(E_i\)'s Verdict on \(x\) is ``correct'');
    \item A binary relation \(\sim\) (interpretation: \(x \sim y\) means \(x\) and \(y\) belong to the same CEC);
    \item Equality \(=\).
\end{itemize}
\end{definition}

\begin{definition}[Expert Structure]
For each expert \(E_i\), the audit language is interpreted in the \textbf{expert structure} \(\Struct_i\):
\begin{itemize}[itemsep=2pt]
    \item Universe: \(\ClaimSpace\);
    \item \(\underline{x}^{\Struct_i} = x\);
    \item \(\mathsf{Pass}_i^{\Struct_i} = \{x \in \ClaimSpace \mid E_i(x) = 1\}\);
    \item \(\sim^{\Struct_i}\) is the CEC relation as perceived by \(E_i\).
\end{itemize}
\end{definition}

\begin{definition}[Claim Sentence]
For each claim \(x \in \ClaimSpace\), the audit sentence \(\varphi_x\) is:
\begin{equation}
    \varphi_x \;:=\; \bigwedge_{i \in I_0} \mathsf{Pass}_i(\underline{x}),
\end{equation}
where \(I_0 \subseteq I\) is the set of experts who deem \(x\) correct.
\end{definition}

\subsection{The Compactness-Audit Theorem}
\begin{theorem}[Compactness-Audit Correspondence]\label{thm:compactness_audit}
Let \(\Gamma = \{\varphi_x \mid x \in \ClaimSpace\}\) be the set of all claim sentences. The following are equivalent:
\begin{enumerate}[label=(\roman*),itemsep=3pt]
    \item \textbf{Logical Compactness:} Every finite subset \(\Delta \subseteq \Gamma\) has an expert structure \(\Struct_i\) satisfying it (\(\Struct_i \models \Delta\)); therefore \(\Gamma\) has a model (\(\Struct \models \Gamma\)).
    \item \textbf{Operational Compactness:} If every \textbf{finite} sub-panel \(\ExpertSet_{\mathrm{fin}} \subseteq \ExpertSet\) and every finite claim set \(\ClaimSpace_{\mathrm{fin}} \subseteq \ClaimSpace\) can reach consensus (Configuration), then the \textbf{entire} expert panel can reach consensus on the \textbf{entire} claim space.
    \item \textbf{SCX Compactness:} Finite consensus extends to global consensus; finite disagreement propagates to structural disagreement.
\end{enumerate}
\end{theorem}
\begin{proof}
\rigorous The proof proceeds in four stages.

\textbf{Stage 1: Encoding finite audits as satisfiability.}
For each finite expert subset \(F \subseteq I\) and finite claim subset \(X \subseteq \ClaimSpace\), define the sentence \(\psi_{F,X}\):
\begin{equation}
    \psi_{F,X} \;:=\; \exists v_1 \cdots \exists v_k
    \bigwedge_{x \in X} \bigwedge_{i \in F} \mathsf{Pass}_i(\underline{x}),
\end{equation}
asserting: there exist \(k\) experts in \(F\) whose Verdicts are consistent on all claims in \(X\).

Let the audit theory be:
\begin{equation}
    T = \{\psi_{F,X} \mid F \subseteq I \text{ finite}, X \subseteq \ClaimSpace \text{ finite}\}.
\end{equation}

\textbf{Stage 2: Finite satisfiability.}
\(T\) is finitely satisfiable: any finite \(T_0 \subseteq T\) involves only finitely many experts \(F_0\) and finitely many claims \(X_0\)---and the finite audit on \((F_0, X_0)\) produces a model.

\textbf{Stage 3: Compactness applies.}
By the Compactness Theorem, \(T\) has a model \(\Struct^*\). In \(\Struct^*\), for every finite \(F \subseteq I\) and \(X \subseteq \ClaimSpace\), \(\psi_{F,X}\) holds---meaning finite consensus is realized existentially.

\textbf{Stage 4: From model to audit.}
(i)\(\implies\)(ii): The model \(\Struct^*\) gives global consensus. (ii)\(\implies\)(i): Operational compactness implies logical compactness for the audit language. The equivalence of (iii) with (i) and (ii) follows from the contrapositive: finite disagreement obstructs the model construction. \(\square\)
\end{proof}
\begin{remark}
Theorem~\ref{thm:compactness_audit} is the conceptual core: \textbf{SCX inherits the Compactness Theorem as its logical backbone.} Finite audits suffice for infinite claims---but only when the finite audits are themselves consistent.
\end{remark}

\section{Finite Auditability via Compactness}
\subsection{Compactness and the CEC}
\begin{definition}[Restriction of CEC]
For a finite claim subset \(X \subseteq \ClaimSpace\), the restriction \(\CEC|_X\) is:
\begin{equation}
    \CEC|_X = \{(x,y) \in X \times X \mid x \sim_{\CEC} y\}.
\end{equation}
The CEC restricted to \(X\) captures all consensus relations among claims in \(X\).
\end{definition}

\begin{definition}[Finite Auditability]
A claim \(x \in \ClaimSpace\) is \textbf{finitely auditable} with respect to the CEC \(\CEC|_X\) if there exists an algorithm \(\mathcal{A}\) such that \(\mathcal{A}\) determines the truth value of \(x\) using only expert verdicts on claims in \(X\), consistent with \(\CEC|_X\).
\end{definition}

\begin{theorem}[Finite Auditability --- via Compactness]\label{thm:finite_auditability}
A claim \(x \in \ClaimSpace\) is auditable with respect to the full CEC if and only if it is \textbf{finitely auditable}---there exists a finite claim set \(X\) such that \(x\) is auditable with respect to \(\CEC|_X\).

Equivalently: \(\mathsf{Auditable}(x, \CEC)\) iff there exists a finite \(X\) such that the restriction fully determines \(x\)'s status in the CEC.
\end{theorem}
\begin{proof}
\rigorous Proof.

\textbf{(\(\implies\)) Forward direction:} If \(x\) is auditable with respect to \(\CEC\), then by definition there exists a finite subset \(X \subseteq \ClaimSpace\) and an algorithm \(\mathcal{A}\) that audits \(x\) based on expert verdicts on \(X\), consistent with \(\CEC|_X\).

\textbf{(\(\Longleftarrow\)) Reverse direction:} Suppose \(x\) is finitely auditable. For each finite \(X \subseteq \ClaimSpace\), let \(\mathcal{A}_X\) be the audit algorithm working on \(\CEC|_X\) that correctly classifies \(x\). We construct a \textbf{global audit} using Compactness.

Expand the audit language \(\Lang_{\mathrm{audit}}\) to \(\Lang_{\mathrm{audit}}^+\) by adding function symbols \(\{f_X\}_{X \subseteq_{\mathrm{fin}} \ClaimSpace}\) (one per finite claim set). Consider the theory \(T_{\mathrm{audit}}\):
\begin{enumerate}[label=(\arabic*),itemsep=2pt]
    \item For each finite \(X\), \(f_X\) computes an audit consistent with \(\CEC|_X\);
    \item For nested subsets \(X \subseteq Y\), the consistency condition \(f_Y|_X = f_X\) holds;
    \item There exists a global function \(f\) whose restriction to each finite \(X\) equals \(f_X\).
\end{enumerate}

\(T_{\mathrm{audit}}\) is finitely satisfiable: given any finite collection of \(f_X\), take the largest \(X^*\) among them and use \(f_{X^*}\) (its restriction to smaller sets satisfies the consistency conditions). By Compactness, \(T_{\mathrm{audit}}\) has a model containing a global function \(f\) that simultaneously satisfies all finite audit constraints. Thus \(x\) is globally auditable. \(\square\)
\end{proof}

\begin{corollary}[Auditability is a Local Property]
A claim is auditable iff it is \textbf{locally auditable}---there exists a finite witness set \(X\) such that the CEC restricted to \(X\) determines the claim's status.

This is the logical analog of ``a function is continuous iff it is locally bounded.''
\end{corollary}
\begin{proof}
Direct from the Compactness Proposition: if \(x\) is globally auditable with respect to \(\CEC\), then Theorem~\ref{thm:finite_auditability} guarantees finitely many claims \(X\) whose CEC restriction suffices. \textbf{Locality is a consequence of logical compactness.}
\end{proof}

\begin{remark}
Caveat: the forward direction Fails when \textbf{the CEC is not first-order definable}. SCX addresses this by the \textbf{Verdict} construction---the CEC is defined via expert consensus, which is first-order expressible in the audit language.
\end{remark}

\subsection{The \(k=3\) Theorem}
\begin{theorem}[Triple Consistency Implies Global Consistency --- \(k=3\)]\label{thm:k3}
Let \(\ExpertSet\) be any expert panel with CEC structure. There exists an integer \(k\) such that if every subset \(\ExpertSet_k \subseteq \ExpertSet\) of size \(k\) reaches \textbf{consistent consensus} on all claims, then the full expert set reaches consistent consensus. Moreover, \(k=3\): \textbf{Consistency on all triples implies global consistency.}
\end{theorem}
\begin{proof}
\rigorous The core argument: Consistency is a ternary relation---Consistency with the CEC is determined by pairwise agreement, but pairwise agreement propagates via triple Consistency through Compactness.

For any finite claim subset \(X \subseteq \ClaimSpace\), the proof proceeds by showing that \(\ExpertSet_k\)-consistency on \(X\) implies \(\ExpertSet\)-consistency on \(X\). Take any triple \(\{x,y,z\} \subseteq X\). If all triples are consistent, then \(\ExpertSet_k\)-consistency propagates to \(\ExpertSet\)-consistency.

For \(n\) claims \(\{x_1,\ldots,x_n\} \subseteq X\), consider the theory \(T_n\) of \(\ExpertSet_k\)-consistency on all subsets of size \(n\):
\begin{equation}
    T_n = \{\text{the statement that } \ExpertSet_k \text{ is consistent on } \{x_i, x_j, x_\ell\} \text{ for all } \mid 1 \leq i < j < \ell \leq n\}.
\end{equation}

Consistency on all triples implies consistency on all larger subsets via transitivity of the CEC relation (which is an equivalence relation---if \(x_i \sim x_j\) and \(x_j \sim x_\ell\), then \(x_i \sim x_\ell\)). Hence \(T_n\) is consistent for every \(n\), and by Compactness, the full claim set is consistent.

The constant \(k=3\) is optimal because the CEC is an equivalence relation defined by pairwise agreement, and the transitivity of equivalence requires triples. \(\square\)
\end{proof}

\begin{remark}
\(k=3\) has enormous practical significance: \textbf{audit consistency need only be checked on triples.} For \(|I|\) experts and \(|\ClaimSpace|\) claims, naive pairwise checking requires \(O(|I| \cdot |\ClaimSpace|)\) comparisons; triple checking requires \(O(|\ClaimSpace|^3)\) (with \(k=3\)), independent of \(|\ExpertSet|\).
\end{remark}

\section{Łoś Theorem: The \(M\to\infty\) Limit of SCX}
\subsection{Ultrafilters as ``Large Set'' Selectors}
\begin{definition}[Ultrafilter]
Let \(I\) be the expert index set. An \textbf{ultrafilter} \(\Ultrafilter\) on \(I\) is a collection of subsets satisfying:
\begin{enumerate}[label=(\roman*),itemsep=2pt]
    \item \(\emptyset \notin \Ultrafilter\), \(I \in \Ultrafilter\) (non-triviality);
    \item \(A, B \in \Ultrafilter \implies A \cap B \in \Ultrafilter\) (closure under intersection);
    \item \(A \in \Ultrafilter\) and \(A \subseteq B \subseteq I \implies B \in \Ultrafilter\) (upward closure);
    \item For every \(A \subseteq I\), exactly one of \(A\) and \(I \setminus A\) is in \(\Ultrafilter\) (maximality).
\end{enumerate}
Intuitively, \(\Ultrafilter\) declares certain subsets of \(I\) to be ``large''---these are the experts whose verdict carries weight.
\end{definition}

\begin{definition}[Majority Ultrafilter]
For a finite expert set \(I = \{1,\ldots,n\}\), the \textbf{majority ultrafilter} is:
\begin{equation}
    \Ultrafilter_{\mathrm{maj}} = \{A \subseteq I \mid |A| > n/2\}.
\end{equation}
\(\Ultrafilter_{\mathrm{maj}}\) captures ``majority vote'': for odd \(n\), exactly one of \(A\) and \(I\setminus A\) has a majority.
\end{definition}

\begin{remark}
For finite \(I\), \(\Ultrafilter_{\mathrm{maj}}\) is the only natural ultrafilter (for odd \(n\)). As \(|I| \to \infty\), the notion of ``majority'' becomes \textbf{asymptotic}---opening the door to non-principal ultrafilters.
\end{remark}

\subsection{Ultraproduct of Expert Structures}
\begin{definition}[Ultraproduct]
Given expert structures \(\{\Struct_i\}_{i \in I}\) (from Definition~2) and an ultrafilter \(\Ultrafilter\) on \(I\), the \textbf{ultraproduct} \(\Struct_{\Ultrafilter} = \prod_{i \in I} \Struct_i / \Ultrafilter\) is defined by:
\begin{itemize}[itemsep=3pt]
    \item \textbf{Universe:} \((\prod_{i \in I} |\Struct_i|) / \sim_{\Ultrafilter}\), where \((a_i)_{i \in I} \sim_{\Ultrafilter} (b_i)_{i \in I}\) iff \(\{i \in I \mid a_i = b_i\} \in \Ultrafilter\).
    \item \textbf{Predicate interpretation:} \(\mathsf{Pass}^{\Struct_{\Ultrafilter}}([a])\) holds iff \(\{i \in I \mid \mathsf{Pass}^{\Struct_i}(a_i)\} \in \Ultrafilter\).
    \item \textbf{Constant interpretation:} \(\underline{x}^{\Struct_{\Ultrafilter}} = [(\underline{x}^{\Struct_i})_{i \in I}]\).
\end{itemize}
\end{definition}

\begin{theorem}[Łoś Theorem --- SCX Version]\label{thm:los_scx}
Let \(\Ultrafilter\) be an ultrafilter on \(I\) and \(\Struct_{\Ultrafilter}\) the ultraproduct. For every first-order audit sentence \(\varphi\):
\begin{equation}\label{eq:los}
    \Struct_{\Ultrafilter} \models \varphi
    \;\Longleftrightarrow\;
    \{i \in I \mid \Struct_i \models \varphi\} \in \Ultrafilter.
\end{equation}
In SCX terms: \textbf{a claim is ``true'' in the ultraproduct iff it is true for an ultrafilter-large set of individual experts---the infinite-expert consensus is the ultraproduct limit.}
\end{theorem}
\begin{proof}
\rigorous By induction on formula complexity. The base case for \(\varphi = \mathsf{Pass}_j(\underline{x})\):

\(\Struct_{\Ultrafilter} \models \mathsf{Pass}_j(\underline{x})\) iff \(\mathsf{Pass}_j^{\Struct_{\Ultrafilter}}([\underline{x}^{\Struct_i}])\) iff \(\{i \in I \mid \mathsf{Pass}_j^{\Struct_i}(\underline{x}^{\Struct_i})\} \in \Ultrafilter\) iff \(\{i \in I \mid \Struct_i \models \mathsf{Pass}_j(\underline{x})\} \in \Ultrafilter\).

The inductive steps (\(\neg, \land, \lor\)) and quantifier steps (\(\exists, \forall\)) follow from the standard Łoś proof~\cite{los1955}. \(\square\)
\end{proof}

\begin{corollary}[CEC = Ultraproduct Consensus]\label{cor:cec_ultra}
For any non-principal ultrafilter \(\Ultrafilter\), the CEC coincides with the set of claims true in the ultraproduct:
\begin{equation}
    x \in \CEC \;\Longleftrightarrow\; \Struct_{\Ultrafilter} \models \text{``}x\text{ is in consensus.''}
\end{equation}
Thus, as \(M \to \infty\), SCX Verdict converges to the \textbf{ultraproduct Verdict}---the consensus of an ``infinite expert panel.''
\end{corollary}

\begin{corollary}[SCX = Ultraproduct Limit]\label{cor:limit_ultra}
Let \(\Struct^{(M)}\) be the structure of \(M\) experts. There exists an ultrafilter such that:
\begin{equation}
    \lim_{M \to \infty} \Th(\Struct^{(M)}) = \Th(\Struct_{\Ultrafilter}),
\end{equation}
where \(\Th(\cdot)\) is the first-order theory (set of all true sentences).
\end{corollary}

\subsection{CEC as Inverse Limit}
\begin{theorem}[CEC as Inverse Limit]\label{thm:cec_limit}
Let \(\{\Struct_i\}_{i \in I}\) be expert structures. For any non-principal ultrafilter \(\Ultrafilter\) on \(I\), the \(\Struct_{\Ultrafilter}\) is \textbf{the inverse limit of the CEC}---that is:
\begin{enumerate}[label=(\roman*),itemsep=3pt]
    \item For each claim \(x \in \ClaimSpace\), \(x\) belongs to the CEC iff it belongs to the \(\Ultrafilter\)-limit of CEC membership.
    \item The ultraproduct \(\Struct_{\Ultrafilter}\) is \textbf{\(\aleph_1\)-saturated}: it realizes all compatible types.
    \item The ultraproduct is the unique structure simultaneously extending all finite expert configurations while preserving the \(\Ultrafilter\)-consensus.
\end{enumerate}
\end{theorem}
\begin{proof}
\rigorous Construction:
\begin{enumerate}[label=(\arabic*),itemsep=3pt]
    \item Form the directed set \(I\) of finite expert subsets.
    \item For each \(F \in I\), let \(\Struct_F\) be the consensus structure of experts in \(F\).
    \item Choose an ultrafilter \(\Ultrafilter\) on \(I\) containing all \textbf{tail sets} \(\{F \in I \mid F \supseteq F_0\}\).
    \item The ultraproduct \(\Struct_{\Ultrafilter}\) is the CEC limit.
\end{enumerate}

(i) follows from Łoś (Theorem~\ref{thm:los_scx}). (ii) is the \(\aleph_1\)-saturation of ultraproducts over countable index sets. (iii) combines (i) with the inverse limit property. \(\square\)
\end{proof}
\begin{remark}
\textbf{The CEC is the inverse limit of finite consensus, and the ultraproduct is its saturated completion.}
\end{remark}

\subsection{Triangle Theorem}
\begin{theorem}[Logical Foundation Triangle]\label{thm:triangle}
\begin{enumerate}[label=\textbf{\arabic*: },itemsep=8pt,leftmargin=*]
    \item \textbf{Galois (Solvability):}
    A claim \(x\) is auditable iff \(\Stab(\{x\}) \leq \AuditGrp\) is a \textbf{Solvable Group}. Equivalently, unauditability arises precisely when the audit group is unsolvable. \textbf{Foundation:} Galois-SCX~\cite{scx_galois}.
    \item \textbf{Compactness (Finite Transfer):}
    A claim \(x\) is auditable iff it is finitely auditable (Theorem~\ref{thm:compactness_audit}). Compactness provides the \textbf{logical transfer}: finite consensus implies infinite consensus (Theorem~\ref{thm:finite_auditability}).
    \item \textbf{Situs (Lipschitz Regularity):}
    The Situs Lipschitz condition ensures that the expert verdict function is Lipschitz-continuous, enabling finite sampling to approximate the global consensus. \textbf{Foundation:} SCX Core Theorems~\cite{scx2026theorems}.
\end{enumerate}
\begin{itemize}[itemsep=3pt]
    \item \textbf{Galois \(\to\) Compactness:} Solvability of the audit group guarantees that finite expert panels can be extended to infinite ones while preserving Galois structure.
    \item \textbf{Compactness \(\to\) Situs:} Compactness guarantees that finite sampling suffices; Situs Lipschitz ensures that the finite samples are representative---the \textbf{finite-to-infinite transfer is both logically guaranteed and geometrically stable}.
    \item \textbf{Situs \(\to\) Galois:} Lipschitz continuity constrains the orbit structure of the audit group---large orbits with small Lipschitz constants correspond to solvable subgroups, while small orbits with large Lipschitz constants \(\iff\) unsolvable subgroups.
\end{itemize}
\end{theorem}
\begin{proof}[Proof of Triangle Coherence]
\rigorous Proof that SCX's three pillars are \textbf{mutually entailing}:

\textbf{Forward direction:} Galois solvability (subgroup lattice with abelian quotients) \(\implies\) Compactness (finite submodel property via abelian extensions) \(\implies\) Situs Lipschitz (controlled growth of the CEC under expert addition).

\textbf{Reverse direction:} Galois Fails (unsolvable) \(\implies\) Compactness Fails (no finite-to-infinite transfer via \(A_5\) obstruction, cf.~\cite{scx_galois}); Compactness Fails \(\implies\) Situs Fails (no Lipschitz bound covers all experts); Situs Fails \(\implies\) Galois unsolvable.

The triangle is a \textbf{vicious/virtuous cycle}: either all three hold (auditability) or all three fail (unauditability). \(\square\)
\end{proof}

\subsection{Inverse Systems}
\begin{definition}[Inverse System]
Let \(\mathcal{F}\) be the directed set of finite expert subsets \(I\) (ordered by \(\subseteq\)). For each \(F \in \mathcal{F}\), there is a finite consensus structure \(\Struct_F\). For \(F \subseteq G\), there is a restriction map \(\pi_{G,F}: \Struct_G \to \Struct_F\). The system \((\{\Struct_F\}_{F \in \mathcal{F}}, \{\pi_{G,F}\}_{F \subseteq G})\) is an \textbf{inverse system}.
\end{definition}

\begin{definition}[Inverse Limit of CEC]
The \textbf{inverse limit} of the inverse system \((\Struct_F, \pi_{G,F})\) is:
\begin{equation}
    \varprojlim_{F \in \mathcal{F}} \Struct_F
    = \left\{ (a_F)_{F \in \mathcal{F}} \in \prod_{F \in \mathcal{F}} |\Struct_F|
    \;\middle|\; \forall F \subseteq G,\; \pi_{G,F}(a_G) = a_F \right\}.
\end{equation}
\end{definition}

\begin{theorem}[Inverse Limit-Ultraproduct Embedding]\label{thm:invlim_ultra}
For any ultrafilter \(\Ultrafilter\) on \(\mathcal{F}\) containing all tail sets \(\{G \in \mathcal{F} \mid G \supseteq F\}\):
\begin{equation}\label{eq:embedding}
    \varprojlim_{F \in \mathcal{F}} \Struct_F \;\hookrightarrow\; \prod_{F \in \mathcal{F}} \Struct_F / \Ultrafilter,
\end{equation}
and the embedding is elementary (preserves all first-order truth).
\end{theorem}
\begin{proof}
\rigorous Define the embedding \(\iota\) by: for \((a_F)_{F \in \mathcal{F}} \in \varprojlim \Struct_F\), set \(\iota((a_F)) = [(a_F)_{F \in \mathcal{F}}]_{\Ultrafilter}\). Injectivity: if \((a_F) \neq (b_F)\), then \(\{F \mid a_F = b_F\}\) is contained in a proper principal set \(F_0\)---which is not in the non-principal \(\Ultrafilter\). Hence the equivalence classes differ.

\textbf{Elementarity:} For any sentence \(\varphi\),
\begin{align*}
    \varprojlim \Struct_F \models \varphi
    &\iff \exists F_0 \forall F \supseteq F_0,\; \Struct_F \models \varphi
    \quad\text{(inverse limit semantics)}\\
    &\iff \{F \in \mathcal{F} \mid \Struct_F \models \varphi\} \text{ is cofinal}\\
    &\iff \{F \in \mathcal{F} \mid \Struct_F \models \varphi\} \in \Ultrafilter
    \quad\text{(by choice of }\Ultrafilter\text{)}\\
    &\iff \prod \Struct_F / \Ultrafilter \models \varphi
    \quad\text{(Łoś Theorem)}.
\end{align*}
Thus \(\iota\) is an elementary embedding. \(\square\)
\end{proof}

\begin{corollary}[Two Views of the CEC]
\begin{enumerate}[label=(\arabic*),itemsep=3pt]
    \item \textbf{Inverse limit view:} \(\CEC_{\infty} \cong \varprojlim_{F \in \mathcal{F}} \Struct_F\) (the CEC as coherent family of finite consensuses);
    \item \textbf{Ultraproduct view:} \(\CEC_{\infty} \cong \prod_{F \in \mathcal{F}} \Struct_F / \Ultrafilter\) (the CEC as majority-consensus limit).
\end{enumerate}
\end{corollary}

\subsection{Compactness Proof via Ultraproducts}
\begin{theorem}[Compactness Proof --- SCX Version]\label{thm:compactness_ultra}
Let \(\Gamma\) be a set of sentences in \(\Lang_{\mathrm{audit}}\). If every finite subset of \(\Gamma\) has an expert model, then \(\Gamma\) itself has an expert model.

\begin{center}
\textbf{SCX interpretation: If every finite set of claims can be consistently audited, then the entire claim space is auditable.}
\end{center}
\end{theorem}
\begin{proof}
\rigorous A complete, self-contained proof of the Compactness Theorem within the SCX framework.

\textbf{Step 1: Index set.} Let \(I = \mathcal{P}_{\mathrm{fin}}(\Gamma)\) be the set of all finite subsets of \(\Gamma\). For each \(\Delta \in I\), by hypothesis there exists an expert structure \(\Struct_\Delta\) such that \(\Struct_\Delta \models \Delta\).

\textbf{Step 2: Ultrafilter construction.} For each \(\varphi \in \Gamma\), define:
\begin{equation}
    A_\varphi = \{\Delta \in I \mid \varphi \in \Delta\}.
\end{equation}
The collection \(\{A_\varphi\}_{\varphi \in \Gamma}\) has the \textbf{finite intersection property}: for any \(\{\varphi_1,\ldots,\varphi_k\} \subseteq \Gamma\), \(A_{\varphi_1} \cap \cdots \cap A_{\varphi_k} = \{\Delta \in I \mid \varphi_1,\ldots,\varphi_k \in \Delta\} \neq \emptyset\) (since \(\{\varphi_1,\ldots,\varphi_k\}\) is itself a finite subset). By the ultrafilter lemma, there exists an ultrafilter \(\Ultrafilter\) on \(I\) containing all sets \(A_\varphi\).

\textbf{Step 3: Ultraproduct model.} Define \(\Struct^* = \prod_{\Delta \in I} \Struct_\Delta / \Ultrafilter\). For any \(\varphi \in \Gamma\):
\begin{align*}
    \{\Delta \in I \mid \Struct_\Delta \models \varphi\}
    &\supseteq \{\Delta \in I \mid \varphi \in \Delta\}
    = A_\varphi \in \Ultrafilter.
\end{align*}
By Łoś (Theorem~\ref{thm:los_scx}), \(\Struct^* \models \varphi\). Since \(\varphi\) was arbitrary in \(\Gamma\), \(\Struct^* \models \Gamma\).

\textbf{Step 4: SCX interpretation.} The model \(\Struct^*\) is the \textbf{ultraproduct consensus}. Every \(\varphi \in \Gamma\) holds in \(\Struct^*\) because it holds for an ultrafilter-large set of finite configurations. The ultrafilter acts as the ``voting rule''---sentences \(\varphi\) in \(A_\varphi\) always win the vote, and by construction \(A_\varphi \in \Ultrafilter\) for every \(\varphi \in \Gamma\). \(\square\)
\end{proof}
\begin{remark}
This proof is \textbf{completely constructive modulo the ultrafilter}: it tells us that the consensus exists as a mathematical object (the ultraproduct), but does not provide an algorithm to compute it---this is the essence of the Compactness Theorem.
\end{remark}

\subsection{Applications to Disagreement Detection}
\subsubsection{Structural Disagreement Modes}
\begin{enumerate}[leftmargin=*,nosep]
    \item \textbf{Galois perspective --- What causes disagreement?} Expert verdicts form an \(M \times N\) matrix \(V \in \{0,1\}^{M \times N}\). Disagreement arises when \(V_{ik} \neq V_{jk}\) for experts \(i, j\) on claim \(k\). If the disagreement pattern has a 5-cycle structure, the audit group contains \(A_5\)---unauditability. Disagreement that is ``perfectly symmetric'' (all pairwise disagreements equal) is precisely the condition for unsolvability.
    \item \textbf{Compactness perspective --- How does disagreement propagate?} The \(k=3\) theorem: if any triple of experts disagrees, Compactness propagates this to global disagreement. The triple disagreement structure determines whether the entire expert panel can reach consensus.
    \item \textbf{Situs perspective --- Where does disagreement concentrate?} Situs Lipschitz bounds the ``geography'' of disagreement on the claim manifold---disagreement concentrates on regions of high curvature. When the curvature exceeds the Lipschitz bound, the Situs condition is violated, triggering the Galois \(A_5\) obstruction.
\end{enumerate}

\subsubsection{Disagreement Taxonomy}
\textbf{Type I (Cyclic Disagreement, \(\mathbb{Z}_p\)):} Experts disagree cyclically. Examples: political polling (left-right), GPS multilateration (satellite disagreements). The cyclic structure is abelian, hence solvable---auditing is possible, but the number of distinct opinions grows with \(p\).

\textbf{Type II (Multi-dimensional disagreement, \(\mathbb{Z}_2 \times \cdots \times \mathbb{Z}_2\)):} CIFAR-10 classification (10 classes), CT diagnosis (3 abnormality types), crystallography (X-ray/NMR/Cryo-EM). Each dimension adds a binary disagreement axis. The group is abelian (elementary abelian 2-group), hence solvable---auditing scales exponentially with dimension.

\textbf{Type III (Paired disagreement, \(\mathbb{Z}_2\)):} DFT vs.\ experiment, LES vs.\ DNS, RCT vs.\ observational study. Simple pairwise disagreement; trivially solvable.

\subsubsection{Disagreement Leading to \(A_5\)}
\textbf{Type IV (Full symmetric disagreement, \(S_M\)):} All experts disagree with all others equally. In social choice: ``market of ideas'' with perfect competition. In science: ``Kuhnian paradigm clash'' with 5 mutually incommensurable schools. This realizes \(S_5\) and hence \(A_5\).

\textbf{Type V (Chained distillation):} Model \(A\) distills to \(B\), \(B\) to \(C\), etc. The iterative nature of distillation creates compounding disagreement---exactly the condition for unsolvability in the distillation-hallucination framework.

\textbf{Type VI (Ideological disagreement, \(A_5\)):} Five mutually opposed frameworks (e.g., Keynesian \(\leftrightarrow\) Monetarist \(\leftrightarrow\) MMT \(\leftrightarrow\) Austrian \(\leftrightarrow\) Marxist), five competing interpretations, five ``schools of thought.'' This is the natural home of \(A_5\) in real-world disagreement.

\subsubsection{The Verdict Protocol}
The SCX Verdict on a claim is: majority consensus + CEC consistency + Situs Lipschitz bound. If all three hold, the claim is auditable. If any fails (ClinVar-style conflict), the Verdict is \textbf{Inconclusive}---flag for human review.

\subsection{Honest Strike: The Non-Constructive Nature of Compactness}
\honestcrit{Compactness is a pure existence theorem---it proves that an infinite consensus EXISTS but provides no algorithm to FIND it.

The ultraproduct model \(\Struct^*\) is a non-constructive object whose existence relies on the Axiom of Choice (via the ultrafilter lemma). This has two consequences:
\begin{itemize}[itemsep=3pt]
    \item Compactness guarantees \textbf{Proof} of existence (like Henkin's completeness proof)---but the proof is non-constructive.
    \item The existence proof, while valid, does not provide an effective procedure for finding the consensus. This is the \textbf{essential tension}: SCX proves that consensus is possible, but does not tell you how to achieve it.
\end{itemize}

\textbf{Response:} This is not a bug but a feature. SCX proves that \textbf{no algorithmic barrier} prevents consensus (the consensus exists mathematically), but leaves the \textit{construction} to the practitioner. This mirrors Gödel's Completeness Theorem---logical existence does not imply algorithmic computability.

\textbf{Heuristic resolution:} The \(k=3\) theorem provides a constructive pathway---check triples, use transitivity, and Compactness guarantees the extension exists. But the \textbf{actual computation} of the consensus may require resources beyond what is feasible.}

\honestcrit{The ultraproduct model is non-standard---it may contain ``infinite expert'' configurations that have no finite realization.}

The SCX consensus in \(\Struct_{\Ultrafilter}\) is \textbf{infinitely precise}---claims are either in the CEC or not, with no ambiguity. Real SCX must operate with \textbf{finite precision}---a claim has an associated Cercis score in \([0,1]\) rather than a binary CEC membership.

\honestcrit{Compactness guarantees consistency, but consistency is not truth. A consistent set of expert verdicts may be systematically wrong.}

\textbf{Distinction:} Compactness ensures \textbf{Consistency} (no logical contradictions among expert verdicts), but does not guarantee \textbf{correctness} (correspondence to ground truth).
\begin{itemize}[itemsep=2pt]
    \item Compactness ensures that expert panels cannot produce irreconcilable disagreements---every finite inconsistency can be resolved.
    \item The resolution may be wrong (systematic bias shared by all experts).
    \item Compactness \textbf{Fails} precisely when the audit group is unsolvable---then there exist finite inconsistencies that cannot be resolved even in the infinite limit.
\end{itemize}

\section{Appendix: Model Theory Background}
\subsection{First-Order Logic}
\begin{definition}[First-Order Language]
A first-order language \(\Lang\) consists of:
\begin{itemize}[itemsep=2pt]
    \item Logical symbols: variables \(\{v_1, v_2, \ldots\}\), connectives \(\neg, \land, \lor, \to\), quantifiers \(\forall, \exists\), and equality \(=\).
    \item Non-logical symbols: constant symbols, function symbols, and relation symbols (predicates).
\end{itemize}
\end{definition}

\begin{definition}[Structure]
A \textbf{structure} \(\Struct\) for \(\Lang\) consists of a non-empty universe \(|\Struct|\) together with interpretations of all non-logical symbols.
\end{definition}

\begin{definition}[Theory and Model]
A \textbf{sentence} is a formula with no free variables. A \textbf{theory} \(T\) is a set of sentences. \(\Struct \models T\) means \(\Struct\) is a model of \(T\).
\end{definition}

\subsection{Compactness Theorem}
\begin{theorem}[Compactness Theorem --- Standard Form]
Let \(\Gamma\) be a set of first-order sentences. Then:
\begin{equation}
    \text{every finite } \Delta \subseteq \Gamma \text{ has a model}
    \;\implies\;
    \Gamma \text{ has a model}.
\end{equation}
\end{theorem}
\begin{proof}[Proof Approaches]
Two standard proofs:
\begin{enumerate}[label=(\arabic*),itemsep=2pt]
    \item \textbf{Henkin construction:} Extend \(\Gamma\) to a maximal consistent Henkin theory, then build a term model.
    \item \textbf{Ultraproduct proof:} Given models for each finite subset, take their ultraproduct (Theorem~\ref{thm:compactness_ultra}).
\end{enumerate}
\end{proof}

\subsection{Ultrafilters and Ultraproducts}
\begin{definition}[Ultrafilter]
A \textbf{filter} \(F \subseteq \mathcal{P}(I)\) satisfies: (i) \(\emptyset \notin F\), \(I \in F\); (ii) \(A, B \in F \implies A \cap B \in F\); (iii) \(A \in F, A \subseteq B \implies B \in F\). If additionally (iv) for every \(A \subseteq I\), exactly one of \(A\) and \(I \setminus A\) is in \(F\), then \(F\) is an \textbf{ultrafilter}.
\end{definition}

\begin{theorem}[Ultrafilter Lemma]
Every filter on \(I\) extends to an ultrafilter.
\end{theorem}

\begin{definition}[Ultraproduct]
Let \(\{\Struct_i\}_{i \in I}\) be \(\Lang\)-structures and \(\Ultrafilter\) an ultrafilter on \(I\). The ultraproduct \(\prod_{i \in I} \Struct_i / \Ultrafilter\) has universe \(\prod_i |\Struct_i|\) modulo the equivalence relation \((a_i) \sim (b_i) \iff \{i \mid a_i = b_i\} \in \Ultrafilter\), with relations interpreted via the \(\Ultrafilter\)-majority rule.
\end{definition}

\begin{theorem}[Łoś Theorem]
For any sentence \(\varphi\):
\[
    \prod_{i \in I} \Struct_i / \Ultrafilter \models \varphi
    \iff \{i \in I \mid \Struct_i \models \varphi\} \in \Ultrafilter.
\]
\end{theorem}

\subsection{Inverse Limits}
\begin{definition}[Inverse System]
A directed set \((I, \leq)\) and structures \(\{\Struct_i\}_{i \in I}\) with maps \(\pi_{ij}: \Struct_j \to \Struct_i\) for \(i \leq j\), satisfying \(\pi_{ii} = \mathrm{id}\) and \(\pi_{ij} \circ \pi_{jk} = \pi_{ik}\). The \textbf{inverse limit} \(\varprojlim \Struct_i\) is the set of coherent families \((a_i)_{i \in I}\) with \(\pi_{ij}(a_j) = a_i\).
\end{definition}

\subsection{Saturated Models}
\begin{definition}[Saturation]
A structure \(\Struct\) is \textbf{\(\kappa\)-saturated} if for every \(A \subseteq |\Struct|\) with \(|A| < \kappa\), every complete 1-type over \(A\) is realized in \(\Struct\).
\end{definition}

\begin{theorem}[Ultraproduct Saturation]
If \(I\) is countable and \(\Ultrafilter\) is non-principal, then \(\prod \Struct_i / \Ultrafilter\) is \(\aleph_1\)-saturated.
\end{theorem}
