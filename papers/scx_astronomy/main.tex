\begin{abstract}
Multi-messenger astronomy — the coordinated observation of astrophysical
sources through electromagnetic (EM), gravitational-wave (GW), neutrino,
and cosmic-ray channels — promises revolutionary insights into the most
extreme phenomena in the universe. However, joint inference across
heterogeneous messenger channels confronts a fundamental challenge:
each messenger is observed by instruments with distinct sensitivity,
systematic errors, and noise models, and their apparent disagreements can
originate from instrumental artifacts, modeling errors, or genuinely new
physics. Without a rigorous certification framework, the scientific
credibility of multi-messenger discoveries remains unquantified.

\medskip
\noindent\textbf{Core Results.} We establish three theorems that provide
mathematically certified bounds for multi-messenger inference within the
\SCX{} auditing framework.
\textbf{Theorem~1} (Fisher-Information-Weighted Joint Detection Guarantee):
For $M$ heterogeneous messenger channels, each with measurement precision
$\sigma_m^2$, the probability of jointly \textit{missing} a genuine
astrophysical event satisfies
$\Pbb(\text{miss}) \leq \exp(-2 \cdot (\sum_m 1/\sigma_m^2) \cdot \Delta^2)$,
where $\Delta$ is the multi-messenger detection gap and the sum
$\sum_m 1/\sigma_m^2$ is the total Fisher information. High-precision
channels (e.g., GW timing with $\sigma \sim 10^{-4}$\,s) dominate the
bound, correctly weighting each messenger by its information content.
\textbf{Theorem~2} (Instrumental Artifact vs.\ New Physics
Unidentifiability): For GW170817, the $\sim$1.74\,s GW–GRB delay admits
three observationally equivalent explanations — calibration error, GRB
model error, or modified gravity with $|\delta c|/c \sim 4 \times
10^{-16}$. We prove these are fundamentally unidentifiable from joint
data alone, via $\TV \leq \varepsilon$ for any pair. \textbf{Theorem~3}
(Situs Celestial Encoding with Relativistic Corrections): We construct
$\Phi_{\Situs}: S^2 \times \R^+ \to \R^D$ respecting Lorentzian
geometry with Shapiro delay and aberration corrections. A bilateral
Lipschitz bound translates localization uncertainty to parameter error:
$\norm{\Delta\hat{\boldsymbol{\theta}}} \leq L_{\mathrm{upper}} \cdot
\sigma_{\mathrm{loc}}$.

\medskip
\noindent\textbf{\Cercis{} Discovery Score.} $S = Q + \eta N$, where
$Q = Q_{\mathrm{loc}} + Q_{\mathrm{spec}}$ combines localization
precision and spectral fidelity, and $N$ quantifies discovery novelty.

\medskip
\noindent\textbf{Benchmarks.} GW170817/GRB 170817A, IceCube-170922A /
TXS 0506+056, and FRB campaigns combining ASKAP/CHIME/FAST.

\medskip
\noindent\textbf{Keywords:} multi-messenger astronomy, gravitational
waves, neutrinos, SCX auditing, \Cercis{} score, \Yajie{} consensus,
\Situs{} encoding, certified inference, 多信使天文学, 引力波, 中微子
\end{abstract}

\section{Introduction 引言}

\subsection{The Multi-Messenger Revolution 多信使革命}
\label{sec:intro-revolution}

On 2017 August 17, LIGO-Virgo detected GW170817, a binary neutron star
inspiral at $D_L \approx 40$\,Mpc \cite{abbott2017gw170817}. Within
1.74\,s, Fermi-GBM triggered on GRB 170817A \cite{goldstein2017grb170817a}.
The optical counterpart AT 2017gfo was identified in NGC 4993
\cite{coulter2017at2017gfo}, inaugurating \textbf{multi-messenger astronomy}
(\emph{多信使天文学}) — coordinated observation through fundamentally
different physical messengers:

\begin{enumerate}[label=(\textbf{M\arabic*})]
    \item \textbf{Electromagnetic (EM)}: Photons across the spectrum.
    \item \textbf{Gravitational waves (GW, 引力波)}: Metric perturbations
    $h \sim 10^{-21}$.
    \item \textbf{Neutrinos ($\nu$, 中微子)}: Weakly interacting particles
    from hadronic acceleration.
    \item \textbf{Cosmic rays (CR)}: Ultra-relativistic charged particles.
\end{enumerate}

Each messenger has distinct noise statistics: Poisson for EM,
matched-filtering SNR for GW, single-event Poisson for neutrinos. The
challenge of \textbf{certified joint inference} across these
heterogeneous channels is the central problem of this paper.

\subsection{The Certification Gap 认证差距}
\label{sec:intro-gap}

Existing analyses rely on: (1) ad-hoc coincidence significance by
multiplying $p$-values without accounting for systematics or differential
information content; (2) unstated auxiliary assumptions when messengers
disagree (e.g., the GW170817 timing delay, missing EM counterparts to
GW events).

The \SCX{} framework \cite{xi2025scx} formalizes each channel as an
\textbf{expert} $f_m$ with state-conditioned guarantees. This paper
adapts and extends \SCX{} to multi-messenger astronomy.

\section{Formalization of Multi-Messenger Inference 多信使推理的形式化}
\label{sec:formalization}

\subsection{Messenger Channels as Heterogeneous Experts}
\label{sec:form-messengers}

\begin{definition}[Messenger Channel]\label{def:messenger}
A messenger channel $m \in \Messengers = \{\EM, \GW, \nu, \CR\}$ has:
\begin{enumerate}[label=(\roman*)]
    \item measurement model $p_m(\mathbf{d}_m \mid \boldsymbol{\theta})$
    with $\boldsymbol{\theta} \in \Theta \subseteq \R^p$;
    \item noise characterization $\sigma_m^2(\boldsymbol{\theta})$;
    \item Fisher information $\FisherMat_m(\boldsymbol{\theta}) =
    \E[(\nabla_{\boldsymbol{\theta}} \ln p_m)(\nabla_{\boldsymbol{\theta}}
    \ln p_m)^\top]$.
\end{enumerate}
\end{definition}

\begin{definition}[Multi-Messenger Observation]\label{def:mm_obs}
Joint data $\mathcal{O} = (\mathbf{d}_1, \ldots, \mathbf{d}_M)$ with
$p(\mathcal{O} \mid \boldsymbol{\theta}) = \prod_{m=1}^M
p_m(\mathbf{d}_m \mid \boldsymbol{\theta})$ (conditional independence
given the source). Joint Fisher: $\FisherMat_{\mathrm{joint}} =
\sum_m \FisherMat_m$.
\end{definition}

\subsection{Instrument-Specific Noise Models}
\label{sec:form-noise}

\begin{definition}[EM — Poisson Photon Counting]\label{def:em_noise}
$N_{\EM} \sim \Poisson(\lambda_{\mathrm{src}} T +
\lambda_{\mathrm{bkg}} T)$. Precision: $\sigma_{\EM}^2 =
(\lambda_{\mathrm{src}} + \lambda_{\mathrm{bkg}})/(\lambda_{\mathrm{src}}^2 T)
= 1/(\SNR_{\EM}^2 \cdot \lambda_{\mathrm{src}})$.
\end{definition}

\begin{definition}[GW — Matched-Filtering SNR]\label{def:gw_noise}
$\SNR_{\GW}^2 = 4\int_0^\infty |\tilde{h}(f)|^2/S_n(f)\,df$.
Fisher: $[\FisherMat_{\GW}]_{ij} = 4\Re\int_0^\infty
\partial_i \tilde{h}^* \partial_j \tilde{h} / S_n(f)\,df$.
For BNS at 40\,Mpc: $\sigma_{\GW}^{t_c} \sim 10^{-3}$\,s.
\end{definition}

\begin{definition}[Neutrino — Single-Event Poisson]\label{def:nu_noise}
$\lambda_\nu = T\int dN/dE_\nu \cdot A_{\mathrm{eff}}^\nu(E_\nu)\,dE_\nu$.
$\SNR_\nu = \lambda_\nu / \sqrt{\lambda_\nu + \lambda_{\mathrm{atm}}}$.
For IceCube, $\lambda_\nu \ll 1$: rare-event Poisson statistics.
\end{definition}

\subsection{State Space and Expert Definitions}
\label{sec:form-states}

\begin{definition}[Multi-Messenger State]\label{def:mm_state}
$s = (\text{source class}, z, \text{multi-messenger signatures},
\text{environment}) \in \cS$.
\end{definition}

\begin{definition}[Multi-Messenger Expert]\label{def:mm_expert}
$f_m: \cD_m \to \cY_m$ maps data to $\hat{y}_m \in \{0,1\}$ and
$\hat{\boldsymbol{\theta}}_m$, characterized by $\alpha_m$ (FAR),
$\beta_m(\boldsymbol{\theta})$ (detection probability), and
$\sigma_m^2(\boldsymbol{\theta})$ (precision).
\end{definition}

\section{Theorem~1: Fisher-Information-Weighted Joint Detection
Guarantee 费希尔信息加权的联合检测保证}
\label{sec:thm1}

\subsection{Motivation}
\label{sec:thm1-motivation}

Standard SCX S1 gives $\exp(-2M\Delta^2)$, treating all experts equally.
In multi-messenger astronomy, GW timing ($\sigma \sim 10^{-4}$\,s) carries
$\sim 10^7 \times$ more Fisher information than EM ($\sigma \sim 0.05$\,s).
Equal-variance Hoeffding would dilute the bound. The Fisher-weighted
approach scales each channel by $w_m = 1/\sigma_m^2$.

\subsection{Assumptions}
\label{sec:thm1-assumptions}

\begin{assumption}[Conditional Independence]\label{asm:A1_mm}
$\hat{y}_m \indep \hat{y}_{m'} \mid \boldsymbol{\theta}$ for $m \neq m'$.
\end{assumption}

\begin{assumption}[Known Precision]\label{asm:A2_mm}
$\frac{1}{2} \leq \hat{\sigma}_m^2 / \sigma_m^2 \leq 2$ for each $m$.
\end{assumption}

\begin{assumption}[Unbiased Detectors]\label{asm:A3_mm}
$\E[\hat{y}_m \mid \text{source}] = \beta_m$,
$\E[\hat{y}_m \mid H_0] = \alpha_m$.
\end{assumption}

\begin{assumption}[Positive Detection Gap]\label{asm:A4_mm}
$\frac{1}{M}\sum_m (\beta_m - \alpha_m) \geq \Delta > 0$.
\end{assumption}

\subsection{Theorem Statement}
\label{sec:thm1-statement}

\begin{theorem}[Fisher-Information-Weighted Detection Guarantee]
\label{thm:fisher_hoeffding}
Let $M$ messenger channels observe a common source. Each channel $m$
has FAR $\alpha_m$, detection probability $\beta_m$, and Fisher
information $w_m = 1/\sigma_m^2$. Define:
\begin{equation}\label{eq:fisher_gap}
    \Delta_w = \frac{\sum_m w_m(\beta_m - \alpha_m)}{\sum_m w_m}.
\end{equation}
Under Assumptions~\ref{asm:A1_mm}--\ref{asm:A4_mm}:
\begin{equation}\label{eq:thm1_box}
    \boxed{\Pbb(\text{Type II error}) \leq
    \exp\!\left(-2 \cdot \left(\sum_{m=1}^M \frac{1}{\sigma_m^2}\right)
    \cdot \Delta^2\right).}
\end{equation}
More precisely, with $W_M = \sum_m w_m$ and $\overline{w^2} =
\frac{1}{M}\sum_m w_m^2$:
\begin{equation}\label{eq:thm1_tight}
    \boxed{\Pbb(\text{miss}) \leq
    \exp\!\left(-\frac{2 W_M^2 \Delta_w^2}{\sum_m w_m^2}\right)
    = \exp\!\left(-2 W_M \Delta_w^2 \cdot
    \frac{M}{1+\mathrm{CV}_w^2}\right).}
\end{equation}
When $\mathrm{CV}_w = 0$ (equal precisions), this reduces to
$\exp(-2M\Delta^2)$, recovering SCX Theorem~S1.
\end{theorem}
\rigorFull

\subsection{Proof}
\label{sec:thm1-proof}

\begin{proof}
Define $D_m = \ind{\hat{y}_m = 1}$ and weighted sum
$T_M = \sum_m w_m D_m$. Each $w_m D_m \in [0, w_m]$. By Hoeffding
\cite{hoeffding1963}:
$\Pbb(T_M - \E[T_M \mid \text{source}] \leq -\delta) \leq
\exp(-2\delta^2 / \sum_m w_m^2)$.
Setting $\delta = \sum_m w_m(\beta_m - \alpha_m) = W_M \Delta_w$
yields the bound. For equal precisions, $\sum_m w_m^2 = M w^2$ and
$W_M^2 = M^2 w^2$, so $W_M^2 / \sum_m w_m^2 = M = \sum_m 1/\sigma_m^2
\cdot \sigma^2$, recovering the stated form.
\end{proof}

\begin{corollary}[Sensitivity Scaling]\label{cor:sensitivity}
Adding any messenger channel never worsens the bound. Gain over best
channel: $G_M = W_M / \max_m w_m$.
\end{corollary}
\rigorFull

\begin{remark}[GW170817 Illustration]
\label{rem:gw170817_thm1}
GW: $\sigma_{t_c} \approx 5 \times 10^{-4}$\,s, $w_{\GW} \approx
4 \times 10^6$\,s$^{-2}$, $\beta_{\GW} \approx 0.999$. EM: $\sigma
\approx 0.05$\,s, $w_{\EM} \approx 400$\,s$^{-2}$. GW dominates:
$\Pbb(\text{miss}) \leq \exp(-2 \times 4\times10^6) \approx 0$.
\end{remark}

\section{Theorem~2: Instrumental Artifact vs.\ New Physics
Unidentifiability 仪器假象与新物理不可区分性}
\label{sec:thm2}

\subsection{Motivation: GW170817 Timing Tension}
\label{sec:thm2-motivation}

The 1.74\,s GW–GRB delay \cite{abbott2017gw170817,goldstein2017grb170817a}
admits three explanations:

\begin{enumerate}[label=(\textbf{H\arabic*})]
    \item \textbf{Calibration ($H_{\mathrm{cal}}$)}: Fermi-GBM flight-software
    latency $\sim 0.1$--$2$\,s.
    \item \textbf{Model error ($H_{\mathrm{model}}$)}: Physical jet
    propagation delay $\sim$1.74\,s through BNS ejecta.
    \item \textbf{Modified gravity ($H_{\mathrm{MG}}$)}:
    $\Delta t = (D_L/c)(c/c_{\GW} - 1)$, giving $|\delta c|/c \approx
    4 \times 10^{-16}$, far below the LIGO-Virgo bound of $10^{-15}$.
\end{enumerate}

\subsection{Assumptions}
\label{sec:thm2-assumptions}

\begin{assumption}[Finite Timing Resolution]\label{asm:B1_mm}
$\sigma_{\EM}^2 \gtrsim 10^{-2}$\,s$^2$; $\sigma_{\GW}^2 \lesssim
10^{-6}$\,s$^2$.
\end{assumption}

\begin{assumption}[Non-Identifiable Latent Parameters]\label{asm:B2_mm}
Each hypothesis introduces unconstrained latent parameters
($\delta_{\mathrm{cal}}$, $\tau_{\mathrm{jet}}$, $\alpha_{\mathrm{MG}}$).
\end{assumption}

\begin{assumption}[Equivalent Likelihoods]\label{asm:B3_mm}
For any pair $H_i, H_j$, there exist latent parameters making
$p(\mathbf{d} \mid H_i) = p(\mathbf{d} \mid H_j)$.
\end{assumption}

\subsection{Theorem Statement}
\label{sec:thm2-statement}

\begin{theorem}[Artifact vs.\ New Physics Unidentifiability]
\label{thm:artifact_unident_mm}
For $\mathcal{H} = \{H_{\mathrm{cal}}, H_{\mathrm{model}},
H_{\mathrm{MG}}\}$ and joint data $\mathbf{d} =
(\mathbf{d}_{\GW}, \mathbf{d}_{\EM})$:
\begin{equation}\label{eq:unident}
    \TV\big(p(\mathbf{d} \mid H_i, \boldsymbol{\xi}_i),\;
           p(\mathbf{d} \mid H_j, \boldsymbol{\xi}_j)\big) \leq
    \varepsilon(\sigma_{\EM}^2, D_L),
\end{equation}
with $\varepsilon \to 0$ as $\sigma_{\EM}^2 \to \infty$ or
$\sigma_{\GW}^2 \to 0$. No statistical test can distinguish these
hypotheses with power $> \frac{1}{2} + \varepsilon$.
\end{theorem}
\rigorFull

\subsection{Proof}
\label{sec:thm2-proof}

\begin{proof}
Under $H_{\mathrm{cal}}$: $t_{\EM}^{\mathrm{obs}} = t_{\GW} +
\delta_{\mathrm{cal}} + \varepsilon_{\EM}$ with $\delta_{\mathrm{cal}}
\sim p_{\mathrm{cal}}$. Under $H_{\mathrm{model}}$: $t_{\EM}^{\mathrm{obs}}
= t_{\GW} + \tau_{\mathrm{jet}} + \varepsilon_{\EM}$ with $\tau_{\mathrm{jet}}
\sim p_{\mathrm{model}}$. By setting $p_{\mathrm{cal}} = p_{\mathrm{model}}$,
the observable $\Delta t$ distributions become identical in the limit
$\sigma_{\GW}^2, \sigma_{\EM}^2 \to 0$.

For $H_{\mathrm{MG}}$: $\Delta t_{\mathrm{true}} = (D_L/c)\alpha_{\mathrm{MG}}$.
For any $\Delta t$, setting $\alpha_{\mathrm{MG}} = c\Delta t/D_L$ or
$\tau_{\mathrm{jet}} = \Delta t$ yields identical likelihoods.
$\TV(H_{\mathrm{MG}}, H_{\mathrm{model}}) = 0$ with unrestricted
parameter spaces. By Neyman-Pearson, the most powerful test has
$1-\beta \leq \alpha + \varepsilon$. For $\alpha \sim 0.05$,
$\varepsilon < 0.95$ implies unidentifiability.
\end{proof}

\begin{corollary}[Declared Assumptions Requirement]\label{cor:declare}
Any claim about multi-messenger timing must declare which competing
explanation is ruled out and on what basis. \SCX{} audit requires
explicit: (i) which hypothesis is assumed false, (ii) the justification
(e.g., Fermi-GBM timing verified to $\pm 2$\,ms via pulsar timing).
\end{corollary}
\rigorFull

\begin{remark}[Ensemble Extension]
\label{rem:scope}
Pointwise unidentifiability does not preclude ensemble identifiability:
$N_{\mathrm{events}}$ observations of consistent $\sim$1.74\,s delays
would distinguish universal $H_{\mathrm{MG}}$ from random
$H_{\mathrm{model}}$. Theorem~\ref{thm:fisher_hoeffding} quantifies
the accumulation: Fisher information $\propto N_{\mathrm{events}} W_M$.
\end{remark}

\section{Theorem~3: Situs Celestial Encoding with Relativistic
Corrections 带相对论修正的Situs天球编码}
\label{sec:thm3}

\subsection{Motivation}
\label{sec:thm3-motivation}

Celestial coordinates $(\alpha, \delta, z)$ live on the past light cone.
We require $\Phi_{\Situs}: S^2 \times \R^+ \to \R^D$ that is bilaterally
Lipschitz, incorporates Shapiro delay and aberration, and preserves
causal structure.

\subsection{Assumptions}
\label{sec:thm3-assumptions}

\begin{assumption}[FLRW Metric]\label{asm:C1_mm}
Flat $\LCDM$, Planck 2018: $H_0 = 67.4$\,km\,s$^{-1}$\,Mpc$^{-1}$,
$\Omega_m = 0.315$ \cite{planck2018}.
\end{assumption}

\begin{assumption}[Weak-Field Potentials]\label{asm:C2_mm}
$|\Phi|/c^2 \ll 1$, $|\Psi|/c^2 \ll 1$. Linear Shapiro delay.
\end{assumption}

\begin{assumption}[Bandlimit]\label{asm:C3_mm}
Angular: $\ell_{\max}$, $\theta_{\min} \approx \pi/\ell_{\max}$.
Radial: $K_r$ Fourier frequencies up to $\omega_{\max}$.
\end{assumption}

\subsection{Situs Construction}
\label{sec:thm3-construction}

\begin{definition}[Relativistic Situs Encoding]\label{def:situs_rel}
$\Phi_{\Situs} = (\Phi_1, \Phi_2, \Phi_3)$ where:

\textbf{C1 — Spherical Harmonics:}
$\Phi_1(\hat{\mathbf{n}}) = \{Y_{\ell m}(\hat{\mathbf{n}})\}_{\ell=0:\ell_{\max},
|m|\leq\ell} \in \R^{(\ell_{\max}+1)^2}$.

\textbf{C2 — Radial Fourier with Relativistic Correction:}
$\Phi_2(z; \hat{\mathbf{n}}, t_{\mathrm{obs}}) = \{\cos(\omega_k
\ln(1+z) + \delta\phi_k), \sin(\omega_k \ln(1+z) + \delta\phi_k)
\}_{k=1}^{K_r}$ where $\omega_k = 2\pi k / \ln(1+z_{\max})$ and:
\begin{equation}\label{eq:rel_phase}
    \delta\phi_k = \omega_k \cdot \Big[
    \frac{\Delta t_{\mathrm{Shapiro}}}{t_H(z)} +
    \frac{\mathbf{v}_{\oplus} \cdot \hat{\mathbf{n}}}{c} +
    \sum_p \frac{2GM_p}{c^3} \ln\frac{R_p +
    \hat{\mathbf{n}} \cdot \mathbf{R}_p}{R_p}\Big],
\end{equation}
with $t_H(z) = \int_0^z dz'/H(z')$ the lookback time, $\mathbf{v}_\oplus$
Earth's barycentric velocity, and the sum over solar-system bodies.

\textbf{C3 — Scale-Attenuated Depth:}
$\Phi_3(z) = (e^{-\chi(z)/\chi_0}, e^{-D_L(z)/D_{L,0}}, \ln(1+z))
\in \R^3$.

Total dimension: $D = (\ell_{\max}+1)^2 + 2K_r + 3$.
\end{definition}

\subsection{Theorem Statement}
\label{sec:thm3-statement}

\begin{theorem}[Relativistic Situs Lipschitz Bound]
\label{thm:situs_relativistic}
For any $\mathbf{p}_1 = (\hat{\mathbf{n}}_1, z_1)$,
$\mathbf{p}_2 = (\hat{\mathbf{n}}_2, z_2)$:
\begin{equation}\label{eq:lipschitz}
    \boxed{L_{\mathrm{lower}} \cdot d_{\mathrm{phys}} \leq
    \norm{\Phi_{\Situs}(\mathbf{p}_1) - \Phi_{\Situs}(\mathbf{p}_2)}
    \leq L_{\mathrm{upper}} \cdot d_{\mathrm{phys}},}
\end{equation}
where $d_{\mathrm{phys}}^2 = D_A(z_1)D_A(z_2)\theta_{12}^2 +
(\chi(z_2)-\chi(z_1))^2$, and:
\begin{equation}\label{eq:lipschitz_consts}
    L_{\mathrm{upper}} = \sqrt{\frac{(\ell_{\max}+1)^2(2\ell_{\max}+1)}
    {3} + 2K_r^3 + 3},\quad
    L_{\mathrm{lower}} = \frac{1}{L_{\mathrm{upper}} \cdot
    \sqrt{1 + \max_z (c/H(z)D_A(z))^2}}.
\end{equation}

\textbf{Corollary — Localization $\to$ Parameter Error:}
\begin{equation}\label{eq:loc_to_param}
    \boxed{|\Delta\hat{\theta}_i| \leq L_{\mathrm{upper}} \cdot
    \sigma_{\mathrm{loc}} \cdot \max_{\mathbf{p}}
    \norm{\nabla_{\mathbf{p}}\theta_i}.}
\end{equation}
\end{theorem}
\rigorFull

\subsection{Proof Sketch}
\label{sec:thm3-proof}

\begin{proof}[Proof Sketch]
Angular: $\norm{\Delta\Phi_1}^2 = \sum_{\ell} \frac{2\ell+1}{2\pi}
[1 - P_\ell(\cos\theta_{12})] \leq \theta_{12}^2 \cdot
(\ell_{\max}+1)^2(2\ell_{\max}+1)/3$.
Radial: $\norm{\Delta\Phi_2}^2 \leq 2\sum_k (\omega_k\Delta\ln(1+z)
+ \Delta\delta\phi_k)^2 \leq 2K_r^3 \cdot
(H(z)/c(1+z))^2 \cdot \Delta\chi^2$ plus $O(\theta_{12})$ relativistic
corrections. Combined: $L_{\mathrm{upper}}$ as stated. Lower bound from
Riesz basis property. Parameter error via Taylor expansion and Lipschitz.
\end{proof}
\rigorSketch

\begin{corollary}[Multi-Messenger Localization Gain]
\label{cor:mm_loc_gain}
$\sigma_{\mathrm{loc}}^{\mathrm{combined}} = (\sum_m
1/\sigma_{\mathrm{loc},m}^2)^{-1/2} \leq \min_m \sigma_{\mathrm{loc},m}$.
Parameter error: $|\Delta\hat{\theta}_i| \leq L_{\mathrm{upper}}
\cdot \sigma_{\mathrm{loc}}^{\mathrm{combined}} \cdot
\max \norm{\nabla_{\mathbf{p}}\theta_i}$.
\end{corollary}
\rigorFull

\begin{remark}[GW170817 Values]
$\ell_{\max} = 180$, $K_r = 32$: $L_{\mathrm{upper}} \approx 1393$.
Optical: $\sigma_{\mathrm{loc}} \approx 5 \times 10^{-7}$\,rad,
$|\Delta\hat{\theta}| \leq 7 \times 10^{-4}$. GW-only:
$\sigma_{\mathrm{loc}} \approx 0.093$\,rad,
$|\Delta\hat{\theta}| \leq 130$ — a $2 \times 10^5$ degradation.
\end{remark}

\section{\Cercis{} Discovery Score for Multi-Messenger Events
多信使事件的Cercis发现评分}
\label{sec:cercis}

\subsection{Definition}
\label{sec:cercis-def}

\begin{definition}[Multi-Messenger \Cercis{} Score]\label{def:cercis_mm}
\begin{equation}\label{eq:cercis_main}
    \boxed{S = Q + \eta \cdot N,}
\end{equation}
where:
\begin{itemize}
    \item $Q = Q_{\mathrm{loc}} + Q_{\mathrm{spec}}$:
    \begin{itemize}
        \item $Q_{\mathrm{loc}} = -\log_{10}(\Omega_{90\%} /
        \Omega_{\mathrm{ref}})$ with $\Omega_{\mathrm{ref}} = 1$\,deg$^2$.
        \item $Q_{\mathrm{spec}} = \sum_{m} \SNR_m / \SNR_m^{\mathrm{ref}}$,
        $\SNR_{\GW}^{\mathrm{ref}} = 8$, $\SNR_{\EM}^{\mathrm{ref}} = 5$,
        $\SNR_\nu^{\mathrm{ref}} = 5$.
    \end{itemize}
    \item $N = \sum_j w_j N_j$: discovery novelty.
    \begin{itemize}
        \item $N_{\mathrm{messenger}}$: new messenger combinations
        (first GW+EM: $w=5$; first GW+$\nu$: $w=5$; routine: $w=0.5$).
        \item $N_{\mathrm{class}}$: new source classes (BNS-sGRB: $w=3$;
        blazar-$\nu$: $w=3$).
        \item $N_{\mathrm{physics}}$: new physics ($c_{\GW}/c$: $w=2$;
        NS EOS: $w=1$).
    \end{itemize}
    \item $\eta_t = \eta_0 \cdot \exp(-t/\tau)$, $\tau \sim 2$\,yr.
\end{itemize}
\end{definition}

\subsection{Interpretation}
\label{sec:cercis-interpret}

$Q_{\mathrm{loc}}$ penalizes poor localization (GW-only 100\,deg$^2$:
$-2.0$; sub-arcsecond: $+10.1$). $Q_{\mathrm{spec}}$ rewards loud
detections. $N$ captures first-discovery premium. $\eta$ ensures
time-decaying novelty.

% 7. YAJIE CONSENSUS
% ======================================================================
\section{\Yajie{} Multi-Messenger Consensus 雅洁多信使共识}
\label{sec:yajie}

\subsection{Weighted Likelihood Combination}
\label{sec:yajie-lik}

\begin{definition}[Yajie Posterior]\label{def:yajie_post}
\begin{equation}\label{eq:yajie_post}
    p_{\Yajie}(\boldsymbol{\theta} \mid \{\mathbf{d}_m\}) \propto
    \pi(\boldsymbol{\theta}) \cdot \prod_{m=1}^M
    p_m(\mathbf{d}_m \mid \boldsymbol{\theta})^{w_m},
\end{equation}
with $w_m = (\SNR_m^2 / \sigma_{m,\mathrm{sys}}^2) /
\sum_{m'} (\SNR_{m'}^2 / \sigma_{m',\mathrm{sys}}^2)$.
\end{definition}

\subsection{Consensus Protocol}
\label{sec:yajie-protocol}

\begin{algorithm}[H]
\caption{\Yajie{} Multi-Messenger Consensus}
\begin{algorithmic}[1]
\State \textbf{Input:} Data $\{\mathbf{d}_m\}$, priors $\pi$
\For{each channel $m$}
    \State Compute $\SNR_m$, likelihood, $\sigma_{m,\mathrm{sys}}$
\EndFor
\State Compute weights $w_m$; joint posterior $p_{\Yajie}$
\State Compute Bayes factor $B_{10}$
\If{$B_{10} > B_{\mathrm{thresh}}$ and $M_{\mathrm{det}} \geq 2$}
    \State $D \leftarrow 1$; $\Omega_{90\%}$ from $p_{\Yajie}$
    \State $S \leftarrow Q_{\mathrm{loc}} + Q_{\mathrm{spec}} + \eta N$
\Else
    \State $D \leftarrow 0$
\EndIf
\State \Return $(D, S, \Omega_{90\%}, B_{10})$
\end{algorithmic}
\end{algorithm}

\subsection{Tension Resolution}
\label{sec:yajie-tension}

Per Theorem~\ref{thm:artifact_unident_mm}, when channels disagree:
(1) Declare which competing hypothesis is ruled out and why.
(2) Down-weight non-detections via upper limits if attributed to
selection effects. (3) Audit: each declared assumption enters the
permanent \Cercis{} trail.

\section{Discussion 讨论}
\label{sec:discussion}

\subsection{Summary}
\label{sec:disc-summary}

Three theorems: (1) Fisher-weighted Hoeffding: $\Pbb(\text{miss}) \leq
\exp(-2 \cdot (\sum_m 1/\sigma_m^2) \cdot \Delta^2)$, correctly
weighting heterogeneous messengers; (2) GW170817 unidentifiability:
calibration/model/gravity indistinguishable without declared
assumptions; (3) Relativistic Situs: Lipschitz bound translating
$\sigma_{\mathrm{loc}}$ to parameter error.

\subsection{Comparison}
\label{sec:disc-compare}

Complements Bayesian inference \cite{ashton2019bilby} with worst-case
guarantees; replaces ad-hoc $p$-value coincidence \cite{icecube_txs}
with principled Fisher weighting; provides per-event confidence bounds
beyond rate projections \cite{abbott2021gw170817_rate}.

\subsection{Limitations 局限性}
\label{sec:disc-limitations}

\begin{enumerate}[label=(\textbf{L\arabic*})]
    \item Conditional independence may be violated by shared calibration
    systematics (extendable via correlation matrices).
    \item Worst-case bound is conservative for loud events but valuable
    in marginal-SNR regimes.
    \item Pointwise unidentifiability; ensemble extension pending.
    \item Situs dimensionality $D \sim \ell_{\max}^2$ infeasible for
    sub-arcsecond resolution.
    \item Cosmic-ray channels treated superficially (magnetic deflection
    adds $E$-dependent systematic).
\end{enumerate}

\subsection{Open Problems 开放问题}
\label{sec:disc-open}

\begin{openproblem}[Ensemble Identifiability]
Prove $N_{\min}$ for distinguishing $H_{\mathrm{MG}}$ from
$H_{\mathrm{model}}$ across an ensemble of GW170817-like events.
\end{openproblem}

\begin{openproblem}[Cosmic-Ray Certification]
Extend Theorem~\ref{thm:fisher_hoeffding} with
$w_{\CR} = 1/(\sigma_{\mathrm{stat}}^2 + \sigma_{\mathrm{deflection}}^2)$.
\end{openproblem}

\begin{openproblem}[Real-Time \Spring{} Triggering]
Adapt \Spring{} \cite{spring_config} to sub-minute multi-messenger
alert decisions with Robbins-Monro convergence.
\end{openproblem}
