\subsubsection{2.2 Three Attack Paths}\label{three-attack-paths}

\paragraph{Attack Path 1: Triviality Attack (``Isn't this just conditional probability?'')}\label{attack-path-1-triviality-attack}

\begin{itemize}
\tightlist
\item
  SCX's three Propositions: Prop 1 is mathematically almost trivial
\item
  Reviewer: ``Conditional risk is a concept from Kolmogorov (1933). What's new here?''
\item
  \textbf{Defense}: Prove \textbf{the cost of ignoring state-conditionality in real material systems}---EGP P1's 9.7$\times$ per-batch RMSE difference
\end{itemize}

\paragraph{Attack Path 2: Patchwork Attack (``Clustering + model selection + active learning pieced together with a new name'')}\label{attack-path-2-patchwork-attack}

\begin{itemize}
\tightlist
\item
  Reviewer: ``State discovery = clustering. Expert reliability = per-cluster accuracy. Data value = heuristic product.''
\item
  \textbf{Defense}: Two-layer descriptors (ErrorDrivenEncoder) are the core---the 2.3$\times$ improvement from F1 0.253 $\rightarrow$ 0.585 proves this is not trivial clustering
\end{itemize}

\paragraph{Attack Path 3: Castle in the Air Attack (``No real experimental verification'')}\label{attack-path-3-castle-in-the-air-attack}

\begin{itemize}
\tightlist
\item
  Reviewer: ``You estimate 29--48\% improvement but never actually retrained.''
\item
  \textbf{Defense}: A single retraining run (4--8 hours compute) can plug this gap---this is currently the most fatal gap
\end{itemize}
