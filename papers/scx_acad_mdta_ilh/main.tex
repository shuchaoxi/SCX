\section{ACAD基础设定 / ACAD Foundations}
\label{sec:acad-found}

\subsection{Core问题 / Core Problem}

\begin{bilingual}{Core问题 / Core Question}
如果两个审计员$A$和$B$的参考数据通过某个秘密约束关联，攻击者在不知道约束的情况下篡改一方的数据，能否以高概率检测到？
\textit{If two auditors $A$ and $B$ have reference data linked by a secret constraint, can tampering with one side be detected with high probability by an attacker ignorant of the constraint?}
\end{bilingual}

这是一个\textbf{信息论安全}问题——与量子密钥分发(QKD)共享直觉但不共享数学。ACAD将纠缠类比修正为诚实的信息论框架：不需要量子力学的任何概念。

\textit{This is an information-theoretic security problem --- sharing intuition with QKD but not its mathematics. ACAD corrects the entanglement analogy into an honest information-theoretic framework requiring no quantum mechanics.}

\subsection{基本定义 / Basic Definitions}

\begin{definition}[审计对 / Audit Pair]
\label{def:audit-pair}
一个审计对是一个三元组$(\cA, \cB, \cC)$，其中：
\begin{itemize}
    \item $\cA \subseteq \cX^n$：审计员$A$的参考数据集空间（$n$个数据点）
    \item $\cB \subseteq \cX^n$：审计员$B$的参考数据集空间
    \item $\cC \subseteq \{C: \cX \times \cX \to \{0,1\}\}$：约束函数空间
\end{itemize}
\end{definition}

\begin{definition}[秘密配对 / Secret Pairing]
\label{def:secret-pairing}
秘密配对是一个双射$\phi: [n] \to [n]$和一个约束函数$C \in \cC$，满足对审计员双方的部分可知性：
\begin{itemize}
    \item 审计员$A$知道$\{x_i^A\}_{i=1}^n$但不知道$\phi$和$\{x_j^B\}_{j=1}^n$
    \item 审计员$B$知道$\{x_j^B\}_{j=1}^n$和$\phi$，但不知道$\{x_i^A\}_{i=1}^n$（在篡改检测阶段之前）
    \item 约束满足：$\forall i \in [n],\; C(x_i^A, x_{\phi(i)}^B) = 1$
\end{itemize}
\end{definition}

\begin{definition}[攻击者模型 / Adversary Model]
\label{def:adversary}
攻击者$\mathcal{E}$具有以下能力与限制：
\begin{itemize}
    \item \textbf{能力：} 可以修改$D_A = \{x_i^A\}_{i=1}^n$中的任意子集为$\tilde{D}_A = \{\tilde{x}_i^A\}_{i=1}^n$
    \item \textbf{限制：} 不知道$\phi$、$D_B$、$C$的具体形式
    \item \textbf{知识：} 知道$\cX$、$n$、以及$D_A$的原始内容
\end{itemize}
\end{definition}

\begin{definition}[篡改检测协议 / Tampering Detection Protocol]
\label{def:protocol}
\begin{enumerate}[leftmargin=*, label=\textbf{步骤 \arabic*}.]
    \item \textbf{初始化：} 审计员$B$拥有$D_B$和$\phi$。安全的带外信道用于最终比对。
    \item \textbf{篡改：} 攻击者将$D_A$修改为$\tilde{D}_A$。
    \item \textbf{挑战：} $B$随机选择$k$个索引$i_1, \ldots, i_k \subset [n]$（均匀无放回抽样）。
    \item \textbf{请求：} $B$向$A$请求$\tilde{x}_{i_1}^A, \ldots, \tilde{x}_{i_k}^A$。
    \item \textbf{验证：} $B$计算检查通过率：
    \begin{equation}
        T_k = \frac{1}{k} \sum_{j=1}^{k} C(\tilde{x}_{i_j}^A, x_{\phi(i_j)}^B)
    \end{equation}
    \item \textbf{决策：} 如果$T_k < \tau$（预设阈值），则标记为``检测到篡改''。
\end{enumerate}
\end{definition}

\section{ACAD定理体系 / ACAD Theorem System}
\label{sec:acad-theorems}

\subsection{定理1：检测概率下界（修正版）}

\begin{theorem}[ACAD检测概率下界 / ACAD Detection Probability Lower Bound]
\label{thm:acad-detection}
设攻击者修改了$m$个数据点（$1 \leq m \leq n$），审计员$B$无放回抽取$k$个样本进行验证。假设约束函数$C$是$\varepsilon$-鲁棒的：对于任何未修改的对，$P(C(x_i^A, x_{\phi(i)}^B) = 1) \geq 1 - \varepsilon$（其中$\varepsilon \geq 0$是自然噪声）。假设攻击者不知道$\phi$且$\phi$是均匀随机双射。则：

\begin{equation}
\boxed{P(\text{检测到篡改}) \geq 1 - \exp\left(-\frac{2k \cdot \alpha^2 \cdot (1 - \varepsilon - \gamma)^2}{1 - (k-1)/n}\right)}
\end{equation}

其中$\alpha = m/n$是篡改比例，$\gamma$是攻击者在不知道$\phi$的情况下能成功伪造约束的最大概率，$\tau$是检测阈值。

\textit{Let the attacker modify $m$ data points. Auditor $B$ draws $k$ samples without replacement. Assume constraint $C$ is $\varepsilon$-robust and the attacker does not know the uniformly random bijection $\phi$. Then the detection probability satisfies the above bound.}
\end{theorem}

\begin{proof}
设$S$为被篡改的索引集合，$|S| = m$。$B$抽取$k$个样本（无放回）。对于均匀随机双射$\phi$，攻击者无法预测哪些$i$被$\phi$映射到哪些$j$。

设$p_1$为篡改条件下单次检查通过的概率：
\begin{equation}
    p_1 = \left(1 - \frac{m}{n}\right)(1 - \varepsilon) + \frac{m}{n} \cdot \gamma
\end{equation}
其中第一项来自未篡改点（自然通过率$1-\varepsilon$），第二项来自篡改点（攻击者最多以$\gamma$的概率通过）。

设$p_0 = 1 - \varepsilon$（无篡改条件下的通过率）。检测条件为$T_k < \tau$。选择$\tau = \frac{p_0 + p_1}{2}$为中间点，则：
\begin{equation}
    \tau - p_1 = \frac{p_0 - p_1}{2} = \frac{m}{n} \cdot \frac{1 - \varepsilon - \gamma}{2} = \frac{\alpha(1 - \varepsilon - \gamma)}{2}
\end{equation}

使用Serfling不awaiting式（无放回抽样的集中不awaiting式）替代原始Version中的Hoeffding不awaiting式：
\begin{equation}
    P(T_k \geq \tau) = P(T_k - p_1 \geq \tau - p_1) \leq \exp\left(-\frac{2k(\tau - p_1)^2}{1 - (k-1)/n}\right)
\end{equation}

代入$\tau - p_1$：
\begin{align}
    P(\text{未检测到}) &\leq \exp\left(-\frac{2k \cdot (\alpha(1 - \varepsilon - \gamma)/2)^2}{1 - (k-1)/n}\right) \\
    &= \exp\left(-\frac{k \cdot \alpha^2 \cdot (1 - \varepsilon - \gamma)^2}{2(1 - (k-1)/n)}\right)
\end{align}

因此$P(\text{检测到}) \geq 1 - \exp\left(-\frac{k \alpha^2 (1 - \varepsilon - \gamma)^2}{2(1 - (k-1)/n)}\right)$。

$\square$ \textbf{注：} 原始Version（定理6.1 in Round 3）使用了独立Bernoulli近似的Hoeffding不awaiting式。本修正版使用Serfling不awaiting式处理无放回抽样的相关性。当$k \ll n$时，修正因子$1 - (k-1)/n \approx 1$，两Versionawaiting价。
\end{proof}

\subsection{定理2：统计安全界}

\begin{theorem}[ACAD统计安全界 / ACAD Statistical Security Bound]
\label{thm:acad-security}
如果攻击者不知道$\phi$和$D_B$，且$|D_B| = n$足够大，则对于任何满足知识限制的攻击策略$\mathcal{E}$，存在常数$c > 0$使得：

\begin{equation}
\boxed{\inf_{\mathcal{E}} P(\text{检测到篡改}) \geq 1 - \exp\left(-c \cdot k \cdot \alpha^2\right)}
\end{equation}

其中$c = \frac{(1 - \varepsilon - \gamma)^2}{2(1 - (k-1)/n)}$，下确界取遍所有满足知识限制的攻击策略。

\textit{If the attacker does not know $\phi$ and $D_B$, then for any attack strategy $\mathcal{E}$ satisfying knowledge constraints, the detection probability is bounded below by the above exponential bound.}
\end{theorem}

\begin{proof}
攻击者的信息Status由$\sigma(\tilde{D}_A, D_A)$（攻击者知道的$\sigma$-代数）决定。$\phi$在此$\sigma$-代数下是不可测的（攻击者不知道配对）。

由信息论数据处理不awaiting式，攻击者对$\phi$的估计误差下界为：
\begin{equation}
    H(\phi \mid \tilde{D}_A, D_A) \geq \log(n!) - I(\phi; D_A)
\end{equation}
其中$H$是Entropy，$I$是Mutual Information。由于$D_A$由独立于$\phi$的过程生成（在初始化阶段），$I(\phi; D_A) = 0$。因此$H(\phi \mid \tilde{D}_A, D_A) = \log(n!)$——攻击者对$\phi$完全无知。

在此无知条件下，攻击者能成功伪造约束的概率受限于随机猜测的水平，从而导出定理\ref{thm:acad-detection}中的指数界。

$\square$ \textbf{重要修正：} 原始Version（定理6.2 in Round 3）声称``信息论安全''。本修正版降级为``统计安全''（statistical security），因为：(1) 约束$C$可能有结构可被利用；(2) 攻击者可通过修改大量数据点实现部分成功；(3) 检查是抽样的而非穷举的。ACAD不达到一次一密级别的完美安全。
\end{proof}

\subsection{定理3：样本复杂度}

\begin{theorem}[样本复杂度 / Sample Complexity]
\label{thm:acad-sample}
为实现检测概率至少$1 - \delta$（给定显著性水平），所需的最小样本量$k^*$满足：

\begin{equation}
\boxed{k^* \geq \frac{2 \cdot (1 - (k^*-1)/n)}{\alpha^2 (1 - \varepsilon - \gamma)^2} \cdot \log\frac{1}{\delta}}
\end{equation}

对于固定篡改比例$\alpha$且$k \ll n$的情况，渐进界为：
\begin{equation}
k^* = O\left(\frac{1}{\alpha^2} \cdot \log\frac{1}{\delta}\right)
\end{equation}

\textit{To achieve detection probability at least $1-\delta$, the minimum required sample size satisfies the above bound. For fixed $\alpha$ with $k \ll n$, the asymptotic bound is $O(\alpha^{-2} \log \delta^{-1})$.}
\end{theorem}

\begin{proof}
从定理\ref{thm:acad-detection}的界出发，设$P(\text{未检测到}) \leq \delta$：
\begin{equation}
    \exp\left(-\frac{k \alpha^2 (1 - \varepsilon - \gamma)^2}{2(1 - (k-1)/n)}\right) \leq \delta
\end{equation}
取对数并整理即得。当$k \ll n$时，$1 - (k-1)/n \approx 1$，得到简化形式。
\end{proof}

\subsection{推论：实际参数下的可行性}

\begin{corollary}[实际参数 / Practical Parameters]
\label{cor:acad-practical}
若$\alpha = 0.1$（攻击者修改10\%的数据），$\varepsilon = 0.05$（5\%自然噪声），$\gamma = 0$（攻击者对约束完全无知），$\delta = 0.01$（99\%检测置信度），且$n = 10^4$，则$k^* \geq 2 \cdot 1 \cdot 100 / (1 \cdot 0.9025) \cdot 4.605 \approx 1020$。只需检查约10\%的数据即可高概率检测篡改。
\end{corollary}

\subsection{定理4：审计设置依赖性检测}

\begin{theorem}[ACAD审计规避检测 / ACAD Audit Evasion Detection]
\label{thm:acad-mm}
设专家系统$E$在审计设置$S_1$和$S_2$下产生预测分布$p_1$和$p_2$。定义审计设置依赖性度量：
\begin{equation}
    \Delta(S_1, S_2) = \MMD(p_1, p_2) = \sup_{f \in \cF} \left| \EE_{x \sim p_1}[f(x)] - \EE_{x \sim p_2}[f(x)] \right|
\end{equation}
其中$\cF$是再生核Hilbert空间(RKHS)的单位球。则在零假设$H_0: p_1 = p_2$（无审计设置依赖）下：
\begin{equation}
\boxed{P\left( \widehat{\MMD}_n > \sqrt{\frac{2K_{\max}}{n}} + \sqrt{\frac{2}{n}\log\frac{1}{\delta}} \right) \leq \delta}
\end{equation}
其中$K_{\max} = \sup_x k(x, x)$是核函数的上界，$\widehat{\MMD}_n$是经验MMD估计。

\textit{Under the null hypothesis of no audit-setting dependence ($p_1 = p_2$), the empirical MMD estimator concentrates around zero with the above exponential tail bound.}
\end{theorem}

\begin{proof}
这是MMD的经典集中不awaiting式。对于有界核$k(\cdot, \cdot) \leq K_{\max}$，经验MMD估计$\widehat{\MMD}_n$满足McDiarmid不awaiting式。具体地，改变单个样本对MMD的影响最多为$2\sqrt{K_{\max}/n}$，因此：
\begin{equation}
    P\left(\widehat{\MMD}_n - \EE[\widehat{\MMD}_n] > t\right) \leq \exp\left(-\frac{nt^2}{2K_{\max}}\right)
\end{equation}
在$H_0$下$\EE[\widehat{\MMD}_n] \leq \sqrt{2K_{\max}/n}$，代入并设$t = \sqrt{\frac{2}{n}\log\frac{1}{\delta}}$即得。
\end{proof}

\subsection{定理5：约束构造方案}

\begin{theorem}[约束构造的密码学基础 / Cryptographic Constraint Construction]
\label{thm:acad-constraint}
存在两类可构造的约束函数，满足ACAD协议的安全要求：

\begin{enumerate}[leftmargin=*, label=(\alph*)]
    \item \textbf{数据流形配对约束：} 选择一个公开的高密度参考点集合$R = \{r_1, \ldots, r_n\}$，对于每个$r_i$在$\varepsilon$-邻域内采样$x_i^A$和$x_i^B$，约束$C(x_i^A, x_i^B) = 1$当且仅当$d_{\cM}(x_i^A, r_i) < \varepsilon$且$d_{\cM}(x_i^B, r_i) < \varepsilon$。此约束不依赖于专家预测（避免循环性）。
    
    \item \textbf{密码学哈希约束：} 使用哈希链构造：$x_i^B = H(x_i^A \| K)$，其中$H$是密码学哈希函数，$K$是共享密钥。约束$C(x_i^A, x_i^B) = 1$当且仅当$H(x_i^A \| K) = x_i^B$。攻击者在不知道$K$的情况下无法构造满足约束的$\tilde{x}_i^A$。
\end{enumerate}

其中方案(b)提供最强的安全保证——安全性归结为哈希函数的抗原像性（preimage resistance）。
\end{theorem}

\begin{proof}
对于方案(a)，约束不依赖专家预测，因此不存在``审计本身验证专家预测、约束又依赖专家预测''的循环性。攻击者需要知道流形的全局结构（$r_i$的邻域）才能伪造约束——这awaiting价于知道$D_B$的分布，与攻击者知识限制一致。

对于方案(b)，攻击者在不知道$K$的情况下构造$\tilde{x}_i^A$使得$H(\tilde{x}_i^A \| K) = x_i^B$的概率awaiting于对随机预言机的原像攻击成功概率，在标准密码学假设下为$O(2^{-|K|})$，即$\gamma = \text{negl}(|K|)$。
\end{proof}

\subsection{Core问题 / Core Problem}

MDTA将``虫洞''类比修正为诚实的流形学习+拓扑数据分析(TDA)框架。``捷径''（shortcut）替代``虫洞''——因为Situs流形通常是连通的，路径一直存在，只是长/短的区别。更像``峡谷''或``隧道''而非``虫洞''。

\textit{MDTA corrects the ``wormhole'' analogy into an honest manifold learning + TDA framework. ``Shortcut'' replaces ``wormhole'' --- since the Situs manifold is typically connected, paths always exist, only shorter or longer. More like ``canyons'' or ``tunnels'' than ``wormholes.''}

\subsection{基本定义 / Basic Definitions}

\begin{definition}[数据流形 / Data Manifold]
\label{def:data-manifold}
设$\cX \subseteq \RR^D$为数据空间（环境空间）。数据流形是一个三元组$(\cM, g, p)$，其中：
\begin{itemize}
    \item $\cM$是$d$-维光滑流形（$d \ll D$），$\cM \subset \RR^D$是嵌入子流形
    \item $g$是$\cM$上的Riemannian度规（由嵌入诱导的拉回度规或由数据密度构造的Fisher信息度规）
    \item $p: \cM \to \RR_{>0}$是$\cM$上的数据密度函数（光滑，可积）
\end{itemize}

\textbf{诚实声明：} 真实SCX数据可能不满足光滑流形假设。MDTA的统计量作为\textbf{描述性几何特征}仍然可计算——不需要流形假设来``成立''，它是算法输出而非理论前提。替代方案：使用\textbf{分层空间}(stratified space)假设，允许不同区域有不同维数。
\end{definition}

\begin{definition}[密集区 / Cluster]
\label{def:cluster}
数据流形$\cM$上的一个$\rho$-密集区是一个连通开集$\cC \subset \cM$，满足：
\begin{enumerate}[leftmargin=*, label=(\arabic*)]
    \item $\forall x \in \cC, p(x) \geq \rho$（密度下界）
    \item $\cC$是道路连通的
    \item 边界$\partial\cC$处$p(x) = \rho$
\end{enumerate}
\end{definition}

\begin{definition}[流形距离 / Manifold Distance]
\label{def:manifold-dist}
两点$x, y \in \cM$之间的流形距离为：
\begin{equation}
    d_{\cM}(x, y) = \inf_{\gamma: [0,1] \to \cM, \gamma(0)=x, \gamma(1)=y} \int_0^1 \sqrt{g_{\gamma(t)}(\dot{\gamma}(t), \dot{\gamma}(t))} \, dt
\end{equation}
两个密集区$\cC_i, \cC_j$之间的流形距离：
\begin{equation}
    d_{\cM}(\cC_i, \cC_j) = \inf_{x \in \cC_i, y \in \cC_j} d_{\cM}(x, y)
\end{equation}
\end{definition}

\begin{definition}[捷径比率 / Shortcut Ratio]
\label{def:shortcut-ratio}
对于两个密集区$\cC_i, \cC_j$，定义捷径比率：
\begin{equation}
    \boxed{r(\cC_i, \cC_j) = \frac{d_{\cM}(\cC_i, \cC_j)}{\|\mu_i - \mu_j\|}}
\end{equation}
其中$\mu_i = \frac{1}{\vol(\cC_i)} \int_{\cC_i} x \, dx$是$\cC_i$的质心。
\end{definition}

\begin{definition}[流形捷径 / Manifold Shortcut]
\label{def:shortcut}
当$r(\cC_i, \cC_j) < \tau$（$\tau \in (0, 1)$为预设阈值）时，称$(\cC_i, \cC_j)$构成一个$\tau$-流形捷径。捷径上的最小密度路径为：
\begin{equation}
    \gamma^*_{ij} = \arg\min_{\gamma: \cC_i \to \cC_j} \int_\gamma ds
\end{equation}
沿$\gamma^*$的密度最小值定义捷径的\textbf{喉}(throat)：
\begin{equation}
    x_{\text{throat}} = \arg\min_{x \in \gamma^*_{ij}} p(x)
\end{equation}
\end{definition}

\section{MDTA定理体系 / MDTA Theorem System}
\label{sec:mdta-theorems}

\subsection{定理1：捷径密度分布的变分特征}

\begin{theorem}[捷径密度分布的变分特征 / Variational Characterization of Shortcut Density]
\label{thm:mdta-variational}
设$\gamma^*$是连接$\cC_i$和$\cC_j$的最小流形测地线。沿$\gamma^*$的密度函数$p(\gamma^*(t))$满足：对于捷径（$r < 1$），存在唯一的极小值点$t^* \in (0, 1)$使得：
\begin{equation}
\boxed{p(\gamma^*(t^*)) = \min_{t \in [0,1]} p(\gamma^*(t)) < \min(\rho_i, \rho_j)}
\end{equation}
其中$\rho_i, \rho_j$是$\cC_i, \cC_j$的密度下界。此外，若$\gamma^*$的曲率满足$|\ddot{\gamma}^*(t^*)| > 0$，则：
\begin{equation}
p(\gamma^*(t^*)) \leq \frac{\rho_i + \rho_j}{2} - \frac{1}{8} |\ddot{\gamma}^*(t^*)| \cdot (\rho_i - \rho_j)^2 + O((\rho_i - \rho_j)^3)
\end{equation}

\textit{The density function along the minimal geodesic connecting two shortcut clusters has a unique interior minimum lower than both cluster density bounds, with the above curvature-dependent bound.}
\end{theorem}

\begin{proof}
由捷径定义$r < 1$，流形路径在环境空间中``绕路''，意味着流形在$\gamma^*$附近有显著曲率或折叠。密度函数沿测地线的二阶Taylor展开为：
\begin{equation}
    p(\gamma^*(t)) = p(\gamma^*(t^*)) + \frac{1}{2} p''(\gamma^*(t^*)) (t - t^*)^2 + O((t - t^*)^3)
\end{equation}

在端点$t = 0, 1$处，$p \geq \min(\rho_i, \rho_j)$。设端点$t=0$处密度为$\rho_i + \delta_i$（$\delta_i \geq 0$），$t=1$处为$\rho_j + \delta_j$。通过匹配端点条件的二阶展开，求解$t^*$和$p(\gamma^*(t^*))$：
\begin{align}
    p(\gamma^*(0)) &= p(t^*) + \frac{1}{2}p''(t^*) \cdot (t^*)^2 = \rho_i + \delta_i \\
    p(\gamma^*(1)) &= p(t^*) + \frac{1}{2}p''(t^*) \cdot (1-t^*)^2 = \rho_j + \delta_j
\end{align}

在曲率存在的条件下，$p''(t^*)$与$|\ddot{\gamma}^*(t^*)|$正相关（密度在曲率大的区域变化快）。联立求解并取$\delta_i = \delta_j = 0$的最坏情况，得到所述界。

$\square$ \textbf{注：} 定理依赖于流形曲率的小性假设以保证Taylor展开的收敛。在高曲率区域（这正是捷径的特征），需要更高阶展开或非参数界。这是一个已知的技术限制。
\end{proof}

\subsection{定理2：捷径存在性条件}

\begin{theorem}[捷径存在性条件 / Shortcut Existence Condition]
\label{thm:mdta-existence}
两个$\rho$-密集区$\cC_i, \cC_j$之间存在$\tau$-捷径的充分条件是：
\begin{equation}
\boxed{\exists \text{ 连接 } \cC_i, \cC_j \text{ 的道路 } \gamma \text{ 使得 } \frac{\int_\gamma ds}{\|\mu_i - \mu_j\|} < \tau}
\end{equation}

awaiting价地，如果$\cM$的reach（到达半径）$\tau_{\cM}$满足：
\begin{equation}
    \tau_{\cM} < \frac{\|\mu_i - \mu_j\|}{2} \cdot \sqrt{\frac{1}{\tau^2} - 1}
\end{equation}
则捷径存在。

\textit{A sufficient condition for a $\tau$-shortcut between clusters is the existence of a path whose length is less than $\tau$ times the ambient centroid distance. Equivalently, if the manifold's reach is sufficiently small relative to the inter-centroid distance, a shortcut exists.}
\end{theorem}

\begin{proof}
由流形的reach定义（从$\cM$到其中轴的最大距离）。reach越小意味着流形折叠越严重。当流形折叠使得沿流形的路径远长于环境空间直线距离时，捷径比率$r < 1$。

设$\gamma$为连接两质心的直线段（在环境空间$\RR^D$中）。流形在$\gamma$附近的管状邻域半径受限于reach $\tau_{\cM}$。沿流形的最短路径长度满足：
\begin{equation}
    \int_{\gamma_{\cM}} ds \leq \|\mu_i - \mu_j\| + 2\tau_{\cM} \cdot \theta
\end{equation}
其中$\theta$是管状邻域的角宽度。当$\tau_{\cM}$足够小时，存在满足捷径比率条件的路径。
\end{proof}

\subsection{定理3：审计风险分类及其LipschitzConsistency}

\begin{theorem}[审计风险分类 / Audit Risk Classification]
\label{thm:mdta-risk}
对于捷径$(\cC_i, \cC_j, \gamma^*)$，定义审计风险评分(SAR)：
\begin{equation}
    \SAR(\gamma^*) = \Var_\gamma[g] \times \left|\frac{d}{dt} \Cercis(\gamma^*(t))\right| \times \frac{1}{p(x_{\text{throat}})}
\end{equation}

基于SAR评分，捷径可分类为：
\begin{itemize}
    \item \textbf{良性捷径}(Benign)：$\SAR < \theta_1$ $\rightarrow$ 真实数据结构
    \item \textbf{对抗性捷径}(Adversarial)：$\theta_1 \leq \SAR < \theta_2$ $\rightarrow$ 潜在审计盲区
    \item \textbf{噪声走廊}(Noise Corridor)：$\SAR \geq \theta_2$ $\rightarrow$ 不可审计区域
\end{itemize}

分类的Consistency由以下Lipschitz条件保证：若Cercis沿捷径是$\alpha$-Lipschitz的，则：
\begin{equation}
\boxed{|\SAR(\gamma_1^*) - \SAR(\gamma_2^*)| \leq L \cdot d_{\cM}(\gamma_1^*, \gamma_2^*)}
\end{equation}
其中$L = L_V \cdot \alpha \cdot L_p$是组合Lipschitz常数（$L_V$：方差Lipschitz常数，$L_p$：密度倒数的Lipschitz常数）。

\textit{Shortcuts are classified as benign, adversarial, or noise corridors based on SAR. Classification consistency is guaranteed by Lipschitz continuity of the component factors.}
\end{theorem}

\begin{proof}
由各因子的Lipschitz性质：
\begin{align}
    |\Var_{\gamma_1}[g] - \Var_{\gamma_2}[g]| &\leq L_V \cdot d_{\cM}(\gamma_1^*, \gamma_2^*) \\
    \left|\frac{d}{dt}\Cercis(\gamma_1^*(t)) - \frac{d}{dt}\Cercis(\gamma_2^*(t))\right| &\leq \alpha \cdot d_{\cM}(\gamma_1^*, \gamma_2^*) \\
    \left|\frac{1}{p(x_{\text{throat},1})} - \frac{1}{p(x_{\text{throat},2})}\right| &\leq L_p \cdot d_{\cM}(\gamma_1^*, \gamma_2^*)
\end{align}
三者乘积的Lipschitz性质由乘积规则导出，$L = L_V \cdot \alpha \cdot L_p + \text{cross-terms}$。

$\square$ \textbf{诚实地：} SAR的乘积组合方式缺乏理论依据——为什么是乘积？为什么是这三个因子？这是一种ad-hocHeuristic。修正方案：使用SAR多指标向量（见第\ref{sec:mdta-fixes}节）。
\end{proof}

\subsection{定理4：离散捷径检测的复杂度}

\begin{theorem}[离散捷径检测的算法复杂度 / Complexity of Discrete Shortcut Detection]
\label{thm:mdta-complexity}
对于$N$个数据点的集合，使用k-NN图（$k = O(\log N)$）进行流形距离估计，捷径检测算法的时间复杂度为：
\begin{equation}
\boxed{O(N \log N \cdot k + C^2 \cdot N \log N)}
\end{equation}
其中$C$是检测到的密集区数量。

\textit{The shortcut detection algorithm runs in $O(N \log N \cdot k + C^2 \cdot N \log N)$ time for $N$ points and $C$ detected clusters.}
\end{theorem}

\begin{proof}
算法分为两个阶段：
\begin{enumerate}[leftmargin=*, label=(\arabic*)]
    \item \textbf{k-NN图构建：} 使用kd-tree在$O(N \log N \cdot k)$时间内完成
    \item \textbf{所有密集区对的最短路径：} 对于$C$个密集区，共有$O(C^2)$个对。在每个对上运行Dijkstra算法，单次复杂度$O(N \log N)$（使用二叉堆），总计$O(C^2 \cdot N \log N)$
\end{enumerate}
总复杂度为两者之和。
\end{proof}

\subsection{定理5：捷径比率估计的Consistency}

\begin{theorem}[捷径比率估计的Consistency / Consistency of Shortcut Ratio Estimation]
\label{thm:mdta-consistency}
设$\hat{d}_{\cM}$是基于k-NN图的流形距离估计，$\hat{r}$是相应的捷径比率估计。则在$N \to \infty$、$k \to \infty$、$k/N \to 0$的条件下：
\begin{equation}
\boxed{\hat{r} \xrightarrow{P} r \quad \text{即} \quad \forall \varepsilon > 0: \lim_{N \to \infty} P(|\hat{r} - r| > \varepsilon) = 0}
\end{equation}

\textit{The k-NN graph-based shortcut ratio estimate converges in probability to the true ratio as $N \to \infty$ under standard manifold learning conditions.}
\end{theorem}

\begin{proof}
由k-NN图距离的IsomapConsistency（Bernstein et al., 2000; Tenenbaum et al., 2000），在流形$\cM$是紧致且测地线凸的条件下，k-NN图最短路径一致地收敛到流形测地距离：
\begin{equation}
    \hat{d}_{\cM}(x, y) \xrightarrow{P} d_{\cM}(x, y)
\end{equation}

质心$\hat{\mu}_i \xrightarrow{P} \mu_i$由大数定律保证。捷径比率$\hat{r} = \hat{d}_{\cM} / \|\hat{\mu}_i - \hat{\mu}_j\|$是连续函数，由连续映射定理得证。

$\square$ \textbf{重要条件：} Consistency依赖于(1)流形紧致，(2)数据采样均匀或密度有下界，(3)度规有界曲率。在捷径发生的低密度区域，k-NN图连接可能不可靠——这是捷径检测的Core困境。
\end{proof}

其中$\pi_{\ell, \ell-1}$由$G_{\ell-1} \subset G_\ell$诱导的轨道空间自然映射。

\textit{The invariant space at layer $\ell$ is the orbit space $\cI_\ell = \cG / G_\ell$, with natural projections between layers induced by the group inclusion hierarchy.}

\begin{proof}

\subsection{定理2：Cercis的平移不变性}

\begin{theorem}[Cercis的平移不变性 / Translation Invariance of Cercis]
\label{thm:ilh-cercis-translation}
设Cercis Score定义为差异向量集合的函数：
\begin{equation}
    \Cercis(\Gamma) = f\left(\{d_{mn}\}_{m<n}\right), \quad d_{mn} = \Gamma_m - \Gamma_n \in \RR^d
\end{equation}
则Cercis在对角线平移$G_1 = (\RR^d, +)^M_{\text{diag}}$（作用于$\Gamma \mapsto \Gamma + \mathbf{1}_M \otimes c^T$，$c \in \RR^d$）下不变：

\boxed{\Cercis(\Gamma + \mathbf{1}_M \otimes c^T) = \Cercis(\Gamma)}
\end{equation}

\textit{Cercis Score is invariant under diagonal translation of all expert predictions by the same vector.}
\end{theorem}

\begin{proof}
$d_{mn}' = (\Gamma_m + c) - (\Gamma_n + c) = \Gamma_m - \Gamma_n = d_{mn}$，因此差异向量不变，Cercis（作为差异向量的函数）不变。
\end{proof}

\subsection{定理3：平移不变量的完备性}

\begin{theorem}[平移不变量的完备性 / Completeness of Translation Invariants]
\label{thm:ilh-completeness}
差异向量集合$\{d_{mn}\}_{m<n}$构成平移群作用的\textbf{完备不变量集合}——即两个Configuration$\Gamma, \Gamma'$在平移下awaiting价当且仅当它们的所有差异向量相awaiting：
\begin{equation}
\boxed{\Gamma' = \Gamma + \mathbf{1}_M \otimes c^T \text{ 对某 } c \in \RR^d \iff d_{mn}' = d_{mn} \; \forall m,n}
\end{equation}

\textit{The set of difference vectors forms a complete set of invariants for the translation group action.}
\end{theorem}

\begin{proof}
($\Rightarrow$) 若$\Gamma' = \Gamma + \mathbf{1}_M \otimes c^T$，则$d_{mn}' = d_{mn}$平凡成立。

($\Leftarrow$) 若$d_{mn}' = d_{mn}$对所有$m,n$成立，固定$m=1$，则$\Gamma_n' = \Gamma_n + (\Gamma_1' - \Gamma_1)$对所有$n$。设$c = \Gamma_1' - \Gamma_1$，则$\Gamma' = \Gamma + \mathbf{1}_M \otimes c^T$。
\end{proof}
